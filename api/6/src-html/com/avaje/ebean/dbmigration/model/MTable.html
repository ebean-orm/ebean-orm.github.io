023/**
024 * Holds the logical model for a given Table and everything associated to it.
025 * <p>
026 * This effectively represents a table, its columns and all associated
027 * constraints, foreign keys and indexes.
028 * </p>
029 * <p>
030 * Migrations can be applied to this such that it represents the state
031 * of a given table after various migrations have been applied.
032 * </p>
033 * <p>
034 * This table model can also be derived from the EbeanServer bean descriptor
035 * and associated properties.
036 * </p>
037 */
038public class MTable {

110  /**
111   * Construct for migration.
112   */
113  public MTable(CreateTable createTable) {
114    this.name = createTable.getName();
115    this.pkName = createTable.getPkName();
116    this.comment = createTable.getComment();
117    this.tablespace = createTable.getTablespace();
118    this.indexTablespace = createTable.getIndexTablespace();
119    this.withHistory = Boolean.TRUE.equals(createTable.isWithHistory());
120    this.sequenceName = createTable.getSequenceName();
121    this.sequenceInitial = toInt(createTable.getSequenceInitial());
122    this.sequenceAllocate = toInt(createTable.getSequenceAllocate());
123    List<Column> cols = createTable.getColumn();
124    for (Column column : cols) {
125      addColumn(column);
126    }
127  }

146  /**
147   * Return the CreateTable migration for this table.
148   */
149  public CreateTable createTable() {
150
151    CreateTable createTable = new CreateTable();
152    createTable.setName(name);
153    createTable.setPkName(pkName);
154    createTable.setComment(comment);
155    createTable.setTablespace(tablespace);
156    createTable.setIndexTablespace(indexTablespace);
157    createTable.setSequenceName(sequenceName);
158    createTable.setSequenceInitial(toBigInteger(sequenceInitial));
159    createTable.setSequenceAllocate(toBigInteger(sequenceAllocate));
160    createTable.setIdentityType(identityType);
161    if (withHistory) {
162      createTable.setWithHistory(Boolean.TRUE);
163    }
164
165    for (MColumn column : this.columns.values()) {
166      createTable.getColumn().add(column.createColumn());
167    }
168
169    for (MCompoundForeignKey compoundKey : compoundKeys) {
170      createTable.getForeignKey().add(compoundKey.createForeignKey());
171    }
172
173    for (MCompoundUniqueConstraint constraint : compoundUniqueConstraints) {
174      UniqueConstraint uq = new UniqueConstraint();
175      uq.setName(constraint.getName());
176      String[] columns = constraint.getColumns();
177      StringBuilder sb = new StringBuilder();
178      for (int i = 0; i < columns.length; i++) {
179        if (i > 0) {
180          sb.append(",");
181        }
182        sb.append(columns[i]);
183      }
184      uq.setColumnNames(sb.toString());
185      createTable.getUniqueConstraint().add(uq);
186    }
187
188    return createTable;
189  }

191  /**
192   * Compare to another version of the same table to perform a diff.
193   */
194  public void compare(ModelDiff modelDiff, MTable newTable) {
195
196    if (withHistory != newTable.withHistory) {
197      if (withHistory) {
198        DropHistoryTable dropHistoryTable = new DropHistoryTable();
199        dropHistoryTable.setBaseTable(name);
200        modelDiff.addDropHistoryTable(dropHistoryTable);
201
202      } else {
203        AddHistoryTable addHistoryTable = new AddHistoryTable();
204        addHistoryTable.setBaseTable(name);
205        modelDiff.addAddHistoryTable(addHistoryTable);
206      }
207    }
208
209    addColumn = null;
210
211    Set<String> mappedColumns = new LinkedHashSet<String>();
212
213    Collection<MColumn> newColumns = newTable.getColumns().values();
214    for (MColumn newColumn : newColumns) {
215      MColumn localColumn = columns.get(newColumn.getName());
216      if (localColumn == null) {
217        diffNewColumn(newColumn);
218      } else {
219        // note that if there are alter column changes in here then
220        // the table withHistory is taken into account
221        localColumn.compare(modelDiff, this, newColumn);
222        mappedColumns.add(newColumn.getName());
223      }
224    }
225
226    Collection<MColumn> existingColumns = columns.values();
227    for (MColumn existingColumn : existingColumns) {
228      if (!mappedColumns.contains(existingColumn.getName())) {
229        diffDropColumn(modelDiff, existingColumn);
230      }
231    }
232
233    if (addColumn != null) {
234      modelDiff.addAddColumn(addColumn);
235    }
236  }

238  /**
239   * Apply AddColumn migration.
240   */
241  public void apply(AddColumn addColumn) {
242    checkTableName(addColumn.getTableName());
243    for (Column column : addColumn.getColumn()) {
244      addColumn(column);
245    }
246  }
247
248  /**
249   * Apply AddColumn migration.
250   */
251  public void apply(AlterColumn alterColumn) {
252    checkTableName(alterColumn.getTableName());
253    String columnName = alterColumn.getColumnName();
254    MColumn existingColumn = columns.get(columnName);
255    if (existingColumn == null) {
256      throw new IllegalStateException("Column [" + columnName + "] does not exist for AlterColumn change?");
257    }
258    existingColumn.apply(alterColumn);
259  }
260
261  /**
262   * Apply DropColumn migration.
263   */
264  public void apply(DropColumn dropColumn) {
265    checkTableName(dropColumn.getTableName());
266    columns.remove(dropColumn.getColumnName());
267  }

365  /**
366   * Return the list of columns that make the primary key.
367   */
368  public List<MColumn> primaryKeyColumns() {
369    List<MColumn> pk = new ArrayList<MColumn>(3);
370    for (MColumn column : columns.values()) {
371      if (column.isPrimaryKey()) {
372        pk.add(column);
373      }
374    }
375    return pk;

378  private void checkTableName(String tableName) {
379    if (!name.equals(tableName)) {
380      throw new IllegalArgumentException("addColumn tableName [" + tableName + "] does not match [" + name + "]");
381    }
382  }

389  }

398  /**
399   * Add a compound unique constraint.

401  public void addCompoundUniqueConstraint(String[] columns, boolean oneToOne, String constraintName) {
402    compoundUniqueConstraints.add(new MCompoundUniqueConstraint(columns, oneToOne, constraintName));
403  }

405  /**
406   * Add a compound unique constraint.
407   */
408  public void addCompoundUniqueConstraint(List<MColumn> columns, boolean oneToOne, String constraintName) {
409    String[] cols = new String[columns.size()];
410    for (int i = 0; i < columns.size(); i++) {
411      cols[i] = columns.get(i).getName();
412    }
413    addCompoundUniqueConstraint(cols, oneToOne, constraintName);
414  }

420    compoundKeys.add(compoundKey);
421  }

423  /**
424   * Add a column checking if it already exists and if so return the existing column.
425   * Sometimes the case for a primaryKey that is also a foreign key.
426   */
427  public MColumn addColumn(String dbCol, String columnDefn, boolean notnull) {
428
429    MColumn existingColumn = columns.get(dbCol);
430    if (existingColumn != null) {
431      if (notnull) {
432        existingColumn.setNotnull(true);
433      }
434      return existingColumn;
435    }
436
437    MColumn newCol = new MColumn(dbCol, columnDefn, notnull);
438    addColumn(newCol);
439    return newCol;
440  }
441
442  /**
443   * Add a 'new column' to the AddColumn migration object.
444   */
445  private void diffNewColumn(MColumn newColumn) {
446
447    if (addColumn == null) {
448      addColumn = new AddColumn();
449      addColumn.setTableName(name);
450      if (withHistory) {
451        // These addColumns need to occur on the history
452        // table as well as the base table
453        addColumn.setWithHistory(Boolean.TRUE);
454      }
455    }
456
457    addColumn.getColumn().add(newColumn.createColumn());
458  }
459
460  /**
461   * Add a 'drop column' to the diff.
462   */
463  private void diffDropColumn(ModelDiff modelDiff, MColumn existingColumn) {
464
465    DropColumn dropColumn = new DropColumn();
466    dropColumn.setTableName(name);
467    dropColumn.setColumnName(existingColumn.getName());
468    if (withHistory) {
469      // These dropColumns should occur on the history
470      // table as well as the base table
471      dropColumn.setWithHistory(Boolean.TRUE);
472    }
473
474    modelDiff.addDropColumn(dropColumn);
475  }
476
477  private int toInt(BigInteger value) {
478    return (value == null) ? 0 : value.intValue();
479  }
480
481  private BigInteger toBigInteger(int value) {
482    return (value == 0) ? null : BigInteger.valueOf(value);
483  }
