001package com.avaje.ebean;
002
003/**
004 * Administrative control of Autofetch during runtime.
005 */
006public interface AdminAutofetch {
007
008  /**
009   * Return true if profiling is enabled.
010   */
011  public boolean isProfiling();
012
013  /**
014   * Set to true to enable profiling.
015   */
016  public void setProfiling(boolean enable);
017
018  /**
019   * Return true if autoFetch automatic query tuning is enabled.
020   */
021  public boolean isQueryTuning();
022
023  /**
024   * Set to true to enable autoFetch automatic query tuning.
025   */
026  public void setQueryTuning(boolean enable);
027
028  /**
029   * Returns the rate which profiling is collected. This is an int between 0 and
030   * 100.
031   */
032  public double getProfilingRate();
033
034  /**
035   * Set the rate at which profiling is collected after the base.
036   * 
037   * @param rate
038   *          a int between 0 and 100.
039   */
040  public void setProfilingRate(double rate);
041
042  /**
043   * Return the number of queries profiled after which profiling is collected at
044   * a percentage rate.
045   */
046  public int getProfilingBase();
047
048  /**
049   * Set a base number of queries to profile per query point.
050   * <p>
051   * After this amount of profiling has been obtained profiling is collected at
052   * the Profiling Percentage rate.
053   * </p>
054   */
055  public void setProfilingBase(int profilingBase);
056
057  /**
058   * Return the minimum number of queries profiled before autoFetch will start
059   * automatically tuning the queries.
060   * <p>
061   * This could be one which means start autoFetch tuning after the first
062   * profiling information is collected.
063   * </p>
064   */
065  public int getProfilingMin();
066
067  /**
068   * Set the minimum number of queries profiled per query point before autoFetch
069   * will automatically tune the queries.
070   * <p>
071   * Increasing this number will mean more profiling is collected before
072   * autoFetch starts tuning the query.
073   * </p>
074   */
075  public void setProfilingMin(int autoFetchMinThreshold);
076
077  /**
078   * Fire a garbage collection (hint to the JVM). Assuming garbage collection
079   * fires this will gather the usage profiling information.
080   */
081  public String collectUsageViaGC();
082
083  /**
084   * This will take the current profiling information and update the "tuned
085   * query detail".
086   * <p>
087   * This is done periodically and can also be manually invoked.
088   * </p>
089   * 
090   * @return a summary of the updates that occurred
091   */
092  public String updateTunedQueryInfo();
093
094  /**
095   * Clear all the tuned query info.
096   * <p>
097   * Should only need do this for testing and playing around.
098   * </p>
099   * 
100   * @return the amount of tuned query information cleared.
101   */
102  public int clearTunedQueryInfo();
103
104  /**
105   * Clear all the profiling information.
106   * <p>
107   * This means the profiling information will need to be re-gathered.
108   * </p>
109   * <p>
110   * Should only need do this for testing and playing around.
111   * </p>
112   * 
113   * @return the amount of profiled information cleared.
114   */
115  public int clearProfilingInfo();
116
117  /**
118   * Clear the query execution statistics.
119   */
120  public void clearQueryStatistics();
121
122  /**
123   * Return the number of queries tuned by AutoFetch.
124   */
125  public int getTotalTunedQueryCount();
126
127  /**
128   * Return the size of the TuneQuery map.
129   */
130  public int getTotalTunedQuerySize();
131
132  /**
133   * Return the size of the profile map.
134   */
135  public int getTotalProfileSize();
136
137}
