355  /**
356   * Load the settings from the PropertiesWrapper.
357   */
358  public void loadSettings(PropertiesWrapper properties, String serverName) {
359
360    migrationPath = properties.get("migration.migrationPath", migrationPath);
361    modelPath = properties.get("migration.modelPath", modelPath);
362    applyPrefix = properties.get("migration.applyPrefix", applyPrefix);
363    applySuffix = properties.get("migration.applySuffix", applySuffix);
364    modelSuffix = properties.get("migration.modelSuffix", modelSuffix);
365    includeGeneratedFileComment = properties.getBoolean("migration.includeGeneratedFileComment", includeGeneratedFileComment);
366    generatePendingDrop = properties.get("migration.generatePendingDrop", generatePendingDrop);
367
368    platform = properties.getEnum(Platform.class, "migration.platform", platform);
369
370    generate = properties.getBoolean("migration.generate", generate);
371    version = properties.get("migration.version", version);
372    name = properties.get("migration.name", name);
373
374    runMigration = properties.getBoolean("migration.run", runMigration);
375    metaTable = properties.get("migration.metaTable", metaTable);
376    runPlaceholders = properties.get("migration.placeholders", runPlaceholders);
377
378    String adminUser = properties.get("datasource."+serverName+".username", dbUsername);
379    adminUser = properties.get("datasource."+serverName+".adminusername", adminUser);
380    dbUsername = properties.get("migration.dbusername", adminUser);
381
382    String adminPwd = properties.get("datasource."+serverName+".password", dbPassword);
383    adminPwd = properties.get("datasource."+serverName+".adminpassword", adminPwd);
384    dbPassword = properties.get("migration.dbpassword", adminPwd);
385  }

387  /**
388   * Return true if the migration should be generated.
389   * <p>
390   * It is expected that when an environment variable <code>ddl.migration.enabled</code>
391   * is set to <code>true</code> then the DB migration will generate the migration DDL.
392   * </p>
393   */
394  public boolean isGenerateOnStart() {
395
396    // environment properties take precedence
397    String envGenerate = readEnvironment("ddl.migration.generate");
398    if (envGenerate != null) {
399      return "true".equalsIgnoreCase(envGenerate.trim());
400    }
401    return generate;
402  }

404  /**
405   * Called by EbeanServer on start.
406   *
407   * <p>
408   * If enabled this generates the migration xml and DDL scripts.
409   * </p>
410   */
411  public void generateOnStart(EbeanServer server) {
412
413    if (isGenerateOnStart()) {
414      if (platform == null) {
415        logger.warn("No platform set for migration DDL generation");
416      } else {
417        // generate the migration xml and platform specific DDL
418        DbMigration migration = new DbMigration(server);
419        migration.setPlatform(platform);
420        try {
421          migration.generateMigration();
422        } catch (Exception e) {
423          throw new RuntimeException("Error generating DB migration", e);
424        }
425      }
426    }
427  }

467  /**
468   * Return the system or environment property.
469   */
470  protected String readEnvironment(String key) {
471
472    String val = System.getProperty(key);
473    if (val == null) {
474      val = System.getenv(key);
475    }
476    return val;
477  }

479  /**
480   * Return true if the string is null or empty.
481   */
482  protected boolean isEmpty(String val) {
483    return val == null || val.trim().isEmpty();
484  }

486  /**
487   * Create the MigrationRunner to run migrations if necessary.
488   */
489  public MigrationRunner createRunner(ClassLoader classLoader) {
490
491    MigrationConfig runnerConfig = new MigrationConfig();
492    runnerConfig.setMetaTable(metaTable);
493    runnerConfig.setApplySuffix(applySuffix);
494    runnerConfig.setMigrationPath(migrationPath);
495    runnerConfig.setRunPlaceholderMap(runPlaceholderMap);
496    runnerConfig.setRunPlaceholders(runPlaceholders);
497    runnerConfig.setDbUsername(getDbUsername());
498    runnerConfig.setDbPassword(getDbPassword());
499    runnerConfig.setClassLoader(classLoader);
500    return new MigrationRunner(runnerConfig);
501  }
