001package com.avaje.ebean.event;
002
003import java.util.Map;
004import java.util.Set;
005
006import com.avaje.ebean.EbeanServer;
007import com.avaje.ebean.Transaction;
008import com.avaje.ebean.ValuePair;
009
010/**
011 * Holds the information available for a bean persist (insert, update or
012 * delete).
013 * <p>
014 * This is made available for the BeanPersistControllers.
015 * </p>
016 */
017public interface BeanPersistRequest<T> {
018
019  /**
020   * Return the server processing the request.
021   */
022  public EbeanServer getEbeanServer();
023
024  /**
025   * Return the Transaction associated with this request.
026   */
027  public Transaction getTransaction();
028
029  /**
030   * For an update or delete of a partially populated bean this is the set of
031   * loaded properties and otherwise returns null.
032   */
033  public Set<String> getLoadedProperties();
034
035  /**
036   * For an update this is the set of properties that where updated.
037   */
038  public Set<String> getUpdatedProperties();
039
040  /**
041   * Returns the bean being inserted updated or deleted.
042   */
043  public T getBean();
044
045  /**
046   * Returns a map of the properties that have changed and their new and old values.
047   */
048  public Map<String,ValuePair> getUpdatedValues();
049
050}
