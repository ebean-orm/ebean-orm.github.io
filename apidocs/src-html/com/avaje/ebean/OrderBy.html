001package com.avaje.ebean;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.Arrays;
006import java.util.List;
007
008/**
009 * Represents an Order By for a Query.
010 * <p>
011 * Is a ordered list of OrderBy.Property objects each specifying a property and
012 * whether it is ascending or descending order.
013 * </p>
014 * <p>
015 * Typically you will not construct an OrderBy yourself but use one that exists
016 * on the Query object.
017 * </p>
018 */
019public final class OrderBy<T> implements Serializable {
020
021  private static final long serialVersionUID = 9157089257745730539L;
022
023  private transient Query<T> query;
024
025  private final List<Property> list;
026
027  /**
028   * Create an empty OrderBy with no associated query.
029   */
030  public OrderBy() {
031    this.list = new ArrayList<Property>(2);
032  }
033  
034  private OrderBy(List<Property> list) {
035    this.list = list;
036  }
037
038  /**
039   * Create an orderBy parsing the order by clause.
040   * <p>
041   * The order by clause follows SQL order by clause with comma's between each
042   * property and optionally "asc" or "desc" to represent ascending or
043   * descending order respectively.
044   * </p>
045   */
046  public OrderBy(String orderByClause) {
047    this(null, orderByClause);
048  }
049
050  /**
051   * Construct with a given query and order by clause.
052   */
053  public OrderBy(Query<T> query, String orderByClause) {
054    this.query = query;
055    this.list = new ArrayList<Property>(2);
056    parse(orderByClause);
057  }
058
059  /**
060   * Reverse the ascending/descending order on all the properties.
061   */
062  public void reverse() {
063    for (int i = 0; i < list.size(); i++) {
064      list.get(i).reverse();
065    }
066  }
067
068  /**
069   * Add a property with ascending order to this OrderBy.
070   */
071  public Query<T> asc(String propertyName) {
072
073    list.add(new Property(propertyName, true));
074    return query;
075  }
076
077  /**
078   * Add a property with descending order to this OrderBy.
079   */
080  public Query<T> desc(String propertyName) {
081
082    list.add(new Property(propertyName, false));
083    return query;
084  }
085
086  /**
087   * Return a copy of this OrderBy with the path trimmed.
088   */
089  public OrderBy<T> copyWithTrim(String path) {
090    List<Property> newList = new ArrayList<Property>(list.size());
091    for (int i = 0; i < list.size(); i++) {
092      newList.add(list.get(i).copyWithTrim(path));
093    }
094    return new OrderBy<T>(newList);
095  }
096  
097  /**
098   * Return the properties for this OrderBy.
099   */
100  public List<Property> getProperties() {
101    // not returning an Immutable list at this point
102    return list;
103  }
104
105  /**
106   * Return true if this OrderBy does not have any properties.
107   */
108  public boolean isEmpty() {
109    return list.isEmpty();
110  }
111
112  /**
113   * Return the associated query if there is one.
114   */
115  public Query<T> getQuery() {
116    return query;
117  }
118
119  /**
120   * Associate this OrderBy with a query.
121   */
122  public void setQuery(Query<T> query) {
123    this.query = query;
124  }
125
126  /**
127   * Return a copy of the OrderBy.
128   */
129  public OrderBy<T> copy() {
130
131    OrderBy<T> copy = new OrderBy<T>();
132    for (int i = 0; i < list.size(); i++) {
133      copy.add(list.get(i).copy());
134    }
135    return copy;
136  }
137
138  /**
139   * Add a property to the order by.
140   */
141  public void add(Property p) {
142    list.add(p);
143  }
144
145  public String toString() {
146    return list.toString();
147  }
148
149  /**
150   * Returns the OrderBy in string format.
151   */
152  public String toStringFormat() {
153    if (list.isEmpty()) {
154      return null;
155    }
156    StringBuilder sb = new StringBuilder();
157    for (int i = 0; i < list.size(); i++) {
158      Property property = list.get(i);
159      if (i > 0) {
160        sb.append(", ");
161      }
162      sb.append(property.toStringFormat());
163    }
164    return sb.toString();
165  }
166
167  @Override
168  public boolean equals(Object obj) {
169    if (obj == this) {
170      return true;
171    }
172    if (!(obj instanceof OrderBy<?>)) {
173      return false;
174    }
175    
176    OrderBy<?> e = (OrderBy<?>) obj;
177    return e.list.equals(list);
178  }
179
180  /**
181   * Return a hash value for this OrderBy. This can be to determine logical
182   * equality for OrderBy clauses.
183   */
184  public int hashCode() {
185    return list.hashCode();
186  }
187
188  /**
189   * A property and its ascending descending order.
190   */
191  public static final class Property implements Serializable {
192
193    private static final long serialVersionUID = 1546009780322478077L;
194
195    private String property;
196
197    private boolean ascending;
198
199    public Property(String property, boolean ascending) {
200      this.property = property;
201      this.ascending = ascending;
202    }
203
204    /**
205     * Return a copy of this Property with the path trimmed.
206     */
207    public Property copyWithTrim(String path) {
208      return new Property(property.substring(path.length() + 1), ascending);
209    }
210
211    public int hashCode() {
212      int hc = property.hashCode();
213      hc = hc * 31 + (ascending ? 0 : 1);
214      return hc;
215    }
216    
217    public boolean equals(Object obj) {
218      if (obj == this) {
219        return true;
220      }
221      if (!(obj instanceof Property)) {
222        return false;
223      }
224      
225      Property e = (Property) obj;
226      return e.ascending == ascending 
227          && e.property.equals(property);
228    }
229
230    public String toString() {
231      return toStringFormat();
232    }
233
234    public String toStringFormat() {
235      if (ascending) {
236        return property;
237      } else {
238        return property + " desc";
239      }
240    }
241
242    /**
243     * Reverse the ascending/descending order for this property.
244     */
245    public void reverse() {
246      this.ascending = !ascending;
247    }
248
249    /**
250     * Trim off the pathPrefix.
251     */
252    public void trim(String pathPrefix) {
253      property = property.substring(pathPrefix.length() + 1);
254    }
255
256    /**
257     * Return a copy of this property.
258     */
259    public Property copy() {
260      return new Property(property, ascending);
261    }
262
263    /**
264     * Return the property name.
265     */
266    public String getProperty() {
267      return property;
268    }
269
270    /**
271     * Set the property name.
272     */
273    public void setProperty(String property) {
274      this.property = property;
275    }
276
277    /**
278     * Return true if the order is ascending.
279     */
280    public boolean isAscending() {
281      return ascending;
282    }
283
284    /**
285     * Set to true if the order is ascending.
286     */
287    public void setAscending(boolean ascending) {
288      this.ascending = ascending;
289    }
290
291  }
292
293  private void parse(String orderByClause) {
294
295    if (orderByClause == null) {
296      return;
297    }
298
299    String[] chunks = orderByClause.split(",");
300    for (int i = 0; i < chunks.length; i++) {
301
302      String[] pairs = chunks[i].split(" ");
303      Property p = parseProperty(pairs);
304      if (p != null) {
305        list.add(p);
306      }
307    }
308  }
309
310  private Property parseProperty(String[] pairs) {
311    if (pairs.length == 0) {
312      return null;
313    }
314
315    ArrayList<String> wordList = new ArrayList<String>(pairs.length);
316    for (int i = 0; i < pairs.length; i++) {
317      if (!isEmptyString(pairs[i])) {
318        wordList.add(pairs[i]);
319      }
320    }
321    if (wordList.isEmpty()) {
322      return null;
323    }
324    if (wordList.size() == 1) {
325      return new Property(wordList.get(0), true);
326    }
327    if (wordList.size() == 2) {
328      boolean asc = isAscending(wordList.get(1));
329      return new Property(wordList.get(0), asc);
330    }
331    String m = "Expecting a max of 2 words in [" + Arrays.toString(pairs)
332        + "] but got " + wordList.size();
333    throw new RuntimeException(m);
334  }
335
336  private boolean isAscending(String s) {
337    s = s.toLowerCase();
338    if (s.startsWith("asc")) {
339      return true;
340    }
341    if (s.startsWith("desc")) {
342      return false;
343    }
344    String m = "Expecting [" + s + "] to be asc or desc?";
345    throw new RuntimeException(m);
346  }
347
348  private boolean isEmptyString(String s) {
349    return s == null || s.length() == 0;
350  }
