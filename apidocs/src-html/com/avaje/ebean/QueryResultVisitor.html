001package com.avaje.ebean;
002
003/**
004 * Used to process a query result one bean at a time via a callback to this
005 * visitor.
006 * <p>
007 * If you wish to stop further processing return false from the accept method.
008 * </p>
009 * <p>
010 * Unlike findList() and findSet() using a QueryResultVisitor does not require
011 * all the beans in the query result to be held in memory at once. This makes
012 * QueryResultVisitor useful for processing large queries.
013 * </p>
014 * 
015 * <pre class="code">
016 * 
017 * Query&lt;Customer&gt; query = server.find(Customer.class)
018 *     .fetch(&quot;contacts&quot;, new FetchConfig().query(2))
019 *     .where().gt(&quot;id&quot;, 0)
020 *     .orderBy(&quot;id&quot;)
021 *     .setMaxRows(2);
022 * 
023 * query.findVisit(new QueryResultVisitor&lt;Customer&gt;() {
024 * 
025 *   public boolean accept(Customer customer) {
026 *     // do something with customer
027 *     System.out.println(&quot;-- visit &quot; + customer);
028 *     return true;
029 *   }
030 * });
031 * </pre>
032 * 
033 * @author rbygrave
034 * 
035 * @param <T>
036 *          the type of entity bean being queried.
037 */
038public interface QueryResultVisitor<T> {
039
040  /**
041   * Process the bean and return true if you want to continue processing more
042   * beans. Return false if you want to stop processing further.
043   * 
044   * @param bean
045   *          the entity bean to process
046   * @return true to continue processing or false to stop.
047   */
048  public boolean accept(T bean);
049}
