001package com.avaje.ebean.text.json;
002
003import com.avaje.ebean.text.PathProperties;
004
005/**
006 * Deprecated in favour of just using PathProperties.
007 *
008 * Provides options for customising the JSON write process.
009 * <p>
010 * You can explicitly state which properties to include in the JSON output for
011 * the root level and each path.
012 * </p>
013 * @deprecated
014 */
015public class JsonWriteOptions {
016
017  protected PathProperties pathProperties;
018
019  /**
020   * Parse and return a PathProperties from nested string format like
021   * (a,b,c(d,e),f(g)) where "c" is a path containing "d" and "e" and "f" is a
022   * path containing "g" and the root path contains "a","b","c" and "f".
023   *
024   * @see com.avaje.ebean.text.PathProperties#parse(String)
025   */
026  public static JsonWriteOptions parsePath(String pathProperties) {
027
028    return pathProperties(PathProperties.parse(pathProperties));
029  }
030
031  /**
032   * Construct JsonWriteOptions with the given pathProperties.
033   */
034  public static JsonWriteOptions pathProperties(PathProperties pathProperties) {
035    JsonWriteOptions o = new JsonWriteOptions();
036    o.setPathProperties(pathProperties);
037    return o;
038  }
039
040  /**
041   * Set the Map of properties to include by path.
042   */
043  public void setPathProperties(PathProperties pathProperties) {
044    this.pathProperties = pathProperties;
045  }
046
047  /**
048   * Return the properties to include by path.
049   */
050  public PathProperties getPathProperties() {
051    return pathProperties;
052  }
053
054}
