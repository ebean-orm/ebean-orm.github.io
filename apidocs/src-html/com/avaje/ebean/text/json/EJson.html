001package com.avaje.ebean.text.json;
002
003import java.io.IOException;
004import java.io.Reader;
005import java.io.Writer;
006import java.util.List;
007import java.util.Map;
008
009import com.fasterxml.jackson.core.JsonGenerator;
010import com.fasterxml.jackson.core.JsonParser;
011import com.fasterxml.jackson.core.JsonToken;
012
013/**
014 * Utility that converts between JSON content and simple java Maps/Lists.
015 */
016public class EJson {
017
018  /**
019   * Write the nested Map/List as json.
020   */
021  public static String write(Object object) throws IOException {
022    return EJsonWriter.write(object);
023  }
024  
025  /**
026   * Write the nested Map/List as json to the writer.
027   */
028  public static void write(Object object, Writer writer) throws IOException {
029    EJsonWriter.write(object, writer);
030  }
031  
032  /**
033   * Write the nested Map/List as json to the jsonGenerator.
034   */
035  public static void write(Object object, JsonGenerator jsonGenerator) throws IOException {
036    EJsonWriter.write(object, jsonGenerator);
037  }
038  
039  /**
040   * Parse the json and return as a Map.
041   */
042  public static Map<String,Object> parseObject(String json) throws IOException {
043    return EJsonReader.parseObject(json);
044  }
045  
046  /**
047   * Parse the json and return as a Map taking a reader.
048   */
049  public static Map<String,Object> parseObject(Reader reader) throws IOException {
050    return EJsonReader.parseObject(reader);
051  }
052  
053  /**
054   * Parse the json and return as a Map taking a JsonParser.
055   */
056  public static Map<String,Object> parseObject(JsonParser parser) throws IOException {
057    return EJsonReader.parseObject(parser);
058  }
059
060  /**
061   * Parse the json and return as a Map taking a JsonParser and a starting token.
062   * <p>
063   * Used when the first token is checked to see if the value is null prior to calling this.
064   * </p>
065   */
066  public static Map<String,Object> parseObject(JsonParser parser, JsonToken token) throws IOException {
067    return EJsonReader.parseObject(parser, token);
068  }
069
070  /**
071   * Parse the json and return as a List.
072   * @throws IOException 
073   */
074  public static List<Object> parseList(String json) throws IOException {
075    return EJsonReader.parseList(json);
076  }
077  
078  /**
079   * Parse the json and return as a List taking a Reader.
080   * @throws IOException 
081   */
082  public static List<Object> parseList(Reader reader) throws IOException {
083    return EJsonReader.parseList(reader);
084  }
085  
086  /**
087   * Parse the json and return as a List taking a JsonParser.
088   */
089  public static List<Object> parseList(JsonParser parser) throws IOException {
090    return EJsonReader.parseList(parser);
091  }
092  
093  
094  /**
095   * Parse the json and return as a List or Map.
096   */
097  public static Object parse(String json) throws IOException {
098    return EJsonReader.parse(json);
099  }
100  
101  /**
102   * Parse the json and return as a List or Map.
103   */
104  public static Object parse(Reader reader) throws IOException {
105    return EJsonReader.parse(reader);
106  }
107  
108  /**
109   * Parse the json and return as a List or Map.
110   */
111  public static Object parse(JsonParser parser) throws IOException {
112    return EJsonReader.parse(parser);
113  }
114
115}
