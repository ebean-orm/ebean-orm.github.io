001package com.avaje.ebean.dbmigration.model.visitor;
002
003import com.avaje.ebeaninternal.server.deploy.BeanProperty;
004import com.avaje.ebeaninternal.server.deploy.BeanPropertyAssocMany;
005import com.avaje.ebeaninternal.server.deploy.BeanPropertyAssocOne;
006import com.avaje.ebeaninternal.server.deploy.BeanPropertyCompound;
007
008/**
009 * Used to visit a BeanProperty given the type of bean property it is.
010 */
011public interface BeanPropertyVisitor {
012
013  /**
014   * Completed visiting all the properties on the bean.
015   */
016  void visitEnd();
017
018  /**
019   * Visit a OneToMany or ManyToMany property.
020   */
021  void visitMany(BeanPropertyAssocMany<?> p);
022
023  /**
024   * Visit the imported side of a OneToOne property.
025   */
026  void visitOneImported(BeanPropertyAssocOne<?> p);
027
028  /**
029   * Visit the exported side of a OneToOne property.
030   */
031  void visitOneExported(BeanPropertyAssocOne<?> p);
032
033  /**
034   * Visit an embedded property.
035   */
036  void visitEmbedded(BeanPropertyAssocOne<?> p);
037
038  /**
039   * Visit the scalar property of an embedded bean.
040   */
041  void visitEmbeddedScalar(BeanProperty p, BeanPropertyAssocOne<?> embedded);
042
043  /**
044   * Visit a scalar property.
045   */
046  void visitScalar(BeanProperty p);
047
048  /**
049   * Visit a compound value object.
050   */
051  void visitCompound(BeanPropertyCompound p);
052
053  /**
054   * Visit the scalar value inside a compound value object.
055   */
056  void visitCompoundScalar(BeanPropertyCompound compound, BeanProperty p);
057
058}
