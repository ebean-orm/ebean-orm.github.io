001package com.avaje.ebean.dbmigration.model;
002
003import com.avaje.ebean.config.DbConstraintNaming;
004import com.avaje.ebean.dbmigration.ddlgeneration.DdlHandler;
005import com.avaje.ebean.dbmigration.ddlgeneration.DdlWrite;
006import com.avaje.ebean.dbmigration.ddlgeneration.platform.DefaultConstraintMaxLength;
007import com.avaje.ebean.dbmigration.migration.ChangeSet;
008import com.avaje.ebean.dbmigration.model.build.ModelBuildBeanVisitor;
009import com.avaje.ebean.dbmigration.model.build.ModelBuildContext;
010import com.avaje.ebean.dbmigration.model.visitor.VisitAllUsing;
011import com.avaje.ebeaninternal.api.SpiEbeanServer;
012
013import java.io.IOException;
014import java.util.List;
015
016/**
017 * Reads EbeanServer bean descriptors to build the current model.
018 */
019public class CurrentModel {
020
021  private final SpiEbeanServer server;
022
023  private final DbConstraintNaming constraintNaming;
024
025  private final DbConstraintNaming.MaxLength maxLength;
026
027  private ModelContainer model;
028
029  private ChangeSet changeSet;
030
031  private DdlWrite write;
032
033  /**
034   * Construct with a given EbeanServer instance.
035   */
036  public CurrentModel(SpiEbeanServer server) {
037    this(server, server.getServerConfig().getConstraintNaming());
038  }
039
040  /**
041   * Construct with a given EbeanServer, platformDdl and constraintNaming convention.
042   * <p>
043   *   Note the EbeanServer is just used to read the BeanDescriptors and platformDdl supplies
044   *   the platform specific handling on
045   * </p>
046   */
047  public CurrentModel(SpiEbeanServer server, DbConstraintNaming constraintNaming) {
048    this.server = server;
049    this.constraintNaming = constraintNaming;
050    this.maxLength = maxLength(server, constraintNaming);
051  }
052
053  public CurrentModel(SpiEbeanServer server, DbConstraintNaming constraintNaming, int maxConstraintLength) {
054    this.server = server;
055    this.constraintNaming = constraintNaming;
056    this.maxLength = new DefaultConstraintMaxLength(maxConstraintLength);
057  }
058
059  private DbConstraintNaming.MaxLength maxLength(SpiEbeanServer server, DbConstraintNaming naming) {
060
061    if (naming.getMaxLength() != null) {
062      return naming.getMaxLength();
063    }
064
065    int maxConstraintNameLength = server.getDatabasePlatform().getMaxConstraintNameLength();
066    return new DefaultConstraintMaxLength(maxConstraintNameLength);
067  }
068
069  /**
070   * Return the current model by reading all the bean descriptors and properties.
071   */
072  public ModelContainer read() {
073    if (model == null) {
074      model = new ModelContainer();
075
076      ModelBuildContext context = new ModelBuildContext(model, constraintNaming, maxLength);
077      ModelBuildBeanVisitor visitor = new ModelBuildBeanVisitor(context);
078      VisitAllUsing visit = new VisitAllUsing(visitor, server);
079      visit.visitAllBeans();
080    }
081    return model;
082  }
083
084  public void setChangeSet(ChangeSet changeSet) {
085   this.changeSet = changeSet;
086  }
087
088  /**
089   * Return as a ChangeSet.
090   */
091  public ChangeSet getChangeSet() {
092    read();
093    if (changeSet == null) {
094      changeSet = asChangeSet();
095    }
096    return changeSet;
097  }
098
099  /**
100   * Return the 'Create' DDL.
101   */
102  public String getCreateDdl() throws IOException {
103
104    createDdl();
105
106    StringBuilder ddl = new StringBuilder(2000);
107    ddl.append(write.apply().getBuffer());
108    ddl.append(write.applyForeignKeys().getBuffer());
109    ddl.append(write.applyHistory().getBuffer());
110
111    return ddl.toString();
112  }
113
114  /**
115   * Return the 'Drop' DDL.
116   */
117  public String getDropDdl() throws IOException {
118
119    createDdl();
120
121    StringBuilder ddl = new StringBuilder(2000);
122    ddl.append(write.rollbackForeignKeys().getBuffer());
123    ddl.append(write.rollback().getBuffer());
124
125    return ddl.toString();
126  }
127
128  /**
129   * Create all the DDL based on the changeSet.
130   */
131  private void createDdl() throws IOException {
132
133    if (write == null) {
134      ChangeSet createChangeSet = getChangeSet();
135
136      write = new DdlWrite(new MConfiguration(), model);
137
138      DdlHandler handler = handler();
139      handler.generate(write, createChangeSet);
140    }
141  }
142
143  /**
144   * Return the platform specific DdlHandler (to generate DDL).
145   */
146  private DdlHandler handler() {
147    return server.getDatabasePlatform().createDdlHandler(server.getServerConfig());
148  }
149
150  /**
151   * Convert the model into a ChangeSet.
152   */
153  private ChangeSet asChangeSet() {
154
155    // empty diff so changes will effectively all be create
156    ModelDiff diff = new ModelDiff();
157    diff.compareTo(model);
158
159    List<Object> applyChanges = diff.getApplyChanges();
160
161    // put the changes into a ChangeSet
162    ChangeSet applyChangeSet = new ChangeSet();
163    applyChangeSet.getChangeSetChildren().addAll(applyChanges);
164    return applyChangeSet;
165  }
166
167}
