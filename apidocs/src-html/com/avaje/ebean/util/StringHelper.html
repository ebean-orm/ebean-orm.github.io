016  /**
017   * parses a String of the form name1='value1' name2='value2'. Note that you
018   * can use either single or double quotes for any particular name value pair
019   * and the end quote must match the begin quote.
020   */
021  public static HashMap<String, String> parseNameQuotedValue(String tag) throws RuntimeException {
022
023    if (tag == null || tag.length() < 1) {
024      return null;
025    }
026
027    // make sure that the quotes are matched...
028    // int remainer = countOccurances(tag, ""+quote) % 2;
029    // if (remainer == 1) {
030    // dp("remainder = "+remainer);
031    // throw new StringParsingException("Unmatched quote in "+tag);
032    // }
033
034    // make sure that th last character is not an equals...
035    // (check now so I don't need to check this every time..)
036    if (tag.charAt(tag.length() - 1) == '=') {
037      throw new RuntimeException("missing quoted value at the end of " + tag);
038    }
039
040    HashMap<String, String> map = new HashMap<String, String>();
041    // recursively parse out the name value pairs...
042    return parseNameQuotedValue(map, tag, 0);
043  }

045  /**
046   * recursively parse out name value pairs (where the value is quoted, with
047   * either single or double quotes).
048   */
049  private static HashMap<String, String> parseNameQuotedValue(HashMap<String, String> map,
050      String tag, int pos) throws RuntimeException {
051
052    int equalsPos = tag.indexOf("=", pos);
053    if (equalsPos > -1) {
054      // check for begin quote...
055      char firstQuote = tag.charAt(equalsPos + 1);
056      if (firstQuote != SINGLE_QUOTE && firstQuote != DOUBLE_QUOTE) {
057        throw new RuntimeException("missing begin quote at " + (equalsPos) + "["
058            + tag.charAt(equalsPos + 1) + "] in [" + tag + "]");
059      }
060
061      // check for end quote...
062      int endQuotePos = tag.indexOf(firstQuote, equalsPos + 2);
063      if (endQuotePos == -1) {
064        throw new RuntimeException("missing end quote [" + firstQuote + "] after " + pos
065            + " in [" + tag + "]");
066      }
067
068      // we have a valid name and value...
069      // dp("pos="+pos+" equalsPos="+equalsPos+"
070      // endQuotePos="+endQuotePos);
071      String name = tag.substring(pos, equalsPos);
072      String value = tag.substring(equalsPos + 2, endQuotePos);
073      // dp("name="+name+"; value="+value+";");
074
075      // trim off any whitespace from the front of name...
076      name = trimFront(name, " ");
077      if ((name.indexOf(SINGLE_QUOTE) > -1) || (name.indexOf(DOUBLE_QUOTE) > -1)) {
078        throw new RuntimeException("attribute name contains a quote [" + name + "]");
079      }
080      map.put(name, value);
081
082      return parseNameQuotedValue(map, tag, endQuotePos + 1);
083
084    } else {
085      // no more equals... stop parsing...
086      return map;
087    }
088  }

090  /**
091   * Returns the number of times a particular String occurs in another String.
092   * e.g. count the number of single quotes.
093   */
094  public static int countOccurances(String content, String occurs) {
095    return countOccurances(content, occurs, 0, 0);
096  }

098  private static int countOccurances(String content, String occurs, int pos, int countSoFar) {
099    int equalsPos = content.indexOf(occurs, pos);
100    if (equalsPos > -1) {
101      countSoFar = countSoFar + 1;
102      pos = equalsPos + occurs.length();
103      // dp("countSoFar="+countSoFar+" pos="+pos);
104      return countOccurances(content, occurs, pos, countSoFar);
105    } else {
106      return countSoFar;
107    }
108  }

110  /**
111   * Parses out a list of Name Value pairs that are delimited together. Will
112   * always return a StringMap. If allNameValuePairs is null, or no name values
113   * can be parsed out an empty StringMap is returned.
114   * 
115   * @param allNameValuePairs
116   *          the entire string to be parsed.
117   * @param listDelimiter
118   *          (typically ';') the delimited between the list
119   * @param nameValueSeparator
120   *          (typically '=') the separator between the name and value
121   */
122  public static Map<String, String> delimitedToMap(String allNameValuePairs,
123      String listDelimiter, String nameValueSeparator) {
124
125    HashMap<String, String> params = new HashMap<String, String>();
126    if ((allNameValuePairs == null) || (allNameValuePairs.length() == 0)) {
127      return params;
128    }
129    // trim off any leading listDelimiter...
130    allNameValuePairs = trimFront(allNameValuePairs, listDelimiter);
131    return getKeyValue(params, 0, allNameValuePairs, listDelimiter, nameValueSeparator);
132  }

134  /**
135   * Trims off recurring strings from the front of a string.
136   * 
137   * @param source
138   *          the source string
139   * @param trim
140   *          the string to trim off the front
141   */
142  public static String trimFront(String source, String trim) {
143    if (source == null) {
144      return null;
145    }
146    if (source.indexOf(trim) == 0) {
147      // dp("trim ...");
148      return trimFront(source.substring(trim.length()), trim);
149    } else {
150      return source;
151    }
152  }

161  /**
162   * Recursively pulls out the key value pairs from a raw string.
163   */
164  private static HashMap<String, String> getKeyValue(HashMap<String, String> map, int pos,
165      String allNameValuePairs, String listDelimiter, String nameValueSeparator) {
166
167    if (pos >= allNameValuePairs.length()) {
168      // dp("end as "+pos+" >= "+allNameValuePairs.length() );
169      return map;
170    }
171
172    int equalsPos = allNameValuePairs.indexOf(nameValueSeparator, pos);
173    int delimPos = allNameValuePairs.indexOf(listDelimiter, pos);
174
175    if (delimPos == -1) {
176      delimPos = allNameValuePairs.length();
177    }
178    if (equalsPos == -1) {
179      // dp("no more equals...");
180      return map;
181    }
182    if (delimPos == (equalsPos + 1)) {
183      // dp("Ignoring as nothing between delim and equals...
184      // delim:"+delimPos+" eq:"+equalsPos);
185      return getKeyValue(map, delimPos + 1, allNameValuePairs, listDelimiter,
186          nameValueSeparator);
187    }
188    if (equalsPos > delimPos) {
189      // there is a key without a value?
190      String key = allNameValuePairs.substring(pos, delimPos);
191      key = key.trim();
192      if (key.length() > 0) {
193        map.put(key, null);
194      }
195      return getKeyValue(map, delimPos + 1, allNameValuePairs, listDelimiter,
196          nameValueSeparator);
197
198    }
199    String key = allNameValuePairs.substring(pos, equalsPos);
200
201    if (delimPos > -1) {
202      String value = allNameValuePairs.substring(equalsPos + 1, delimPos);
203      // dp("cont "+key+","+value+" pos:"+pos+"
204      // len:"+allNameValuePairs.length());
205      key = key.trim();
206
207      map.put(key, value);
208      pos = delimPos + 1;
209
210      // recurse the rest of the values...
211      return getKeyValue(map, pos, allNameValuePairs, listDelimiter, nameValueSeparator);
212    } else {
213      // dp("ERROR: delimPos < 0 ???");
214      return map;
215    }
216  }

218  /**
219   * Convert a string that has delimited values (say comma delimited) in a
220   * String[]. You must explicitly choose whether or not to include empty values
221   * (say two commas that a right beside each other.
222   * 
223   * <P>
224   * e.g. "alpha,beta,,theta"<br>
225   * With keepEmpties true, this results in a String[] of size 4 with the third
226   * one having a String of 0 length. With keepEmpties false, this results in a
227   * String[] of size 3.
228   * </P>
229   * <P>
230   * </P>
231   * <P>
232   * e.g. ",alpha,beta,,theta,"<br>
233   * With keepEmpties true, this results in a String[] of size 6 with the
234   * 1st,4th and 6th one having a String of 0 length. With keepEmpties false,
235   * this results in a String[] of size 3.
236   * </P>
237   */
238  public static String[] delimitedToArray(String str, String delimiter, boolean keepEmpties) {
239
240    ArrayList<String> list = new ArrayList<String>();
241    int startPos = 0;
242    delimiter(str, delimiter, keepEmpties, startPos, list);
243    String[] result = new String[list.size()];
244    return list.toArray(result);
245  }
246
247  private static void delimiter(String str, String delimiter, boolean keepEmpties, int startPos,
248      ArrayList<String> list) {
249
250    int endPos = str.indexOf(delimiter, startPos);
251    if (endPos == -1) {
252      if (startPos <= str.length()) {
253        String lastValue = str.substring(startPos, str.length());
254        if (keepEmpties || lastValue.length() != 0) {
255          list.add(lastValue);
256        }
257      }
258      // we have finished parsing the string...
259
260    } else {
261      // get the delimited value... add it..
262      String value = str.substring(startPos, endPos);
263      if (keepEmpties || value.length() != 0) {
264        list.add(value);
265      }
266      // recursively search as we are not at the end yet...
267      delimiter(str, delimiter, keepEmpties, endPos + 1, list);
268    }
269  }

271  /**
272   * This returns the FIRST string in str that is bounded on the left by
273   * leftBound, and bounded on the right by rightBound. This will return null if
274   * the leftBound is not found within str.
275   * 
276   * <P>
277   * If leftBound can't be found this returns null.
278   * </P>
279   * <P>
280   * This rightBound can't be found then this throws a
281   * StringIndexOutOfBoundsException.
282   * </P>
283   * 
284   * @param str
285   *          the base string that we will search for the bounded string.
286   * @param leftBound
287   *          the left bound of the string.
288   * @param rightBound
289   *          the right bound of the string.
290   */
291  public static String getBoundedString(String str, String leftBound, String rightBound)
292      throws RuntimeException {
293
294    if (str == null) {
295      throw new RuntimeException("string to parse is null?");
296    }
297    int startPos = str.indexOf(leftBound);
298    if (startPos > -1) {
299      startPos = startPos + leftBound.length();
300      int endPos = str.indexOf(rightBound, startPos);
301      // dp(str+" start:"+startPos+" end:"+endPos);
302      if (endPos == -1) {
303        throw new RuntimeException("Can't find rightBound: " + rightBound);
304      }
305      return str.substring(startPos, endPos);
306    } else {
307      // if no leftBound can be found.. return null... could be in a
308      // search n parse type loop?
309      // this keeps "no tag"==null different from "tag not formed
310      // properly"==StringParsingException
311      return null;
312    }
313  }

315  /**
316   * Takes the String bounded by leftBound & rightBound, and replaces it with
317   * replaceString. Actually removes the left and right bound strings aswell.
318   */
319  public static String setBoundedString(String str, String leftBound, String rightBound,
320      String replaceString) {
321
322    int startPos = str.indexOf(leftBound);
323    if (startPos > -1) {
324      // startPos = startPos;
325      int endPos = str.indexOf(rightBound, startPos + leftBound.length());
326      if (endPos > -1) {
327        String toReplace = str.substring(startPos, endPos + 1);
328        return replaceString(str, toReplace, replaceString);
329      } else {
330        return str;
331      }
332    } else {
333      return str;
334    }
335  }

364  /**
365   * This method takes a String and will replace all occurrences of the match
366   * String with that of the replace String.
367   * 
368   * @param source
369   *          the source string
370   * @param match
371   *          the string used to find a match
372   * @param replace
373   *          the string used to replace match with
374   * @return the source string after the search and replace
375   */
376  public static String replaceString(String source, String match, String replace) {
377    if (source == null) {
378      return null;
379    }
380    if (replace == null) {
381      return source;
382    }
383    if (match == null) {
384      throw new NullPointerException("match is null?");
385    }
386    if (match.equals(replace)) {
387      return source;
388    }
389    return replaceString(source, match, replace, 30, 0, source.length());
390  }

392  /**
393   * Additionally specify the additionalSize to add to the buffer. This will
394   * make the buffer bigger so that it doesn't have to grow when replacement
395   * occurs.
396   */
397  public static String replaceString(String source, String match, String replace,
398      int additionalSize, int startPos, int endPos) {
399
400    if (source == null) {
401      return source;
402    }
403
404    char match0 = match.charAt(0);
405
406    int matchLength = match.length();
407
408    if (matchLength == 1 && replace.length() == 1) {
409      char replace0 = replace.charAt(0);
410      return source.replace(match0, replace0);
411    }
412    if (matchLength >= replace.length()) {
413      additionalSize = 0;
414    }
415
416    int sourceLength = source.length();
417    int lastMatch = endPos - matchLength;
418
419    StringBuilder sb = new StringBuilder(sourceLength + additionalSize);
420
421    if (startPos > 0) {
422      sb.append(source.substring(0, startPos));
423    }
424
425    char sourceChar;
426    boolean isMatch;
427    int sourceMatchPos;
428
429    for (int i = startPos; i < sourceLength; i++) {
430      sourceChar = source.charAt(i);
431      if (i > lastMatch || sourceChar != match0) {
432        sb.append(sourceChar);
433
434      } else {
435        // check to see if this is a match
436        isMatch = true;
437        sourceMatchPos = i;
438
439        // check each following character...
440        for (int j = 1; j < matchLength; j++) {
441          sourceMatchPos++;
442          if (source.charAt(sourceMatchPos) != match.charAt(j)) {
443            isMatch = false;
444            break;
445          }
446        }
447        if (isMatch) {
448          i = i + matchLength - 1;
449          sb.append(replace);
450        } else {
451          // was not a match
452          sb.append(sourceChar);
453        }
454      }
455    }
456
457    return sb.toString();
458  }

477  /**
478   * Additionally specify an additional size estimate for the buffer plus start
479   * and end positions.
480   * <p>
481   * The start and end positions can limit the search and replace. Otherwise
482   * these default to startPos = 0 and endPos = source.length().
483   * </p>
484   */
485  public static String replaceStringMulti(String source, String[] match, String replace,
486      int additionalSize, int startPos, int endPos) {
487
488    int shortestMatch = match[0].length();
489
490    char[] match0 = new char[match.length];
491    for (int i = 0; i < match0.length; i++) {
492      match0[i] = match[i].charAt(0);
493      if (match[i].length() < shortestMatch) {
494        shortestMatch = match[i].length();
495      }
496    }
497
498    StringBuilder sb = new StringBuilder(source.length() + additionalSize);
499
500    char sourceChar;
501
502    int len = source.length();
503    int lastMatch = endPos - shortestMatch;
504
505    if (startPos > 0) {
506      sb.append(source.substring(0, startPos));
507    }
508
509    int matchCount;
510
511    for (int i = startPos; i < len; i++) {
512      sourceChar = source.charAt(i);
513      if (i > lastMatch) {
514        sb.append(sourceChar);
515      } else {
516        matchCount = 0;
517        for (int k = 0; k < match0.length; k++) {
518          if (matchCount == 0 && sourceChar == match0[k]) {
519            if (match[k].length() + i <= len) {
520
521              ++matchCount;
522              int j = 1;
523              for (; j < match[k].length(); j++) {
524                if (source.charAt(i + j) != match[k].charAt(j)) {
525                  --matchCount;
526                  break;
527                }
528              }
529              if (matchCount > 0) {
530                i = i + j - 1;
531                sb.append(replace);
532                break;
533              }
534            }
535          }
536        }
537        if (matchCount == 0) {
538          sb.append(sourceChar);
539        }
540      }
541    }
542
543    return sb.toString();
544  }

546  /**
547   * This method takes a String as an argument and removes all occurrences of
548   * the supplied Char. It returns the resulting String.
549   */
550  public static String removeChar(String s, char chr) {
551
552    StringBuilder sb = new StringBuilder(s.length());
553
554    for (int i = 0; i < s.length(); i++) {
555      char c = s.charAt(i);
556      if (c != chr) {
557        sb.append(c);
558      }
559    }
560
561    return sb.toString();
562  }

564  /**
565   * This method takes a String as an argument and removes all occurrences of
566   * the supplied Chars. It returns the resulting String.
567   */
568  public static String removeChars(String s, char[] chr) {
569
570    StringBuilder sb = new StringBuilder(s.length());
571
572    for (int i = 0; i < s.length(); i++) {
573      char c = s.charAt(i);
574      if (!charMatch(c, chr)) {
575        sb.append(c);
576      }
577    }
578
579    return sb.toString();
580  }
581
582  private static boolean charMatch(int iChr, char[] chr) {
583    for (int i = 0; i < chr.length; i++) {
584      if (iChr == chr[i]) {
585        return true;
586      }
587    }
588    return false;
589  }
