020/**
021 * The configuration used for creating a EbeanServer.
022 * <p>
023 * Used to programmatically construct an EbeanServer and optionally register it
024 * with the Ebean singleton.
025 * </p>
026 * <p>
027 * If you just use Ebean without this programmatic configuration Ebean will read
028 * the ebean.properties file and take the configuration from there. This usually
029 * includes searching the class path and automatically registering any entity
030 * classes and listeners etc.
031 * </p>
032 * 
033 * <pre>{@code
034 * ServerConfig c = new ServerConfig();
035 * c.setName("ordh2");
036 * 
037 * // read the ebean.properties and load
038 * // those settings into this serverConfig object
039 * c.loadFromProperties();
040 * 
041 * // generate DDL and run it
042 * c.setDdlGenerate(true);
043 * c.setDdlRun(true);
044 * 
045 * // add any classes found in the app.data package
046 * c.addPackage("app.data");
047 * 
048 * // add the names of Jars that contain entities
049 * c.addJar("myJarContainingEntities.jar");
050 * c.addJar("someOtherJarContainingEntities.jar");
051 * 
052 * // register as the 'Default' server
053 * c.setDefaultServer(true);
054 * 
055 * EbeanServer server = EbeanServerFactory.create(c);
056 * 
057 * }</pre>
058 * 
059 * @see EbeanServerFactory
060 * 
061 * @author emcgreal
062 * @author rbygrave
063 */

1697   */

1699
1700    namingConvention = createNamingConvention(p, namingConvention);
1701    if (namingConvention != null) {
1702      namingConvention.loadFromProperties(p);
1703    }
1704    if (autofetchConfig == null) {
1705      autofetchConfig = new AutofetchConfig();
1706    }
1707    loadAutofetchSettings(p);
1708
1709    if (dataSourceConfig == null) {
1710      dataSourceConfig = new DataSourceConfig();

1712    loadDataSourceSettings(p);
1713
1714    autoCommitMode = p.getBoolean("autoCommitMode", autoCommitMode);

1722    serverCacheFactory = createInstance(p, ServerCacheFactory.class, "serverCacheFactory");
1723    serverCacheManager = createInstance(p, ServerCacheManager.class, "serverCacheManager");
1724    cacheWarmingDelay = p.getInt("cacheWarmingDelay", cacheWarmingDelay);
1725    classPathReaderClassName = p.get("classpathreader");
1726    
1727    String jarsProp = p.get("search.jars", p.get("jars", null));
1728    if (jarsProp != null) {
1729      searchJars = getSearchJarsPackages(jarsProp);
1730    }
1731
1732    if (packages != null) {
1733      String packagesProp = p.get("search.packages", p.get("packages", null));

1743    updatesDeleteMissingChildren = p.getBoolean("updatesDeleteMissingChildren", defaultDeleteMissingChildren);
1744
1745    if (p.get("batch.mode") != null || p.get("persistBatching") != null) {
1746      throw new IllegalArgumentException("Property 'batch.mode' or 'persistBatching' is being set but no longer used. Please change to use 'persistBatchMode'");
1747    }
1748
1749    persistBatch = p.getEnum(PersistBatch.class, "persistBatch", persistBatch);

1756

1787  /**
1788   * Build the list of classes from the comma delimited string.
1789   * 
1790   * @param properties
1791   *          the properties
1792   * 
1793   * @return the classes
1794   */
1795  private List<Class<?>> getClasses(PropertiesWrapper properties) {
1796
1797    String classNames = properties.get("classes", null);
1798    if (classNames == null) {
1799
1800      return null;
1801    }
1802
1803    List<Class<?>> classes = new ArrayList<Class<?>>();
1804
1805    String[] split = classNames.split("[ ,;]");
1806    for (int i = 0; i < split.length; i++) {
1807      String cn = split[i].trim();
1808      if (cn.length() > 0 && !"class".equalsIgnoreCase(cn)) {
1809        try {
1810          classes.add(Class.forName(cn));
1811        } catch (ClassNotFoundException e) {
1812          String msg = "Error registering class [" + cn + "] from [" + classNames + "]";
1813          throw new RuntimeException(msg, e);
1814        }
1815      }
1816    }
1817    return classes;
1818  }

1821
1822    List<String> hitList = new ArrayList<String>();
1823
1824    if (searchPackages != null) {
1825
1826      String[] entries = searchPackages.split("[ ,;]");
1827      for (int i = 0; i < entries.length; i++) {
1828        hitList.add(entries[i].trim());
1829      }
1830    }
1831    return hitList;
1832  }
