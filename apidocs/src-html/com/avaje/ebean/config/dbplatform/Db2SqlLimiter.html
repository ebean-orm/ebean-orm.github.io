001package com.avaje.ebean.config.dbplatform;
002
003public class Db2SqlLimiter implements SqlLimiter {
004
005  @Override
006  public SqlLimitResponse limit(SqlLimitRequest request) {
007
008    StringBuilder sb = new StringBuilder(512);
009    sb.append("select ");
010    sb.append(request.getDbSql());
011
012    int maxRows = request.getMaxRows();
013    if (maxRows > 0) {
014      sb.append(" ").append(NEW_LINE).append("FETCH FIRST ").append(maxRows).append(" ROWS ONLY");
015    }
016
017    String sql = request.getDbPlatform().completeSql(sb.toString(), request.getOrmQuery());
018    return new SqlLimitResponse(sql, false);
019  }
020}
