001package com.avaje.ebean.config.dbplatform;
002
003import com.avaje.ebean.BackgroundExecutor;
004
005import javax.sql.DataSource;
006import java.sql.Types;
007
008/**
009 * Postgres v9 specific platform.
010 * <p>
011 * Uses serial types and getGeneratedKeys.
012 * </p>
013 */
014public class PostgresPlatform extends DatabasePlatform {
015
016  /**
017   * Unique jdbc type id defined for hstore type.
018   */
019  public static final int TYPE_HSTORE = 4001;
020  
021  public PostgresPlatform() {
022    super();
023    this.name = "postgres";
024    // OnQueryOnly.CLOSE as a performance optimisation on Postgres
025    this.onQueryOnly = OnQueryOnly.CLOSE;
026    this.likeClause = "like ? escape''";
027    
028    this.dbDdlSyntax = new PostgresDdlSyntax();
029    
030    this.selectCountWithAlias = true;
031    this.blobDbType = Types.LONGVARBINARY;
032    this.clobDbType = Types.VARCHAR;
033
034    this.dbEncrypt = new PostgresDbEncrypt();
035
036    // Use Identity and getGeneratedKeys
037    this.dbIdentity.setIdType(IdType.IDENTITY);
038    this.dbIdentity.setSupportsGetGeneratedKeys(true);
039    this.dbIdentity.setSupportsSequence(true);
040
041    this.columnAliasPrefix = "as c";
042
043    this.openQuote = "\"";
044    this.closeQuote = "\"";
045
046    dbTypeMap.put(TYPE_HSTORE, new DbType("hstore"));
047    
048    dbTypeMap.put(Types.INTEGER, new DbType("integer", false));
049    dbTypeMap.put(Types.DOUBLE, new DbType("float"));
050    dbTypeMap.put(Types.TINYINT, new DbType("smallint"));
051    dbTypeMap.put(Types.DECIMAL, new DbType("decimal", 38));
052
053    dbTypeMap.put(Types.BINARY, new DbType("bytea", false));
054    dbTypeMap.put(Types.VARBINARY, new DbType("bytea", false));
055
056    dbTypeMap.put(Types.BLOB, new DbType("bytea", false));
057    dbTypeMap.put(Types.CLOB, new DbType("text"));
058    dbTypeMap.put(Types.LONGVARBINARY, new DbType("bytea", false));
059    dbTypeMap.put(Types.LONGVARCHAR, new DbType("text"));
060
061    dbDdlSyntax.setDropTableCascade("cascade");
062    dbDdlSyntax.setDropIfExists("if exists");
063
064  }
065
066  /**
067   * Create a Postgres specific sequence IdGenerator.
068   */
069  @Override
070  public IdGenerator createSequenceIdGenerator(BackgroundExecutor be, DataSource ds,
071      String seqName, int batchSize) {
072
073    return new PostgresSequenceIdGenerator(be, ds, seqName, batchSize);
074  }
075
076  @Override
077  protected String withForUpdate(String sql) {
078    return sql + " for update";
079  }
080}
