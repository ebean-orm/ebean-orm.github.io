001package com.avaje.ebean.config.dbplatform;
002
003/**
004 * Postgres v8.3 specific platform.
005 * <p>
006 * No support for getGeneratedKeys.
007 * </p>
008 */
009public class Postgres8Platform extends PostgresPlatform {
010
011  public Postgres8Platform() {
012    super();
013    this.dbIdentity.setSupportsGetGeneratedKeys(false);
014    this.dbIdentity.setIdType(IdType.SEQUENCE);
015    this.dbIdentity.setSupportsSequence(true);
016    this.dbIdentity.setSupportsIdentity(true);
017    this.columnAliasPrefix = "as c";
018  }
019
020}
