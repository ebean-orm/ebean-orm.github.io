001package com.avaje.ebean.config.dbplatform;
002
003import com.avaje.ebean.config.PersistBatch;
004
005/**
006 * Microsoft SQL Server 2000 specific platform.
007 * <p>
008 * <ul>
009 * <li>supportsGetGeneratedKeys = false</li>
010 * <li>Use select @@IDENTITY to return the generated Id instead</li>
011 * <li>Uses LIMIT OFFSET clause</li>
012 * <li>Uses [ & ] for quoted identifiers</li>
013 * </ul>
014 * </p>
015 */
016public class MsSqlServer2000Platform extends DatabasePlatform {
017
018  public MsSqlServer2000Platform() {
019    super();
020    this.name = "mssqlserver2000";
021    this.persistBatchOnCascade = PersistBatch.NONE;
022    this.dbIdentity.setIdType(IdType.IDENTITY);
023    this.dbIdentity.setSupportsGetGeneratedKeys(false);
024    this.dbIdentity.setSelectLastInsertedIdTemplate("select @@IDENTITY as X");
025    this.dbIdentity.setSupportsIdentity(true);
026
027    this.openQuote = "[";
028    this.closeQuote = "]";
029
030    dbTypeMap.put(DbType.BOOLEAN, new DbPlatformType("bit default 0"));
031
032    dbTypeMap.put(DbType.BIGINT, new DbPlatformType("numeric", 19));
033    dbTypeMap.put(DbType.REAL, new DbPlatformType("float(16)"));
034    dbTypeMap.put(DbType.DOUBLE, new DbPlatformType("float(32)"));
035    dbTypeMap.put(DbType.TINYINT, new DbPlatformType("smallint"));
036    dbTypeMap.put(DbType.DECIMAL, new DbPlatformType("numeric", 28));
037
038    dbTypeMap.put(DbType.BLOB, new DbPlatformType("image"));
039    dbTypeMap.put(DbType.CLOB, new DbPlatformType("text"));
040    dbTypeMap.put(DbType.LONGVARBINARY, new DbPlatformType("image"));
041    dbTypeMap.put(DbType.LONGVARCHAR, new DbPlatformType("text"));
042
043    dbTypeMap.put(DbType.DATE, new DbPlatformType("datetime"));
044    dbTypeMap.put(DbType.TIME, new DbPlatformType("datetime"));
045    dbTypeMap.put(DbType.TIMESTAMP, new DbPlatformType("datetime"));
046
047  }
048
049}
