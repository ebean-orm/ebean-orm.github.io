020  /**
021   * Behavior used when ending a query only transaction (at read committed isolation level).

033     */
034    CLOSE,

036    /**
037     * Commit the transaction
038     */
039    COMMIT
040  }

353  /**
354   * Convert backticks to the platform specific open quote and close quote
355   * 
356   * <p>
357   * Specific plugins may implement this method to cater for platform specific
358   * naming rules.
359   * </p>
360   * 
361   * @param dbName
362   *          the db name
363   * 
364   * @return the string
365   */
366  public String convertQuotedIdentifiers(String dbName) {
367    // Ignore null values e.g. schema name or catalog
368    if (dbName != null && dbName.length() > 0) {
369      if (dbName.charAt(0) == BACK_TICK) {
370        if (dbName.charAt(dbName.length() - 1) == BACK_TICK) {
371
372          String quotedName = getOpenQuote();
373          quotedName += dbName.substring(1, dbName.length() - 1);
374          quotedName += getCloseQuote();
375
376          return quotedName;
377
378        } else {
379          logger.error("Missing backquote on [" + dbName + "]");
380        }
381      }
382    }
383    return dbName;
384  }

393  public String completeSql(String sql, Query<?> query) {
394    if (Boolean.TRUE.equals(query.isForUpdate())) {
395      sql = withForUpdate(sql);
396    }
397
398    return sql;
399  }
400
401  protected String withForUpdate(String sql) {
402    // silently assume the database does not support the "for update" clause.
403    logger.info("it seems your database does not support the 'for update' clause");
404
405    return sql;
406  }
