001package com.avaje.ebean.config.dbplatform;
002
003import java.sql.Types;
004import java.util.HashMap;
005import java.util.Map;
006
007/**
008 * Used to map bean property types to DB specific types for DDL generation.
009 */
010public class DbTypeMap {
011
012  private final Map<Integer, DbType> typeMap = new HashMap<Integer, DbType>();
013
014  public DbTypeMap() {
015    loadDefaults();
016  }
017
018  /**
019   * Load the standard types. These can be overridden by DB specific platform.
020   */
021  private void loadDefaults() {
022
023    put(Types.BOOLEAN, new DbType("boolean"));
024    put(Types.BIT, new DbType("bit"));
025
026    put(Types.INTEGER, new DbType("integer"));
027    put(Types.BIGINT, new DbType("bigint"));
028    put(Types.REAL, new DbType("float"));
029    put(Types.DOUBLE, new DbType("double"));
030    put(Types.SMALLINT, new DbType("smallint"));
031    put(Types.TINYINT, new DbType("tinyint"));
032    put(Types.DECIMAL, new DbType("decimal", 38));
033
034    put(Types.VARCHAR, new DbType("varchar", 255));
035    put(Types.CHAR, new DbType("char", 1));
036
037    put(Types.BLOB, new DbType("blob"));
038    put(Types.CLOB, new DbType("clob"));
039    put(Types.LONGVARBINARY, new DbType("longvarbinary"));
040    put(Types.LONGVARCHAR, new DbType("lonvarchar"));
041    put(Types.VARBINARY, new DbType("varbinary", 255));
042    put(Types.BINARY, new DbType("binary", 255));
043
044    put(Types.DATE, new DbType("date"));
045    put(Types.TIME, new DbType("time"));
046    put(Types.TIMESTAMP, new DbType("timestamp"));
047
048  }
049
050  /**
051   * Override the type for a given JDBC type.
052   */
053  public void put(int jdbcType, DbType dbType) {
054    typeMap.put(Integer.valueOf(jdbcType), dbType);
055  }
056
057  /**
058   * Return the type for a given jdbc type.
059   */
060  public DbType get(int jdbcType) {
061
062    DbType dbType = typeMap.get(Integer.valueOf(jdbcType));
063    if (dbType == null) {
064      String m = "No DB type for JDBC type " + jdbcType;
065      throw new RuntimeException(m);
066    }
067
068    return dbType;
069  }
070}
