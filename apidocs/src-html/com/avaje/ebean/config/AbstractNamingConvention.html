001package com.avaje.ebean.config;
002
003import javax.persistence.Inheritance;
004import javax.persistence.Table;
005
006import com.avaje.ebean.config.dbplatform.DatabasePlatform;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;

010/**
011 * Provides some base implementation for NamingConventions.
012 *
013 * @author emcgreal
014 */
015public abstract class AbstractNamingConvention implements NamingConvention {
016
017  /** The Constant logger. */
018  private static final Logger logger = LoggerFactory.getLogger(AbstractNamingConvention.class);
019
020  /** The Constant DEFAULT_SEQ_FORMAT. */
021  public static final String DEFAULT_SEQ_FORMAT = "{table}_seq";
022
023  /** Sequence Format that includes the Primary Key column */
024  public static final String TABLE_PKCOLUMN_SEQ_FORMAT = "{table}_{column}_seq";
025
026  /** The catalog. */
027  private String catalog;
028
029  /** The schema. */
030  private String schema;
031
032  /** The sequence format. */
033  private String sequenceFormat;
034
035  /** The database platform. */
036  protected DatabasePlatform databasePlatform;
037
038  /** The max length of constraint names. */
039  protected int maxConstraintNameLength;
040
041  /** Used to trim off extra prefix for M2M. */
042  protected int rhsPrefixLength = 3;
043
044  protected boolean useForeignKeyPrefix;
045
046  /**
047   * Construct with a sequence format and useForeignKeyPrefix setting.
048   */
049  public AbstractNamingConvention(String sequenceFormat, boolean useForeignKeyPrefix) {
050    this.sequenceFormat = sequenceFormat;
051    this.useForeignKeyPrefix = useForeignKeyPrefix;
052  }
053
054  /**
055   * Construct with a sequence format.
056   *
057   * @param sequenceFormat the sequence format
058   */
059  public AbstractNamingConvention(String sequenceFormat) {
060    this.sequenceFormat = sequenceFormat;
061    this.useForeignKeyPrefix = true;
062  }
063
064  /**
065   * Construct with the default sequence format ("{table}_seq") and
066   * useForeignKeyPrefix as true.
067   */
068  public AbstractNamingConvention() {
069    this(DEFAULT_SEQ_FORMAT);
070  }
071
072  public void setDatabasePlatform(DatabasePlatform databasePlatform) {
073    this.databasePlatform = databasePlatform;
074    this.maxConstraintNameLength = databasePlatform.getDbDdlSyntax().getMaxConstraintNameLength();
075
076    logger.trace("Using maxConstraintNameLength of " + maxConstraintNameLength);
077  }
078
079  public String getSequenceName(String tableName, String pkColumn) {
080    String s = sequenceFormat.replace("{table}", tableName);
081    if (pkColumn == null) {
082      pkColumn = "";
083    }
084    return s.replace("{column}", pkColumn);
085  }
086
087  /**
088   * Return the catalog.
089   */
090  public String getCatalog() {
091    return catalog;
092  }
093
094  /**
095   * Sets the catalog.
096   */
097  public void setCatalog(String catalog) {
098    this.catalog = catalog;
099  }
100
101  /**
102   * Return the schema.
103   */
104  public String getSchema() {
105    return schema;
106  }
107
108  /**
109   * Sets the schema.
110   */
111  public void setSchema(String schema) {
112    this.schema = schema;
113  }
114
115  /**
116   * Returns the sequence format.
117   */
118  public String getSequenceFormat() {
119    return sequenceFormat;
120  }
121
122  /**
123   * Set the sequence format used to generate the sequence name.
124   * <p>
125   * The format should include "{table}". When generating the sequence name
126   * {table} is replaced with the actual table name.
127   * </p>
128   *
129   * @param sequenceFormat string containing "{table}" which is replaced with the actual
130   *                       table name to generate the sequence name.
131   */
132  public void setSequenceFormat(String sequenceFormat) {
133    this.sequenceFormat = sequenceFormat;
134  }
135
136  /**
137   * Return true if a prefix should be used building a foreign key name.
138   * <p>
139   * This by default is true and this works well when the primary key column
140   * names are simply "ID". In this case a prefix (such as "ORDER" and
141   * "CUSTOMER" etc) is added to the foreign key column producing "ORDER_ID" and
142   * "CUSTOMER_ID".
143   * </p>
144   * <p>
145   * This should return false when your primary key columns are the same as the
146   * foreign key columns. For example, when the primary key columns are
147   * "ORDER_ID", "CUST_ID" etc ... and they are the same as the foreign key
148   * column names.
149   * </p>
150   */
151  public boolean isUseForeignKeyPrefix() {
152    return useForeignKeyPrefix;
153  }
154
155  /**
156   * Set this to false when the primary key columns matching your foreign key
157   * columns.
158   */
159  public void setUseForeignKeyPrefix(boolean useForeignKeyPrefix) {
160    this.useForeignKeyPrefix = useForeignKeyPrefix;
161  }
162
163  /**
164   * Return the tableName using the naming convention (rather than deployed
165   * Table annotation).
166   */
167  protected abstract TableName getTableNameByConvention(Class<?> beanClass);
168
169  /**
170   * Returns the table name for a given entity bean.
171   * <p>
172   * This first checks for the @Table annotation and if not present uses the
173   * naming convention to define the table name.
174   * </p>
175   *
176   * @see #getTableNameFromAnnotation(Class)
177   * @see #getTableNameByConvention(Class)
178   */
179  public TableName getTableName(Class<?> beanClass) {
180
181    TableName tableName = getTableNameFromAnnotation(beanClass);
182    if (tableName == null) {
183
184      Class<?> supCls = beanClass.getSuperclass();
185      Inheritance inheritance = supCls.getAnnotation(Inheritance.class);
186      if (inheritance != null) {
187        // get the table as per inherited class in case their
188        // is not a table annotation in the inheritance hierarchy
189        return getTableName(supCls);
190      }
191
192      tableName = getTableNameByConvention(beanClass);
193    }
194
195    // Use naming convention for catalog or schema,
196    // if not set in the annotation.
197    String catalog = tableName.getCatalog();
198    if (isEmpty(catalog)) {
199      catalog = getCatalog();
200    }
201    String schema = tableName.getSchema();
202    if (isEmpty(schema)) {
203      schema = getSchema();
204    }
205    return new TableName(catalog, schema, tableName.getName());
206  }
207
208  public TableName getM2MJoinTableName(TableName lhsTable, TableName rhsTable) {
209
210    StringBuilder buffer = new StringBuilder();
211    buffer.append(lhsTable.getName());
212    buffer.append("_");
213
214    String rhsTableName = rhsTable.getName();
215    if (rhsTableName.indexOf('_') < rhsPrefixLength) {
216      // trim off a xx_ prefix if there is one
217      rhsTableName = rhsTableName.substring(rhsTableName.indexOf('_') + 1);
218    }
219    buffer.append(rhsTableName);
220
221    int maxConstraintNameLength = databasePlatform.getDbDdlSyntax().getMaxConstraintNameLength();
222
223    // maxConstraintNameLength is used as the max table name length.
224    if (buffer.length() > maxConstraintNameLength) {
225      buffer.setLength(maxConstraintNameLength);
226    }
227
228    return new TableName(lhsTable.getCatalog(), lhsTable.getSchema(), buffer.toString());
229  }
230
231  /**
232   * Gets the table name from annotation.
233   */
234  protected TableName getTableNameFromAnnotation(Class<?> beanClass) {
235
236    final Table t = findTableAnnotation(beanClass);
237
238    // Take the annotation if defined
239    if (t != null && !isEmpty(t.name())) {
240      // Note: empty catalog and schema are converted to null
241      // Only need to convert quoted identifiers from annotations
242      return new TableName(quoteIdentifiers(t.catalog()), quoteIdentifiers(t.schema()),  quoteIdentifiers(t.name()));
243    }
244
245    // No annotation
246    return null;
247  }
248
249  /**
250   * Search recursively for an @Table in the class hierarchy.
251   */
252  protected Table findTableAnnotation(Class<?> cls) {
253    if (cls.equals(Object.class)) {
254      return null;
255    }
256    Table table = cls.getAnnotation(Table.class);
257    if (table != null) {
258      return table;
259    }
260    return findTableAnnotation(cls.getSuperclass());
261  }
262
263  /**
264   * Replace back ticks (if they are used) with database platform specific
265   * quoted identifiers.
266   */
267  protected String quoteIdentifiers(String s) {
268    return databasePlatform.convertQuotedIdentifiers(s);
269  }
270
271  /**
272   * Checks string is null or empty .
273   */
274  protected boolean isEmpty(String s) {
275    if (s == null || s.trim().length() == 0) {
276      return true;
277    }
278    return false;
279  }
280
281  /**
282   * Load settings from properties.
283   */
284  @Override
285  public void loadFromProperties(PropertiesWrapper properties) {
286
287    useForeignKeyPrefix = properties.getBoolean("namingConvention.useForeignKeyPrefix", useForeignKeyPrefix);
288    sequenceFormat = properties.get("namingConvention.sequenceFormat", sequenceFormat);
289    schema = properties.get("namingConvention.schema", schema);
290  }
