001package com.avaje.ebean;
002
003import java.beans.PropertyChangeListener;
004import java.util.Map;
005import java.util.Set;
006
007/**
008 * Provides access to the internal state of an entity bean.
009 */
010public interface BeanState {
011
012  /**
013   * Return true if this is a lazy loading reference bean.
014   * <p>
015   * If so the this bean only holds the Id property and will invoke lazy loading
016   * if any other property is get or set.
017   * </p>
018   */
019  public boolean isReference();
020
021  /**
022   * Return true if the bean is new (and not yet saved).
023   */
024  public boolean isNew();
025
026  /**
027   * Return true if the bean is new or dirty (and probably needs to be saved).
028   */
029  public boolean isNewOrDirty();
030
031  /**
032   * Return true if the bean has been changed but not yet saved.
033   */
034  public boolean isDirty();
035
036  /**
037   * For partially populated beans returns the properties that are loaded on the
038   * bean.
039   * <p>
040   * Accessing another property will cause lazy loading to occur.
041   * </p>
042   */
043  public Set<String> getLoadedProps();
044
045  /**
046   * Return the set of changed properties.
047   */
048  public Set<String> getChangedProps();
049
050  /**
051   * Return a map of the updated properties and their new and old values.
052   */
053  public Map<String,ValuePair> getDirtyValues();
054  
055  /**
056   * Return true if the bean is readOnly.
057   * <p>
058   * If a setter is called on a readOnly bean it will throw an exception.
059   * </p>
060   */
061  public boolean isReadOnly();
062
063  /**
064   * Set the readOnly status for the bean.
065   */
066  public void setReadOnly(boolean readOnly);
067
068  /**
069   * Add a propertyChangeListener.
070   */
071  public void addPropertyChangeListener(PropertyChangeListener listener);
072
073  /**
074   * Remove a propertyChangeListener.
075   */
076  public void removePropertyChangeListener(PropertyChangeListener listener);
077
078  /**
079   * Advanced - Used to programmatically build a partially or fully loaded
080   * entity bean. First create an entity bean via
081   * {@link EbeanServer#createEntityBean(Class)}, then populate its properties
082   * and then call this method specifying which properties where loaded or null
083   * for a fully loaded entity bean.
084   */
085  public void setLoaded();
086}
