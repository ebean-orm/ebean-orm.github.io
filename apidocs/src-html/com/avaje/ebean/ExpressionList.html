011/**
012 * List of Expressions that make up a where or having clause.
013 * <p>
014 * An ExpressionList is returned from {@link Query#where()}.
015 * </p>
016 * <p>
017 * The ExpressionList has a list of convenience methods that create the standard
018 * expressions and add them to this list.
019 * </p>
020 * <p>
021 * The ExpressionList also duplicates methods that are found on the Query such
022 * as findList() and orderBy(). The purpose of these methods is provide a fluid
023 * API. The upside of this approach is that you can build and execute a query
024 * via chained methods. The down side is that this ExpressionList object has
025 * more methods than you would initially expect (the ones duplicated from
026 * Query).
027 * </p>
028 * 
029 * @see Query#where()
030 */
031public interface ExpressionList<T> extends Serializable {
032
033  /**
034   * Return the query that owns this expression list.
035   * <p>
036   * This is a convenience method solely to support a fluid API where the
037   * methods are chained together. Adding expressions returns this expression
038   * list and this method can be used after that to return back the original
039   * query so that further things can be added to it.
040   * </p>
041   */
042  public Query<T> query();
043
044  /**
045   * Set the order by clause replacing the existing order by clause if there is
046   * one.
047   * <p>
048   * This follows SQL syntax using commas between each property with the
049   * optional asc and desc keywords representing ascending and descending order
050   * respectively.
051   * </p>
052   * <p>
053   * This is EXACTLY the same as {@link #orderBy(String)}.
054   * </p>
055   */
056  public Query<T> order(String orderByClause);
057
058  /**
059   * Return the OrderBy so that you can append an ascending or descending
060   * property to the order by clause.
061   * <p>
062   * This will never return a null. If no order by clause exists then an 'empty'
063   * OrderBy object is returned.
064   * </p>
065   */
066  public OrderBy<T> order();
067
068  /**
069   * Return the OrderBy so that you can append an ascending or descending
070   * property to the order by clause.
071   * <p>
072   * This will never return a null. If no order by clause exists then an 'empty'
073   * OrderBy object is returned.
074   * </p>
075   */
076  public OrderBy<T> orderBy();
077
078  /**
079   * Add an orderBy clause to the query.
080   * 
081   * @see Query#orderBy(String)
082   */
083  public Query<T> orderBy(String orderBy);
084
085  /**
086   * Add an orderBy clause to the query.
087   * 
088   * @see Query#orderBy(String)
089   */
090  public Query<T> setOrderBy(String orderBy);
091
092  /**
093   * Apply the path properties to the query replacing the select and fetch clauses.
094   */
095  public Query<T> apply(PathProperties pathProperties);
096

099   *

101   */
102  public QueryIterator<T> findIterate();

104  /**

177   * <p>

211  /**
212   * Return a PagedList for this query.
213   * <p>
214   * The benefit of using this over just using the normal {@link Query#setFirstRow(int)} and
215   * {@link Query#setMaxRows(int)} is that it additionally wraps an optional call to
216   * {@link Query#findFutureRowCount()} to determine total row count, total page count etc.

218   * <p>
219   * Internally this works using {@link Query#setFirstRow(int)} and {@link Query#setMaxRows(int)} on
220   * the query. This translates into SQL that uses limit offset, rownum or row_number
221   * function to limit the result set.
222   * </p>
223   * 
224   * @param pageIndex
225   *          The zero based index of the page.
226   * @param pageSize
227   *          The number of beans to return per page.
228   * @return The PagedList
229   */
230  public PagedList<T> findPagedList(int pageIndex, int pageSize);

366  /**
367   * A "Query By Example" type of expression.
368   * <p>
369   * Pass in an example entity and for each non-null scalar properties an
370   * expression is added.
371   * </p>
372   * <p>
373   * By Default this case sensitive, will ignore numeric zero values and will
374   * use a Like for string values (you must put in your own wildcards).
375   * </p>
376   * <p>
377   * To get control over the options you can create an ExampleExpression and set
378   * those options such as case insensitive etc.
379   * </p>
380   * 
381   * <pre class="code">
382   * // create an example bean and set the properties
383   * // with the query parameters you want
384   * Customer example = new Customer();
385   * example.setName(&quot;Rob%&quot;);
386   * example.setNotes(&quot;%something%&quot;);
387   * 
388   * List&lt;Customer&gt; list = Ebean.find(Customer.class).where()
389   *     // pass the bean into the where() clause
390   *     .exampleLike(example)
391   *     // you can add other expressions to the same query
392   *     .gt(&quot;id&quot;, 2).findList();
393   * 
394   * </pre>
395   * 
396   * Similarly you can create an ExampleExpression
397   * 
398   * <pre>
399   * Customer example = new Customer();
400   * example.setName(&quot;Rob%&quot;);
401   * example.setNotes(&quot;%something%&quot;);
402   * 
403   * // create a ExampleExpression with more control
404   * ExampleExpression qbe = new ExampleExpression(example, true, LikeType.EQUAL_TO).includeZeros();
405   * 
406   * List&lt;Customer&gt; list = Ebean.find(Customer.class).where().add(qbe).findList();
407   * </pre>
408   */
409  public ExpressionList<T> exampleLike(Object example);
410

412   * Case insensitive version of {@link #exampleLike(Object)}
413   */
414  public ExpressionList<T> iexampleLike(Object example);
