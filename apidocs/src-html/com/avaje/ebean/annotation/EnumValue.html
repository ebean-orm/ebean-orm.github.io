001package com.avaje.ebean.annotation;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * Enables you to specify a value to use to persist for an enum value.
010 * 
011 * <pre class="code">
012 * public enum Status {
013 *   &#064;EnumValue(&quot;N&quot;)
014 *   NEW,
015 * 
016 *   &#064;EnumValue(&quot;A&quot;)
017 *   ACTIVE,
018 * 
019 *   &#064;EnumValue(&quot;I&quot;)
020 *   INACTIVE,
021 * }
022 * 
023 * </pre>
024 * <p>
025 * This is an alternative to using the JPA standard approach or Ebean's
026 * {@link EnumMapping} annotation.
027 * </p>
028 * <p>
029 * Note that if all the EnumValue values are parsable as Integers then Ebean
030 * will persist and fetch them as integers - otherwise they will be persisted
031 * and fetched as strings.
032 * </p>
033 */
034@Target({ ElementType.FIELD })
035@Retention(RetentionPolicy.RUNTIME)
036public @interface EnumValue {
037
038  /**
039   * Specify the value to persist for a specific enum value.
040   * <p>
041   * If all the values are parsable as Integers then Ebean will persist and
042   * fetch them as integers rather than strings.
043   * </p>
044   */
045  String value();
046};
