001package io.ebean;
002
003/**
004 * A SqlUpdate for executing insert update or delete statements.
005 * <p>
006 * Provides a simple way to execute raw SQL insert update or delete statements
007 * without having to resort to JDBC.
008 * </p>
009 * <p>
010 * Supports the use of positioned or named parameters and can automatically
011 * notify Ebean of the table modified so that Ebean can maintain its cache.
012 * </p>
013 * <p>
014 * Note that {@link #setAutoTableMod(boolean)} and
015 * Ebean#externalModification(String, boolean, boolean, boolean)} can be to
016 * notify Ebean of external changes and enable Ebean to maintain it's "L2"
017 * server cache.
018 * </p>
019 * <p>
020 * <pre>{@code
021 *
022 *   // example that uses 'named' parameters
023 *
024 *   String s = "UPDATE f_topic set post_count = :count where id = :id";
025 *
026 *   SqlUpdate update = Ebean.createSqlUpdate(s);
027 *   update.setParameter("id", 1);
028 *   update.setParameter("count", 50);
029 *
030 *   int modifiedCount = update.execute();
031 *
032 *   String msg = "There were " + modifiedCount + " rows updated";
033 *
034 * }</pre>
035 * <p>
036 * <h3>Example: Using setNextParameter()</h3>
037 * <pre>{@code
038 *
039 *  String sql = "insert into audit_log (id, description, modified_description) values (?,?,?)";
040 *  SqlUpdate insert = Ebean.createSqlUpdate(sql);
041 *
042 *  try (Transaction txn = Ebean.beginTransaction()) {
043 *    txn.setBatchMode(true);
044 *
045 *    insert.setNextParameter(10000);
046 *    insert.setNextParameter("hello");
047 *    insert.setNextParameter("rob");
048 *    insert.execute();
049 *
050 *    insert.setNextParameter(10001);
051 *    insert.setNextParameter("goodbye");
052 *    insert.setNextParameter("rob");
053 *    insert.execute();
054 *
055 *    insert.setNextParameter(10002);
056 *    insert.setNextParameter("chow");
057 *    insert.setNextParameter("bob");
058 *    insert.execute();
059 *
060 *    txn.commit();
061 *  }
062 * }</pre>
063 * <p>
064 * An alternative to the batch mode on the transaction is to use addBatch() and executeBatch() like:
065 * </p>
066 * <pre>{@code
067 *
068 *   try (Transaction txn = Ebean.beginTransaction()) {
069 *
070 *     insert.setNextParameter(10000);
071 *     insert.setNextParameter("hello");
072 *     insert.setNextParameter("rob");
073 *     insert.addBatch();
074 *
075 *     insert.setNextParameter(10001);
076 *     insert.setNextParameter("goodbye");
077 *     insert.setNextParameter("rob");
078 *     insert.addBatch();
079 *
080 *     insert.setNextParameter(10002);
081 *     insert.setNextParameter("chow");
082 *     insert.setNextParameter("bob");
083 *     insert.addBatch();
084 *
085 *     int[] rows = insert.executeBatch();
086 *
087 *     txn.commit();
088 *   }
089 *
090 * }</pre>
091 *
092 * @see Update
093 * @see SqlQuery
094 * @see CallableSql
095 */
096public interface SqlUpdate {
097
098  /**
099   * Execute the update returning the number of rows modified.
100   * <p>
101   * After you have executed the SqlUpdate you can bind new variables using
102   * {@link #setParameter(String, Object)} etc and then execute the SqlUpdate
103   * again.
104   * </p>
105   * <p>
106   * For JDBC batch processing refer to
107   * {@link Transaction#setBatchMode(boolean)} and
108   * {@link Transaction#setBatchSize(int)}.
109   * </p>
110   *
111   * @see Ebean#execute(SqlUpdate)
112   */
113  int execute();
114
115  /**
116   * Execute when addBatch() has been used to batch multiple bind executions.
117   *
118   * @return The row counts for each of the batched statements.
119   */
120  int[] executeBatch();
121
122  /**
123   * Add the statement to batch processing to then later execute via executeBatch().
124   */
125  void addBatch();
126
127  /**
128   * Return the generated key value.
129   */
130  Object getGeneratedKey();
131
132  /**
133   * Execute and return the generated key. This is effectively a short cut for:
134   * <p>
135   * <pre>{@code
136   *
137   *   sqlUpdate.execute();
138   *   Object key = sqlUpdate.getGeneratedKey();
139   *
140   * }</pre>
141   *
142   * @return The generated key value
143   */
144  Object executeGetKey();
145
146  /**
147   * Return true if eBean should automatically deduce the table modification
148   * information and process it.
149   * <p>
150   * If this is true then cache invalidation and text index management are aware
151   * of the modification.
152   * </p>
153   */
154  boolean isAutoTableMod();
155
156  /**
157   * Set this to false if you don't want eBean to automatically deduce the table
158   * modification information and process it.
159   * <p>
160   * Set this to false if you don't want any cache invalidation or text index
161   * management to occur. You may do this when say you update only one column
162   * and you know that it is not important for cached objects or text indexes.
163   * </p>
164   */
165  SqlUpdate setAutoTableMod(boolean isAutoTableMod);
166
167  /**
168   * Return the label that can be seen in the transaction logs.
169   */
170  String getLabel();
171
172  /**
173   * Set a descriptive text that can be put into the transaction log.
174   * <p>
175   * Useful when identifying the statement in the transaction log.
176   * </p>
177   */
178  SqlUpdate setLabel(String label);
179
180  /**
181   * Set to true when we want to use getGeneratedKeys with this statement.
182   */
183  SqlUpdate setGetGeneratedKeys(boolean getGeneratedKeys);
184
185  /**
186   * Return the sql statement.
187   */
188  String getSql();
189
190  /**
191   * Return the generated sql that has named parameters converted to positioned parameters.
192   */
193  String getGeneratedSql();
194
195  /**
196   * Return the timeout used to execute this statement.
197   */
198  int getTimeout();
199
200  /**
201   * Set the timeout in seconds. Zero implies no limit.
202   * <p>
203   * This will set the query timeout on the underlying PreparedStatement. If the
204   * timeout expires a SQLException will be throw and wrapped in a
205   * PersistenceException.
206   * </p>
207   */
208  SqlUpdate setTimeout(int secs);
209
210  /**
211   * Set the next positioned parameter.
212   *
213   * @param value The value to bind
214   */
215  SqlUpdate setNextParameter(Object value);
216
217  /**
218   * Set a parameter via its index position.
219   */
220  SqlUpdate setParameter(int position, Object value);
221
222  /**
223   * Set a null parameter via its index position.
224   */
225  SqlUpdate setNull(int position, int jdbcType);
226
227  /**
228   * Set a null valued parameter using its index position.
229   */
230  SqlUpdate setNullParameter(int position, int jdbcType);
231
232  /**
233   * Set a named parameter value.
234   */
235  SqlUpdate setParameter(String name, Object param);
236
237  /**
238   * Set a named parameter that has a null value. Exactly the same as
239   * {@link #setNullParameter(String, int)}.
240   */
241  SqlUpdate setNull(String name, int jdbcType);
242
243  /**
244   * Set a named parameter that has a null value.
245   */
246  SqlUpdate setNullParameter(String name, int jdbcType);
247
248}
