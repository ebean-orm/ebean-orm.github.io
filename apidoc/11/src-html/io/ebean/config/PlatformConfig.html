184  /**
185   * Add a custom type mapping.
186   * <p>
187   * <pre>{@code
188   *
189   *   // set the default mapping for BigDecimal.class/decimal
190   *   serverConfig.addCustomMapping(DbType.DECIMAL, "decimal(18,6)");
191   *
192   *   // set the default mapping for String.class/varchar but only for Postgres
193   *   serverConfig.addCustomMapping(DbType.VARCHAR, "text", Platform.POSTGRES);
194   *
195   * }</pre>
196   *
197   * @param type             The DB type this mapping should apply to
198   * @param columnDefinition The column definition that should be used
199   * @param platform         Optionally specify the platform this mapping should apply to.
200   */
201  public void addCustomMapping(DbType type, String columnDefinition, Platform platform) {
202    customDbTypeMappings.add(new CustomDbTypeMapping(type, columnDefinition, platform));
203  }
204
205  /**
206   * Add a custom type mapping that applies to all platforms.
207   * <p>
208   * <pre>{@code
209   *
210   *   // set the default mapping for BigDecimal/decimal
211   *   serverConfig.addCustomMapping(DbType.DECIMAL, "decimal(18,6)");
212   *
213   *   // set the default mapping for String/varchar
214   *   serverConfig.addCustomMapping(DbType.VARCHAR, "text");
215   *
216   * }</pre>
217   *
218   * @param type             The DB type this mapping should apply to
219   * @param columnDefinition The column definition that should be used
220   */
221  public void addCustomMapping(DbType type, String columnDefinition) {
222    customDbTypeMappings.add(new CustomDbTypeMapping(type, columnDefinition));
223  }
224

232  public void loadSettings(PropertiesWrapper p) {
233
234    idType = p.getEnum(IdType.class, "idType", idType);
235    databaseSequenceBatchSize = p.getInt("databaseSequenceBatchSize", databaseSequenceBatchSize);
236    databaseBooleanTrue = p.get("databaseBooleanTrue", databaseBooleanTrue);
237    databaseBooleanFalse = p.get("databaseBooleanFalse", databaseBooleanFalse);
238
239    DbUuid dbUuid = p.getEnum(DbUuid.class, "dbuuid", null);
240    if (dbUuid != null) {
241      setDbUuid(dbUuid);
242    }
243    if (p.getBoolean("uuidStoreAsBinary", false)) {
244      setDbUuid(DbUuid.BINARY);
245    }
246
247    int srid = p.getInt("geometrySRID", 0);
248    if (srid > 0) {
249      setGeometrySRID(srid);
250    }
251
252    // Mapping is specified in the form: BOOLEAN=int(1);BIT=int(1);
253    String mapping = p.get("mapping");
254    if (mapping != null && !mapping.isEmpty()) {
255      Map<String, String> map = StringHelper.delimitedToMap(mapping, ";", "=");
256      for (Entry<String, String> entry : map.entrySet()) {
257        addCustomMapping(DbType.valueOf(entry.getKey()), entry.getValue());
258      }
259    }
260
261    boolean quotedIdentifiers = p.getBoolean("allQuotedIdentifiers", allQuotedIdentifiers);
262    if (quotedIdentifiers != allQuotedIdentifiers) {
263      // potentially also set to use matching naming convention
264      setAllQuotedIdentifiers(quotedIdentifiers);
265    }
266  }

268  /**
269   * Specify how UUID is stored.
270   */
271  public enum DbUuid {
272
273
274    /**
275     * Store using native UUID in H2 and Postgres and otherwise fallback to VARCHAR(40).
276     */
277    AUTO_VARCHAR(true, false, false),
278
279    /**
280     * Store using native UUID in H2 and Postgres and otherwise fallback to BINARY(16).
281     */
282    AUTO_BINARY(true, true, false),
283
284    /**
285     * Store using native UUID in H2 and Postgres and otherwise fallback to BINARY(16) with optimized packing.
286     */
287    AUTO_BINARY_OPTIMIZED(true, true, true),
288
289    /**
290     * Store using DB VARCHAR(40).
291     */
292    VARCHAR(false, false, false),
293
294    /**
295     * Store using DB BINARY(16).
296     */
297    BINARY(false, true, false),
298
299    /**
300     * Store using DB BINARY(16).
301     */
302    BINARY_OPTIMIZED(false, true, true);
303
304    boolean nativeType;
305    boolean binary;
306    boolean binaryOptimized;
307
308    DbUuid(boolean nativeType, boolean binary, boolean binaryOptimized) {
309      this.nativeType = nativeType;
310      this.binary = binary;
311      this.binaryOptimized = binaryOptimized;
312    }
313
314    /**
315     * Return true if native UUID type is preferred.
316     */
317    public boolean useNativeType() {
318      return nativeType;
319    }
320
321    /**
322     * Return true if BINARY(16) storage is preferred over VARCHAR(40).
323     */
324    public boolean useBinary() {
325      return binary;
326    }
327
328    /**
329     * Return true, if optimized packing should be used.
330     */
331    public boolean useBinaryOptimized() {
332      return binaryOptimized;
333    }
334  }
