001package io.ebean.config.dbplatform.postgres;
002
003import io.ebean.BackgroundExecutor;
004import io.ebean.Query;
005import io.ebean.annotation.Platform;
006import io.ebean.config.dbplatform.DatabasePlatform;
007import io.ebean.config.dbplatform.DbPlatformType;
008import io.ebean.config.dbplatform.DbType;
009import io.ebean.config.dbplatform.IdType;
010import io.ebean.config.dbplatform.PlatformIdGenerator;
011import io.ebean.config.dbplatform.SqlErrorCodes;
012
013import javax.sql.DataSource;
014import java.sql.Types;
015
016/**
017 * Postgres v9 specific platform.
018 * <p>
019 * Uses serial types and getGeneratedKeys.
020 * </p>
021 */
022public class PostgresPlatform extends DatabasePlatform {
023
024  public PostgresPlatform() {
025    super();
026    this.platform = Platform.POSTGRES;
027    this.supportsNativeIlike = true;
028    this.selectCountWithAlias = true;
029    this.blobDbType = Types.LONGVARBINARY;
030    this.clobDbType = Types.VARCHAR;
031    this.nativeUuidType = true;
032    this.columnAliasPrefix = null;
033
034    this.dbEncrypt = new PostgresDbEncrypt();
035    this.historySupport = new PostgresHistorySupport();
036
037    // Use Identity and getGeneratedKeys
038    this.dbIdentity.setIdType(IdType.IDENTITY);
039    this.dbIdentity.setSupportsGetGeneratedKeys(true);
040    this.dbIdentity.setSupportsSequence(true);
041
042    this.dbDefaultValue.setNow("current_timestamp");
043
044    this.exceptionTranslator =
045      new SqlErrorCodes()
046        .addAcquireLock("55P03")
047        .addDuplicateKey("23505")
048        .addDataIntegrity("23000","23502","23503","23514")
049        .build();
050
051    this.openQuote = "\"";
052    this.closeQuote = "\"";
053
054    DbPlatformType dbTypeText = new DbPlatformType("text", false);
055    DbPlatformType dbBytea = new DbPlatformType("bytea", false);
056
057    dbTypeMap.put(DbType.UUID, new DbPlatformType("uuid", false));
058    dbTypeMap.put(DbType.HSTORE, new DbPlatformType("hstore", false));
059    dbTypeMap.put(DbType.JSON, new DbPlatformType("json", false));
060    dbTypeMap.put(DbType.JSONB, new DbPlatformType("jsonb", false));
061
062    dbTypeMap.put(DbType.INTEGER, new DbPlatformType("integer", false));
063    dbTypeMap.put(DbType.DOUBLE, new DbPlatformType("float"));
064    dbTypeMap.put(DbType.TINYINT, new DbPlatformType("smallint"));
065    dbTypeMap.put(DbType.DECIMAL, new DbPlatformType("decimal", 38));
066    dbTypeMap.put(DbType.TIMESTAMP, new DbPlatformType("timestamptz"));
067
068    dbTypeMap.put(DbType.BINARY, dbBytea);
069    dbTypeMap.put(DbType.VARBINARY, dbBytea);
070
071    dbTypeMap.put(DbType.BLOB, dbBytea);
072    dbTypeMap.put(DbType.CLOB, dbTypeText);
073    dbTypeMap.put(DbType.LONGVARBINARY, dbBytea);
074    dbTypeMap.put(DbType.LONGVARCHAR, dbTypeText);
075  }
076
077  @Override
078  protected void addGeoTypes(int srid) {
079    dbTypeMap.put(DbType.POINT, geoType("point", srid));
080    dbTypeMap.put(DbType.POLYGON, geoType("polygon", srid));
081    dbTypeMap.put(DbType.LINESTRING, geoType("linestring", srid));
082    dbTypeMap.put(DbType.MULTIPOINT, geoType("multipoint", srid));
083    dbTypeMap.put(DbType.MULTILINESTRING, geoType("multilinestring", srid));
084    dbTypeMap.put(DbType.MULTIPOLYGON, geoType("multipolygon", srid));
085  }
086
087  private DbPlatformType geoType(String type, int srid) {
088    return new DbPlatformType("geometry(" + type + "," + srid + ")");
089  }
090
091  /**
092   * So we can generate varchar[], int[], uuid[] column definitions and use the associated scalar types.
093   */
094  @Override
095  public boolean isNativeArrayType() {
096    return true;
097  }
098
099  /**
100   * Create a Postgres specific sequence IdGenerator.
101   */
102  @Override
103  public PlatformIdGenerator createSequenceIdGenerator(BackgroundExecutor be, DataSource ds, int stepSize, String seqName) {
104
105    return new PostgresSequenceIdGenerator(be, ds, seqName, sequenceBatchSize);
106  }
107
108  @Override
109  protected String withForUpdate(String sql, Query.ForUpdate forUpdateMode) {
110    switch (forUpdateMode) {
111      case SKIPLOCKED:
112        return sql + " for update skip locked";
113      case NOWAIT:
114        return sql + " for update nowait";
115      default:
116        return sql + " for update";
117    }
118  }
119}
