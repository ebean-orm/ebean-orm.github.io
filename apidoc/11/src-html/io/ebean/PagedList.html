001package io.ebean;
002
003import javax.annotation.Nonnull;
004import java.util.List;
005import java.util.concurrent.Future;
006
007/**
008 * Represents a page of results.
009 * <p>
010 * The benefit of using PagedList over just using the normal Query with
011 * {@link Query#setFirstRow(int)} and {@link Query#setMaxRows(int)} is that it additionally wraps
012 * functionality that can call {@link Query#findFutureCount()} to determine total row count,
013 * total page count etc.
014 * </p>
015 * <p>
016 * Internally this works using {@link Query#setFirstRow(int)} and {@link Query#setMaxRows(int)} on
017 * the query. This translates into SQL that uses limit offset, rownum or row_number function to
018 * limit the result set.
019 * </p>
020 * <p>
021 * <h4>Example: typical use including total row count</h4>
022 * <pre>{@code
023 *
024 *     // We want to find the first 50 new orders
025 *     //  ... so we don't really need setFirstRow(0)
026 *
027 *     PagedList<Order> pagedList
028 *       = ebeanServer.find(Order.class)
029 *       .where().eq("status", Order.Status.NEW)
030 *       .order().asc("id")
031 *       .setFirstRow(0)
032 *       .setMaxRows(50)
033 *       .findPagedList();
034 *
035 *     // Optional: initiate the loading of the total
036 *     // row count in a background thread
037 *     pagedList.loadRowCount();
038 *
039 *     // fetch and return the list in the foreground thread
040 *     List<Order> orders = pagedList.getList();
041 *
042 *     // get the total row count (from the future)
043 *     int totalRowCount = pagedList.getTotalRowCount();
044 *
045 * }</pre>
046 * <p>
047 * <h4>Example: No total row count required</h4>
048 * <pre>{@code
049 *
050 *     // If you are not getting the 'first page' often
051 *     // you do not bother getting the total row count again
052 *     // so instead just get the page list of data
053 *
054 *     // fetch and return the list in the foreground thread
055 *     List<Order> orders = pagedList.getList();
056 *
057 * }</pre>
058 *
059 * @param <T> the entity bean type
060 * @see Query#findPagedList()
061 */
062public interface PagedList<T> {
063
064  /**
065   * Initiate the loading of the total row count in the background.
066   * <pre>{@code
067   *
068   *     // initiate the loading of the total row count
069   *     // in a background thread
070   *     pagedList.loadRowCount();
071   *
072   *     // fetch and return the list in the foreground thread
073   *     List<Order> orders = pagedList.getList();
074   *
075   *     // get the total row count (from the future)
076   *     int totalRowCount = pagedList.getTotalRowCount();
077   *
078   * }</pre>
079   * <p>
080   * Also note that using loadRowCount() and getTotalRowCount() rather than getFutureRowCount()
081   * means that exceptions ExecutionException, InterruptedException, TimeoutException are instead
082   * wrapped in the unchecked PersistenceException (which might be preferrable).
083   * </p>
084   */
085  void loadCount();
086
087  /**
088   * Return the Future row count. You might get this if you wish to cancel the total row count query
089   * or specify a timeout for the row count query.
090   * <p>
091   * The loadRowCount() & getTotalRowCount() methods internally make use of this getFutureRowCount() method.
092   * Generally I expect people to prefer loadRowCount() & getTotalRowCount() over getFutureRowCount().
093   * </p>
094   * <pre>{@code
095   *
096   *     // initiate the row count query in the background thread
097   *     Future<Integer> rowCount = pagedList.getFutureRowCount();
098   *
099   *     // fetch and return the list in the foreground thread
100   *     List<Order> orders = pagedList.getList();
101   *
102   *     // now get the total count with a timeout
103   *     Integer totalRowCount = rowCount.get(30, TimeUnit.SECONDS);
104   *
105   *     // or ge the total count without a timeout
106   *     Integer totalRowCountViaFuture = rowCount.get();
107   *
108   *     // which is actually the same as ...
109   *     int totalRowCount = pagedList.getTotalRowCount();
110   *
111   * }</pre>
112   */
113  @Nonnull
114  Future<Integer> getFutureCount();
115
116  /**
117   * Return the list of entities for this page.
118   */
119  @Nonnull
120  List<T> getList();
121
122  /**
123   * Return the total row count for all pages.
124   * <p>
125   * If loadRowCount() has already been called then the row count query is already executing in a background thread
126   * and this gets the associated Future and gets the value waiting for the future to finish.
127   * </p>
128   * <p>
129   * If loadRowCount() has not been called then this executes the find row count query and returns the result and this
130   * will just occur in the current thread and not use a background thread.
131   * </p>
132   * <pre>{@code
133   *
134   *     // Optional: initiate the loading of the total
135   *     // row count in a background thread
136   *     pagedList.loadRowCount();
137   *
138   *     // fetch and return the list in the foreground thread
139   *     List<Order> orders = pagedList.getList();
140   *
141   *     // get the total row count (which was being executed
142   *     // in a background thread if loadRowCount() was used)
143   *     int totalRowCount = pagedList.getTotalRowCount();
144   *
145   * }</pre>
146   */
147  int getTotalCount();
148
149  /**
150   * Return the total number of pages based on the page size and total row count.
151   * <p>
152   * This method requires that the total row count has been fetched and will invoke
153   * the total row count query if it has not already been invoked.
154   * </p>
155   */
156  int getTotalPageCount();
157
158  /**
159   * Return the page size used for this query. This is the same value as maxRows used by the query.
160   */
161  int getPageSize();
162
163  /**
164   * Return the index position of this page (Zero based).
165   * <p>
166   * This is a calculated value based on firstRow/maxRows.
167   * </p>
168   */
169  int getPageIndex();
170
171  /**
172   * Return true if there is a next page.
173   * <p>
174   * This method requires that the total row count has been fetched and will invoke
175   * the total row count query if it has not already been invoked.
176   * </p>
177   */
178  boolean hasNext();
179
180  /**
181   * Return true if there is a previous page.
182   */
183  boolean hasPrev();
184
185  /**
186   * Helper method to return a "X to Y of Z" string for this page where X is the first row, Y the
187   * last row and Z the total row count.
188   * <p>
189   * This method requires that the total row count has been fetched and will invoke
190   * the total row count query if it has not already been invoked.
191   * </p>
192   *
193   * @param to String to put between the first and last row
194   * @param of String to put between the last row and the total row count
195   * @return String of the format XtoYofZ.
196   */
197  String getDisplayXtoYofZ(String to, String of);
198}
