001package io.ebean.plugin;
002
003import io.ebean.Query;
004import io.ebean.config.dbplatform.IdType;
005import io.ebean.event.BeanFindController;
006import io.ebean.event.BeanPersistController;
007import io.ebean.event.BeanPersistListener;
008import io.ebean.event.BeanQueryAdapter;
009import io.ebeanservice.docstore.api.mapping.DocumentMapping;
010
011import java.util.Collection;
012
013/**
014 * Information and methods on BeanDescriptors made available to plugins.
015 */
016public interface BeanType<T> {
017
018  /**
019   * Return the short name of the bean type.
020   */
021  String getName();
022
023  /**
024   * Return the profileId of the bean type.
025   */
026  short getProfileId();
027
028  /**
029   * Return the full name of the bean type.
030   */
031  String getFullName();
032
033  /**
034   * Return the class type this BeanDescriptor describes.
035   */
036  Class<T> getBeanType();
037
038  /**
039   * Return the type bean for an OneToMany or ManyToOne or ManyToMany property.
040   */
041  BeanType<?> getBeanTypeAtPath(String propertyName);
042
043  /**
044   * Return all the properties for this bean type.
045   */
046  Collection<? extends Property> allProperties();
047
048  /**
049   * Return the Id property.
050   */
051  Property getIdProperty();
052
053  /**
054   * Return the when modified property if there is one defined.
055   */
056  Property getWhenModifiedProperty();
057
058  /**
059   * Return the when created property if there is one defined.
060   */
061  Property getWhenCreatedProperty();
062
063  /**
064   * Return the Property to read values from a bean.
065   */
066  Property getProperty(String propertyName);
067
068  /**
069   * Return the ExpressionPath for a given property path.
070   * <p>
071   * This can return a property or nested property path.
072   * </p>
073   */
074  ExpressionPath getExpressionPath(String path);
075
076  /**
077   * Return true if the property is a valid known property or path for the given bean type.
078   */
079  boolean isValidExpression(String property);
080
081  /**
082   * Return true if bean caching is on for this bean type.
083   */
084  boolean isBeanCaching();
085
086  /**
087   * Return true if query caching is on for this bean type.
088   */
089  boolean isQueryCaching();
090
091  /**
092   * Clear the bean cache.
093   */
094  void clearBeanCache();
095
096  /**
097   * Clear the query cache.
098   */
099  void clearQueryCache();
100
101  /**
102   * Return true if the type is document store only.
103   */
104  boolean isDocStoreOnly();
105
106  /**
107   * Return the base table this bean type maps to.
108   */
109  String getBaseTable();
110
111  /**
112   * Create a new instance of the bean.
113   */
114  T createBean();
115
116  /**
117   * Return the bean id. This is the same as getBeanId() but without the generic type.
118   */
119  Object beanId(Object bean);
120
121  /**
122   * Return the id value for the given bean.
123   */
124  Object getBeanId(T bean);
125
126  /**
127   * Set the id value to the bean.
128   */
129  void setBeanId(T bean, Object idValue);
130
131  /**
132   * Return the bean persist controller.
133   */
134  BeanPersistController getPersistController();
135
136  /**
137   * Return the bean persist listener.
138   */
139  BeanPersistListener getPersistListener();
140
141  /**
142   * Return the beanFinder. Usually null unless overriding the finder.
143   */
144  BeanFindController getFindController();
145
146  /**
147   * Return the BeanQueryAdapter or null if none is defined.
148   */
149  BeanQueryAdapter getQueryAdapter();
150
151  /**
152   * Return the identity generation type.
153   */
154  IdType getIdType();
155
156  /**
157   * Return the sequence name associated to this entity bean type (if there is one).
158   */
159  String getSequenceName();
160
161  /**
162   * Return true if this bean type has doc store backing.
163   */
164  boolean isDocStoreMapped();
165
166  /**
167   * Return the DocumentMapping for this bean type.
168   * <p>
169   * This is the document structure and mapping options for how this bean type is mapped
170   * for the document store.
171   * </p>
172   */
173  DocumentMapping getDocMapping();
174
175  /**
176   * Return the doc store queueId for this bean type.
177   */
178  String getDocStoreQueueId();
179
180  /**
181   * Return the doc store support for this bean type.\
182   */
183  BeanDocType<T> docStore();
184
185  /**
186   * Add the discriminator value to the query if needed.
187   */
188  void addInheritanceWhere(Query<?> query);
189
190  /**
191   * Return the root bean type for an inheritance hierarchy.
192   */
193  BeanType<?> root();
194
195  /**
196   * Return true if this bean type has an inheritance hierarchy.
197   */
198  boolean hasInheritance();
199
200  /**
201   * Return the discriminator column.
202   */
203  String getDiscColumn();
204
205  /**
206   * Create a bean given the discriminator value.
207   */
208  T createBeanUsingDisc(Object discValue);
209}
