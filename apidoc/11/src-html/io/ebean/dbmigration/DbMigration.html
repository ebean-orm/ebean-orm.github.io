001package io.ebean.dbmigration;
002
003import io.ebean.EbeanServer;
004import io.ebean.annotation.Platform;
005import io.ebean.config.ServerConfig;
006import io.ebean.config.dbplatform.DatabasePlatform;
007
008import java.io.IOException;
009import java.util.Iterator;
010import java.util.ServiceLoader;
011
012/**
013 * Generates DDL migration scripts based on changes to the current model.
014 *
015 * <p>
016 * Typically this is run as a main method in src/test once a developer is happy
017 * with the next set of changes to the model.
018 * </p>
019 *
020 * <h3>Example: Run for a single specific platform</h3>
021 *
022 * <pre>{@code
023 *
024 *    // optionally specify the version and name
025 *    //System.setProperty("ddl.migration.version", "1.1");
026 *    //System.setProperty("ddl.migration.name", "add bars");
027 *
028 *    // generate a migration using drops from a prior version
029 *    //System.setProperty("ddl.migration.pendingDropsFor", "1.2");
030 *
031 *    DbMigration migration = DbMigration.create();
032 *
033 *    migration.setPlatform(Platform.POSTGRES);
034 *    migration.generateMigration();
035 *
036 * }</pre>
037 */
038public interface DbMigration {
039
040  /**
041   * Create a DbMigration implementation to use.
042   */
043  static DbMigration create() {
044
045    Iterator<DbMigration> loader = ServiceLoader.load(DbMigration.class).iterator();
046    if (loader.hasNext()) {
047      return loader.next();
048    }
049    throw new IllegalStateException("No service implementation found for DbMigration?");
050  }
051
052  /**
053   * Set the path from the current working directory to the application resources.
054   * <p>
055   * This defaults to maven style 'src/main/resources'.
056   */
057  void setPathToResources(String pathToResources);
058
059  /**
060   * Set the server to use to determine the current model.
061   * Typically this is not called explicitly.
062   */
063  void setServer(EbeanServer ebeanServer);
064
065  /**
066   * Set the serverConfig to use. Typically this is not called explicitly.
067   */
068  void setServerConfig(ServerConfig config);
069
070  /**
071   * Set the specific platform to generate DDL for.
072   * <p>
073   * If not set this defaults to the platform of the default server.
074   * </p>
075   */
076  void setPlatform(Platform platform);
077
078  /**
079   * Set the specific platform to generate DDL for.
080   * <p>
081   * If not set this defaults to the platform of the default server.
082   * </p>
083   */
084  void setPlatform(DatabasePlatform databasePlatform);
085
086  /**
087   * Set to false to turn off strict mode.
088   * <p>
089   * Strict mode checks that a column changed to non-null on an existing table via DB migration has a default
090   * value specified. Set this to false if that isn't the case but it is known that all the existing rows have
091   * a value specified (there are no existing null values for the column).
092   * </p>
093   */
094  void setStrictMode(boolean strictMode);
095
096  /**
097   * Set to true to include a generated header comment in the DDL script.
098   */
099  void setIncludeGeneratedFileComment(boolean includeGeneratedFileComment);
100
101  /**
102   * Set the header that is included in the generated DDL script.
103   */
104  void setHeader(String header);
105
106  /**
107   * Set the prefix for the version. Set this to "V" for use with Flyway.
108   */
109  void setApplyPrefix(String applyPrefix);
110
111  /**
112   * Set the version of the migration to be generated.
113   */
114  void setVersion(String version);
115
116  /**
117   * Set the name of the migration to be generated.
118   */
119  void setName(String name);
120
121  /**
122   * Generate a migration for the version specified that contains pending drops.
123   *
124   * @param generatePendingDrop The version of a prior migration that holds pending drops.
125   */
126  void setGeneratePendingDrop(String generatePendingDrop);
127
128  /**
129   * Add an additional platform to write the migration DDL.
130   * <p>
131   * Use this when you want to generate sql scripts for multiple database platforms
132   * from the migration (e.g. generate migration sql for MySql, Postgres and Oracle).
133   * </p>
134   */
135  void addPlatform(Platform platform, String prefix);
136
137  /**
138   * Add an additional databasePlatform to write the migration DDL.
139   * <p>
140   * Use this when you want to add preconfigured database platforms.
141   * </p>
142   */
143  void addDatabasePlatform(DatabasePlatform databasePlatform, String prefix);
144
145  /**
146   * Generate the next migration xml file and associated apply and rollback sql scripts.
147   * <p>
148   * This does not run the migration or ddl scripts but just generates them.
149   * </p>
150   * <h3>Example: Run for a single specific platform</h3>
151   * <pre>{@code
152   *
153   *   DbMigration migration = DbMigration.create();
154   *   migration.setPlatform(Platform.POSTGRES);
155   *
156   *   migration.generateMigration();
157   *
158   * }</pre>
159   * <p>
160   * <h3>Example: Run migration generating DDL for multiple platforms</h3>
161   * <pre>{@code
162   *
163   *   DbMigration migration = DbMigration.create();
164   *
165   *   migration.setPathToResources("src/main/resources");
166   *
167   *   migration.addPlatform(Platform.POSTGRES, "pg");
168   *   migration.addPlatform(Platform.MYSQL, "mysql");
169   *   migration.addPlatform(Platform.ORACLE, "oracle");
170   *
171   *   migration.generateMigration();
172   *
173   * }</pre>
174   *
175   * @return the version of the generated migration or null
176   */
177  String generateMigration() throws IOException;
178}
