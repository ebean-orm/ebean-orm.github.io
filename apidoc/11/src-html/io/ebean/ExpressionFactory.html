012/**
013 * Expression factory for creating standard expressions.
014 * <p>
015 * Creates standard common expressions for using in a Query Where or Having
016 * clause.
017 * </p>
018 * <p>
019 * You will often not use this class directly but instead just add expressions
020 * via the methods on ExpressionList such as
021 * {@link ExpressionList#gt(String, Object)}.
022 * </p>
023 * <p>
024 * The ExpressionList is returned from {@link Query#where()}.
025 * </p>
026 * <pre>{@code
027 * // Example: fetch orders where status equals new or orderDate > lastWeek.
028 *
029 * Expression newOrLastWeek =
030 *     Expr.or(Expr.eq("status", Order.Status.NEW),
031 *             Expr.gt("orderDate", lastWeek));
032 *
033 * Query<Order> query = Ebean.createQuery(Order.class);
034 * query.where().add(newOrLastWeek);
035 * List<Order> list = query.findList();
036 * ...
037 * }</pre>
038 *
039 * @see Query#where()
040 */
041public interface ExpressionFactory {
042

053  /**

058  /**

075   */

082

084   * Between - for the given path in a JSON document.
085   */
086  Expression jsonBetween(String propertyName, String path, Object lowerValue, Object upperValue);

104  /**

106   * <p>

112  /**

116   * </p>
117   */
118  Expression arrayIsNotEmpty(String propertyName);
119

122   */
123  Expression eq(String propertyName, Object value);
124

127   */
128  Expression ne(String propertyName, Object value);
129

155

157   * Greater Than or Equal to - property greater than or equal to the given
158   * value.
159   */
160  Expression ge(String propertyName, Object value);
161
162  /**
163   * Less Than - property less than the given value.
164   */
165  Expression lt(String propertyName, Object value);
166
167  /**
168   * Less Than or Equal to - property less than or equal to the given value.
169   */
170  Expression le(String propertyName, Object value);
171
172  /**

174   */
175  Expression isNull(String propertyName);

179   */

183   * Case insensitive {@link #exampleLike(Object)}
184   */
185  ExampleExpression iexampleLike(Object example);

187  /**

190   */
191  ExampleExpression exampleLike(Object example);
192
193  /**
194   * Create the query by Example expression specifying more options.
195   */
196  ExampleExpression exampleLike(Object example, boolean caseInsensitive, LikeType likeType);
197

199   * Like with support for named parameters.
200   */
201  Expression like(String propertyName, Object value, boolean caseInsensitive, LikeType likeType);

208

213   */
214  Expression ilike(String propertyName, String value);

239   * Contains - property like %value%.

246   */

282  Expression notIn(String propertyName, Query<?> subQuery);

426

428   * Return a list of expressions that will be joined by AND's.
429   */

434   */
435  <T> Junction<T> disjunction(Query<T> query, ExpressionList<T> parent);

437  /**

443  <T> Junction<T> junction(Junction.Type type, Query<T> query, ExpressionList<T> parent);
