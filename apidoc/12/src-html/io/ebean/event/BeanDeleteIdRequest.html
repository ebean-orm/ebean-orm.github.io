001package io.ebean.event;
002
003import io.ebean.EbeanServer;
004import io.ebean.Transaction;
005
006/**
007 * A request to delete a bean by Id value.
008 */
009public interface BeanDeleteIdRequest {
010
011  /**
012   * Return the server processing the request.
013   */
014  EbeanServer getEbeanServer();
015
016  /**
017   * Return the Transaction associated with this request.
018   */
019  Transaction getTransaction();
020
021  /**
022   * Returns the Id value of the bean being deleted.
023   */
024  Object getId();
025
026}
