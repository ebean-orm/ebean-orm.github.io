001package io.ebean.meta;
002
003/**
004 * The type of Metric.
005 */
006public enum MetricType {
007
008  /**
009   * Transactions.
010   */
011  TXN,
012
013  /**
014   * ORM Insert Update or Delete.
015   */
016  IUD,
017
018  /**
019   * ORM queries.
020   */
021  ORM,
022
023  /**
024   * DTO queries.
025   */
026  DTO,
027
028  /**
029   * SQL queries with a label will have metrics collected.
030   * <p>
031   * SqlQuery and SqlUpdate without a label have no metrics collected.
032   */
033  SQL,
034
035  /**
036   * L2 cache metrics.
037   */
038  L2
039
040}
