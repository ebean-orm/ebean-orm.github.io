001package io.ebean.meta;
002
003/**
004 * An individual metric.
005 */
006public class MetricData {
007
008  private String name;
009  private String type;
010  private String hash;
011  private String loc;
012  private String sql;
013
014  private Long count;
015  private Long mean;
016  private Long max;
017  private Long total;
018
019  public MetricData(String name, String type) {
020    this.name = name;
021    this.type = type;
022  }
023
024  public MetricData() {
025  }
026
027  public String getName() {
028    return name;
029  }
030
031  public void setName(String name) {
032    this.name = name;
033  }
034
035  public String getType() {
036    return type;
037  }
038
039  public void setType(String type) {
040    this.type = type;
041  }
042
043  public String getHash() {
044    return hash;
045  }
046
047  public void setHash(String hash) {
048    this.hash = hash;
049  }
050
051  public String getLoc() {
052    return loc;
053  }
054
055  public void setLoc(String loc) {
056    this.loc = loc;
057  }
058
059  public String getSql() {
060    return sql;
061  }
062
063  public void setSql(String sql) {
064    this.sql = sql;
065  }
066
067  public Long getCount() {
068    return count;
069  }
070
071  public void setCount(Long count) {
072    this.count = count;
073  }
074
075  public Long getMean() {
076    return mean;
077  }
078
079  public void setMean(Long mean) {
080    this.mean = mean;
081  }
082
083  public Long getMax() {
084    return max;
085  }
086
087  public void setMax(Long max) {
088    this.max = max;
089  }
090
091  public Long getTotal() {
092    return total;
093  }
094
095  public void setTotal(Long total) {
096    this.total = total;
097  }
098}
