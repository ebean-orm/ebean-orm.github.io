001package io.ebean.dbmigration;
002
003import io.ebean.Database;
004import io.ebean.annotation.Platform;
005import io.ebean.config.ServerConfig;
006import io.ebean.config.dbplatform.DatabasePlatform;
007
008import java.io.IOException;
009import java.util.Iterator;
010import java.util.List;
011import java.util.ServiceLoader;
012
013/**
014 * Generates DDL migration scripts based on changes to the current model.
015 *
016 * <p>
017 * Typically this is run as a main method in src/test once a developer is happy
018 * with the next set of changes to the model.
019 * </p>
020 *
021 * <h3>Example: Run for a single specific platform</h3>
022 *
023 * <pre>{@code
024 *
025 *    // optionally specify the version and name
026 *    //System.setProperty("ddl.migration.version", "1.1");
027 *    //System.setProperty("ddl.migration.name", "add bars");
028 *
029 *    // generate a migration using drops from a prior version
030 *    //System.setProperty("ddl.migration.pendingDropsFor", "1.2");
031 *
032 *    DbMigration migration = DbMigration.create();
033 *
034 *    migration.setPlatform(Platform.POSTGRES);
035 *    migration.generateMigration();
036 *
037 * }</pre>
038 */
039public interface DbMigration {
040
041  /**
042   * Create a DbMigration implementation to use.
043   */
044  static DbMigration create() {
045
046    Iterator<DbMigration> loader = ServiceLoader.load(DbMigration.class).iterator();
047    if (loader.hasNext()) {
048      return loader.next();
049    }
050    throw new IllegalStateException("No service implementation found for DbMigration?");
051  }
052
053  /**
054   * Set to false to suppress logging to System out.
055   */
056  void setLogToSystemOut(boolean logToSystemOut);
057
058  /**
059   * Set the path from the current working directory to the application resources.
060   * <p>
061   * This defaults to maven style 'src/main/resources'.
062   */
063  void setPathToResources(String pathToResources);
064
065  /**
066   * Set the path where migrations are generated to (which defaults to "dbmigration").
067   * <p>
068   * Normally we only use this when we use Ebean to generate the database migrations
069   * and then use some other tool like FlywayDB to run the migrations.
070   * <p>
071   * Example: with <code>setMigrationPath("db/migration")</code> ... the migrations
072   * are generated into <code>src/resources/db/migration</code>.
073   * <p>
074   * Note that if Ebean migration runner is used we should not use this method but
075   * instead set the migrationPath via a property such that both the migration generator
076   * and migration runner both use the same path.
077   *
078   * @param migrationPath The path that migrations are generated into.
079   */
080  void setMigrationPath(String migrationPath);
081
082  /**
083   * Set the server to use to determine the current model.
084   * Typically this is not called explicitly.
085   */
086  void setServer(Database database);
087
088  /**
089   * Set the serverConfig to use. Typically this is not called explicitly.
090   */
091  void setServerConfig(ServerConfig config);
092
093  /**
094   * Set the specific platform to generate DDL for.
095   * <p>
096   * If not set this defaults to the platform of the default database.
097   * </p>
098   */
099  void setPlatform(Platform platform);
100
101  /**
102   * Set the specific platform to generate DDL for.
103   * <p>
104   * If not set this defaults to the platform of the default database.
105   * </p>
106   */
107  void setPlatform(DatabasePlatform databasePlatform);
108
109  /**
110   * Set to false to turn off strict mode.
111   * <p>
112   * Strict mode checks that a column changed to non-null on an existing table via DB migration has a default
113   * value specified. Set this to false if that isn't the case but it is known that all the existing rows have
114   * a value specified (there are no existing null values for the column).
115   * </p>
116   */
117  void setStrictMode(boolean strictMode);
118
119  /**
120   * Set to true to include a generated header comment in the DDL script.
121   */
122  void setIncludeGeneratedFileComment(boolean includeGeneratedFileComment);
123
124  /**
125   * Set this to false to exclude the builtin support for table partitioning (with @DbPartition).
126   */
127  void setIncludeBuiltInPartitioning(boolean includeBuiltInPartitioning);
128
129  /**
130   * Set the header that is included in the generated DDL script.
131   */
132  void setHeader(String header);
133
134  /**
135   * Set the prefix for the version. Set this to "V" for use with Flyway.
136   */
137  void setApplyPrefix(String applyPrefix);
138
139  /**
140   * Set the version of the migration to be generated.
141   */
142  void setVersion(String version);
143
144  /**
145   * Set the name of the migration to be generated.
146   */
147  void setName(String name);
148
149  /**
150   * Generate a migration for the version specified that contains pending drops.
151   *
152   * @param generatePendingDrop The version of a prior migration that holds pending drops.
153   */
154  void setGeneratePendingDrop(String generatePendingDrop);
155
156  /**
157   * Set to true if ALTER TABLE ADD FOREIGN KEY should be generated with an option to skip validation.
158   * <p>
159   * Currently this is only useful for Postgres DDL adding the <code>NOT VALID</code> option.
160   * </p>
161   */
162  void setAddForeignKeySkipCheck(boolean addForeignKeySkipCheck);
163
164  /**
165   * Set the lock timeout to be included with the DDL generation.
166   * <p>
167   * Currently this is only useful for Postgres migrations adding a <code>set lock_timeout</code>
168   * statement to the generated database migration.
169   * </p>
170   */
171  void setLockTimeout(int seconds);
172
173  /**
174   * Add an additional platform to write the migration DDL.
175   * <p>
176   * Use this when you want to generate sql scripts for multiple database platforms
177   * from the migration (e.g. generate migration sql for MySql, Postgres and Oracle).
178   * </p>
179   */
180  void addPlatform(Platform platform, String prefix);
181
182  /**
183   * Add an additional databasePlatform to write the migration DDL.
184   * <p>
185   * Use this when you want to add preconfigured database platforms.
186   * </p>
187   */
188  void addDatabasePlatform(DatabasePlatform databasePlatform, String prefix);
189
190  /**
191   * Return the list of versions that contain pending drops.
192   */
193  List<String> getPendingDrops();
194
195  /**
196   * Generate the next migration sql script and associated model xml.
197   * <p>
198   * This does not run the migration or ddl scripts but just generates them.
199   * </p>
200   * <h3>Example: Run for a single specific platform</h3>
201   * <pre>{@code
202   *
203   *   DbMigration migration = DbMigration.create();
204   *   migration.setPlatform(Platform.POSTGRES);
205   *
206   *   migration.generateMigration();
207   *
208   * }</pre>
209   * <p>
210   * <h3>Example: Run migration generating DDL for multiple platforms</h3>
211   * <pre>{@code
212   *
213   *   DbMigration migration = DbMigration.create();
214   *
215   *   migration.setPathToResources("src/main/resources");
216   *
217   *   migration.addPlatform(Platform.POSTGRES, "pg");
218   *   migration.addPlatform(Platform.MYSQL, "mysql");
219   *   migration.addPlatform(Platform.ORACLE, "oracle");
220   *
221   *   migration.generateMigration();
222   *
223   * }</pre>
224   *
225   * @return the version of the generated migration or null
226   */
227  String generateMigration() throws IOException;
228
229  /**
230   * Generate an "init" migration which has all changes.
231   * <p>
232   * An "init" migration can only be executed and used on a database that has had no
233   * prior migrations run on it.
234   * </p>
235   *
236   * @return the version of the generated migration
237   */
238  String generateInitMigration() throws IOException;
239
240}
