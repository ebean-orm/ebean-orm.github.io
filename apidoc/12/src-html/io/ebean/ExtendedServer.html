001package io.ebean;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005import javax.persistence.NonUniqueResultException;
006import java.time.Clock;
007import java.util.List;
008import java.util.Map;
009import java.util.Optional;
010import java.util.Set;
011import java.util.function.Consumer;
012import java.util.function.Predicate;
013import java.util.stream.Stream;
014
015/**
016 * The extended API for Database.
017 * <p>
018 * This provides the finder methods that take an explicit transaction rather than obtaining
019 * the transaction from the usual mechanism (which is ThreadLocal based).
020 * </p>
021 * <p>
022 * In general we only want to use this ExtendedServer API when we want to avoid / bypass
023 * the use of the mechanism to get the current transaction and instead explicitly supply
024 * the transaction to use.
025 * </p>
026 * <p>
027 * Note that in all cases the transaction supplied can be null and in this case the Database
028 * will use the normal mechanism to obtain the transaction to use.
029 * </p>
030 */
031public interface ExtendedServer {
032
033  /**
034   * Return the NOW time from the Clock.
035   */
036  long clockNow();
037
038  /**
039   * Set the Clock to use for <code>@WhenCreated</code> and <code>@WhenModified</code>.
040   * <p>
041   * Note that we only expect to change the Clock for testing purposes.
042   * </p>
043   */
044  void setClock(Clock clock);
045
046  /**
047   * Execute the query returning true if a row is found.
048   * <p>
049   * The query is executed using max rows of 1 and will only select the id property.
050   * This method is really just a convenient way to optimise a query to perform a
051   * 'does a row exist in the db' check.
052   * </p>
053   *
054   * <h2>Example:</h2>
055   * <pre>{@code
056   *
057   *   boolean userExists = query().where().eq("email", "rob@foo.com").exists();
058   *
059   * }</pre>
060   *
061   * <h2>Example using a query bean:</h2>
062   * <pre>{@code
063   *
064   *   boolean userExists = new QContact().email.equalTo("rob@foo.com").exists();
065   *
066   * }</pre>
067   *
068   * @return True if the query finds a matching row in the database
069   */
070  <T> boolean exists(Query<?> ormQuery, Transaction transaction);
071
072  /**
073   * Return the number of 'top level' or 'root' entities this query should return.
074   *
075   * @see Query#findCount()
076   * @see Query#findFutureCount()
077   */
078  <T> int findCount(Query<T> query, Transaction transaction);
079
080  /**
081   * Return the Id values of the query as a List.
082   *
083   * @see Query#findIds()
084   */
085  @Nonnull
086  <A, T> List<A> findIds(Query<T> query, Transaction transaction);
087
088  /**
089   * Return a QueryIterator for the query.
090   * <p>
091   * Generally using {@link #findEach(Query, Consumer, Transaction)} or
092   * {@link #findEachWhile(Query, Predicate, Transaction)} is preferred
093   * to findIterate(). The reason is that those methods automatically take care of
094   * closing the queryIterator (and the underlying jdbc statement and resultSet).
095   * </p>
096   * <p>
097   * This is similar to findEach in that not all the result beans need to be held
098   * in memory at the same time and as such is good for processing large queries.
099   * </p>
100   *
101   * @see Query#findIterate()
102   * @see Query#findEach(Consumer)
103   * @see Query#findEachWhile(Predicate)
104   */
105  @Nonnull
106  <T> QueryIterator<T> findIterate(Query<T> query, Transaction transaction);
107
108  /**
109   * Return the query result as a Stream using a single persistence context.
110   * <p>
111   * Note that the stream needs to be closed so use with try with resources.
112   * </p>
113   */
114  @Nonnull
115  <T> Stream<T> findStream(Query<T> query, Transaction transaction);
116
117  /**
118   * Return the query result as a Stream (with multiple persistence contexts).
119   * <p>
120   * Note that the stream needs to be closed so use with try with resources.
121   * </p>
122   */
123  @Nonnull
124  <T> Stream<T> findLargeStream(Query<T> query, Transaction transaction);
125
126  /**
127   * Execute the query visiting the each bean one at a time.
128   * <p>
129   * Unlike findList() this is suitable for processing a query that will return
130   * a very large resultSet. The reason is that not all the result beans need to be
131   * held in memory at the same time and instead processed one at a time.
132   * </p>
133   * <p>
134   * Internally this query using a PersistenceContext scoped to each bean (and the
135   * beans associated object graph).
136   * </p>
137   * <p>
138   * <pre>{@code
139   *
140   *     DB.find(Order.class)
141   *       .where().eq("status", Order.Status.NEW)
142   *       .order().asc("id")
143   *       .findEach((Order order) -> {
144   *
145   *         // do something with the order bean
146   *         System.out.println(" -- processing order ... " + order);
147   *       });
148   *
149   * }</pre>
150   *
151   * @see Query#findEach(Consumer)
152   * @see Query#findEachWhile(Predicate)
153   */
154  <T> void findEach(Query<T> query, Consumer<T> consumer, Transaction transaction);
155
156  /**
157   * Execute the query visiting the each bean one at a time.
158   * <p>
159   * Compared to findEach() this provides the ability to stop processing the query
160   * results early by returning false for the Predicate.
161   * </p>
162   * <p>
163   * Unlike findList() this is suitable for processing a query that will return
164   * a very large resultSet. The reason is that not all the result beans need to be
165   * held in memory at the same time and instead processed one at a time.
166   * </p>
167   * <p>
168   * Internally this query using a PersistenceContext scoped to each bean (and the
169   * beans associated object graph).
170   * </p>
171   * <p>
172   * <pre>{@code
173   *
174   *     DB.find(Order.class)
175   *       .where().eq("status", Order.Status.NEW)
176   *       .order().asc("id")
177   *       .findEachWhile((Order order) -> {
178   *
179   *         // do something with the order bean
180   *         System.out.println(" -- processing order ... " + order);
181   *
182   *         boolean carryOnProcessing = ...
183   *         return carryOnProcessing;
184   *       });
185   *
186   * }</pre>
187   *
188   * @see Query#findEach(Consumer)
189   * @see Query#findEachWhile(Predicate)
190   */
191  <T> void findEachWhile(Query<T> query, Predicate<T> consumer, Transaction transaction);
192
193  /**
194   * Return versions of a @History entity bean.
195   * <p>
196   * Generally this query is expected to be a find by id or unique predicates query.
197   * It will execute the query against the history returning the versions of the bean.
198   * </p>
199   */
200  @Nonnull
201  <T> List<Version<T>> findVersions(Query<T> query, Transaction transaction);
202
203  /**
204   * Execute a query returning a list of beans.
205   * <p>
206   * Generally you are able to use {@link Query#findList()} rather than
207   * explicitly calling this method. You could use this method if you wish to
208   * explicitly control the transaction used for the query.
209   * </p>
210   * <p>
211   * <pre>{@code
212   *
213   * List<Customer> customers = DB.find(Customer.class)
214   *     .where().ilike("name", "rob%")
215   *     .findList();
216   *
217   * }</pre>
218   *
219   * @param <T>         the type of entity bean to fetch.
220   * @param query       the query to execute.
221   * @param transaction the transaction to use (can be null).
222   * @return the list of fetched beans.
223   * @see Query#findList()
224   */
225  @Nonnull
226  <T> List<T> findList(Query<T> query, Transaction transaction);
227
228  /**
229   * Execute find row count query in a background thread.
230   * <p>
231   * This returns a Future object which can be used to cancel, check the
232   * execution status (isDone etc) and get the value (with or without a
233   * timeout).
234   * </p>
235   *
236   * @param query       the query to execute the row count on
237   * @param transaction the transaction (can be null).
238   * @return a Future object for the row count query
239   * @see Query#findFutureCount()
240   */
241  @Nonnull
242  <T> FutureRowCount<T> findFutureCount(Query<T> query, Transaction transaction);
243
244  /**
245   * Execute find Id's query in a background thread.
246   * <p>
247   * This returns a Future object which can be used to cancel, check the
248   * execution status (isDone etc) and get the value (with or without a
249   * timeout).
250   * </p>
251   *
252   * @param query       the query to execute the fetch Id's on
253   * @param transaction the transaction (can be null).
254   * @return a Future object for the list of Id's
255   * @see Query#findFutureIds()
256   */
257  @Nonnull
258  <T> FutureIds<T> findFutureIds(Query<T> query, Transaction transaction);
259
260  /**
261   * Execute find list query in a background thread returning a FutureList object.
262   * <p>
263   * This returns a Future object which can be used to cancel, check the
264   * execution status (isDone etc) and get the value (with or without a timeout).
265   * <p>
266   * This query will execute in it's own PersistenceContext and using its own transaction.
267   * What that means is that it will not share any bean instances with other queries.
268   *
269   * @param query       the query to execute in the background
270   * @param transaction the transaction (can be null).
271   * @return a Future object for the list result of the query
272   * @see Query#findFutureList()
273   */
274  @Nonnull
275  <T> FutureList<T> findFutureList(Query<T> query, Transaction transaction);
276
277  /**
278   * Return a PagedList for this query using firstRow and maxRows.
279   * <p>
280   * The benefit of using this over findList() is that it provides functionality to get the
281   * total row count etc.
282   * </p>
283   * <p>
284   * If maxRows is not set on the query prior to calling findPagedList() then a
285   * PersistenceException is thrown.
286   * </p>
287   * <p>
288   * <pre>{@code
289   *
290   *  PagedList<Order> pagedList = DB.find(Order.class)
291   *       .setFirstRow(50)
292   *       .setMaxRows(20)
293   *       .findPagedList();
294   *
295   *       // fetch the total row count in the background
296   *       pagedList.loadRowCount();
297   *
298   *       List<Order> orders = pagedList.getList();
299   *       int totalRowCount = pagedList.getTotalRowCount();
300   *
301   * }</pre>
302   *
303   * @return The PagedList
304   * @see Query#findPagedList()
305   */
306  @Nonnull
307  <T> PagedList<T> findPagedList(Query<T> query, Transaction transaction);
308
309  /**
310   * Execute the query returning a set of entity beans.
311   * <p>
312   * Generally you are able to use {@link Query#findSet()} rather than
313   * explicitly calling this method. You could use this method if you wish to
314   * explicitly control the transaction used for the query.
315   * </p>
316   * <p>
317   * <pre>{@code
318   *
319   * Set<Customer> customers = DB.find(Customer.class)
320   *     .where().ilike("name", "rob%")
321   *     .findSet();
322   *
323   * }</pre>
324   *
325   * @param <T>         the type of entity bean to fetch.
326   * @param query       the query to execute
327   * @param transaction the transaction to use (can be null).
328   * @return the set of fetched beans.
329   * @see Query#findSet()
330   */
331  @Nonnull
332  <T> Set<T> findSet(Query<T> query, Transaction transaction);
333
334  /**
335   * Execute the query returning the entity beans in a Map.
336   * <p>
337   * Generally you are able to use {@link Query#findMap()} rather than
338   * explicitly calling this method. You could use this method if you wish to
339   * explicitly control the transaction used for the query.
340   * </p>
341   *
342   * @param <T>         the type of entity bean to fetch.
343   * @param query       the query to execute.
344   * @param transaction the transaction to use (can be null).
345   * @return the map of fetched beans.
346   * @see Query#findMap()
347   */
348  @Nonnull
349  <K, T> Map<K, T> findMap(Query<T> query, Transaction transaction);
350
351  /**
352   * Execute the query returning a list of values for a single property.
353   * <p>
354   * <h3>Example 1:</h3>
355   * <pre>{@code
356   *
357   *  List<String> names =
358   *    DB.find(Customer.class)
359   *      .select("name")
360   *      .order().asc("name")
361   *      .findSingleAttributeList();
362   *
363   * }</pre>
364   * <h3>Example 2:</h3>
365   * <pre>{@code
366   *
367   *  List<String> names =
368   *    DB.find(Customer.class)
369   *      .setDistinct(true)
370   *      .select("name")
371   *      .where().eq("status", Customer.Status.NEW)
372   *      .order().asc("name")
373   *      .setMaxRows(100)
374   *      .findSingleAttributeList();
375   *
376   * }</pre>
377   *
378   * @return the list of values for the selected property
379   * @see Query#findSingleAttributeList()
380   */
381  @Nonnull
382  <A, T> List<A> findSingleAttributeList(Query<T> query, Transaction transaction);
383
384  /**
385   * Execute the query returning at most one entity bean or null (if no matching
386   * bean is found).
387   * <p>
388   * This will throw a NonUniqueResultException if the query finds more than one result.
389   * </p>
390   * <p>
391   * Generally you are able to use {@link Query#findOne()} rather than
392   * explicitly calling this method. You could use this method if you wish to
393   * explicitly control the transaction used for the query.
394   * </p>
395   *
396   * @param <T>         the type of entity bean to fetch.
397   * @param query       the query to execute.
398   * @param transaction the transaction to use (can be null).
399   * @return the list of fetched beans.
400   * @throws NonUniqueResultException if more than one result was found
401   * @see Query#findOne()
402   */
403  @Nullable
404  <T> T findOne(Query<T> query, Transaction transaction);
405
406  /**
407   * Similar to findOne() but returns an Optional (rather than nullable).
408   */
409  @Nonnull
410  <T> Optional<T> findOneOrEmpty(Query<T> query, Transaction transaction);
411
412  /**
413   * Execute as a delete query deleting the 'root level' beans that match the predicates
414   * in the query.
415   * <p>
416   * Note that if the query includes joins then the generated delete statement may not be
417   * optimal depending on the database platform.
418   * </p>
419   *
420   * @param query       the query used for the delete
421   * @param transaction the transaction to use (can be null)
422   * @param <T>         the type of entity bean to fetch.
423   * @return the number of beans/rows that were deleted
424   */
425  <T> int delete(Query<T> query, Transaction transaction);
426
427  /**
428   * Execute the update query returning the number of rows updated.
429   * <p>
430   * The update query must be created using {@link Database#update(Class)}.
431   * </p>
432   *
433   * @param query       the update query to execute
434   * @param transaction the optional transaction to use for the update (can be null)
435   * @param <T>         the type of entity bean
436   * @return The number of rows updated
437   */
438  <T> int update(Query<T> query, Transaction transaction);
439
440  /**
441   * Execute the sql query returning a list of MapBean.
442   * <p>
443   * Generally you are able to use {@link SqlQuery#findList()} rather than
444   * explicitly calling this method. You could use this method if you wish to
445   * explicitly control the transaction used for the query.
446   * </p>
447   *
448   * @param query       the query to execute.
449   * @param transaction the transaction to use (can be null).
450   * @return the list of fetched MapBean.
451   * @see SqlQuery#findList()
452   */
453  @Nonnull
454  List<SqlRow> findList(SqlQuery query, Transaction transaction);
455
456  /**
457   * Execute the SqlQuery iterating a row at a time.
458   * <p>
459   * This streaming type query is useful for large query execution as only 1 row needs to be held in memory.
460   * </p>
461   */
462  void findEach(SqlQuery query, Consumer<SqlRow> consumer, Transaction transaction);
463
464  /**
465   * Execute the SqlQuery iterating a row at a time with the ability to stop consuming part way through.
466   * <p>
467   * Returning false after processing a row stops the iteration through the query results.
468   * </p>
469   * <p>
470   * This streaming type query is useful for large query execution as only 1 row needs to be held in memory.
471   * </p>
472   */
473  void findEachWhile(SqlQuery query, Predicate<SqlRow> consumer, Transaction transaction);
474
475  /**
476   * Execute the sql query returning a single MapBean or null.
477   * <p>
478   * This will throw a PersistenceException if the query found more than one
479   * result.
480   * </p>
481   * <p>
482   * Generally you are able to use {@link SqlQuery#findOne()} rather than
483   * explicitly calling this method. You could use this method if you wish to
484   * explicitly control the transaction used for the query.
485   * </p>
486   *
487   * @param query       the query to execute.
488   * @param transaction the transaction to use (can be null).
489   * @return the fetched MapBean or null if none was found.
490   * @see SqlQuery#findOne()
491   */
492  @Nullable
493  SqlRow findOne(SqlQuery query, Transaction transaction);
