048   * Helper class for managing fast and safe access and creation of Databases.
049   */
050  private static final class ServerManager {
051
052    /**
053     * Cache for fast concurrent read access.
054     */
055    private final ConcurrentHashMap<String, EbeanServer> concMap = new ConcurrentHashMap<>();
056
057    /**
058     * Cache for synchronized read, creation and put. Protected by the monitor object.
059     */
060    private final HashMap<String, EbeanServer> syncMap = new HashMap<>();
061
062    private final Object monitor = new Object();
063
064    /**
065     * The 'default' Database.
066     */
067    private EbeanServer defaultServer;
068
069    private ServerManager() {
070
071      try {
072        if (!PrimaryServer.isSkip()) {
073          // look to see if there is a default server defined
074          String defaultName = PrimaryServer.getDefaultServerName();
075          logger.debug("defaultName:{}", defaultName);
076          if (defaultName != null && !defaultName.trim().isEmpty()) {
077            defaultServer = getWithCreate(defaultName.trim());
078          }
079        }
080      } catch (BeanNotEnhancedException e) {
081        throw e;
082
083      } catch (DataSourceConfigurationException e) {
084        String msg = "Configuration error creating DataSource for the default Database." +
085          " This typically means a missing application-test.yaml or missing ebean-test-config dependency." +
086          " See https://ebean.io/docs/trouble-shooting#datasource";
087        throw new DataSourceConfigurationException(msg, e);
088
089      } catch (Throwable e) {
090        logger.error("Error trying to create the default Database", e);
091        throw new RuntimeException(e);
092      }
093    }
094
095    private EbeanServer getDefaultServer() {
096      if (defaultServer == null) {
097        String msg = "The default Database has not been defined?";
098        msg += " This is normally set via the ebean.datasource.default property.";
099        msg += " Otherwise it should be registered programmatically via registerServer()";
100        throw new PersistenceException(msg);
101      }
102      return defaultServer;
103    }
104
105    private EbeanServer get(String name) {
106      if (name == null || name.isEmpty()) {
107        return defaultServer;
108      }
109      // non-synchronized read
110      EbeanServer server = concMap.get(name);
111      if (server != null) {
112        return server;
113      }
114      // synchronized read, create and put
115      return getWithCreate(name);
116    }
117
118    /**
119     * Synchronized read, create and put of Databases.
120     */
121    private EbeanServer getWithCreate(String name) {
122
123      synchronized (monitor) {
124
125        EbeanServer server = syncMap.get(name);
126        if (server == null) {
127          // register when creating server this way
128          server = EbeanServerFactory.create(name);
129          register(server, false);
130        }
131        return server;
132      }
133    }
134
135    /**
136     * Register a server so we can get it by its name.
137     */
138    private void register(EbeanServer server, boolean isDefaultServer) {
139      registerWithName(server.getName(), server, isDefaultServer);
140    }
141
142    private void registerWithName(String name, EbeanServer server, boolean isDefaultServer) {
143      synchronized (monitor) {
144        concMap.put(name, server);
145        syncMap.put(name, server);
146        if (isDefaultServer) {
147          defaultServer = server;
148        }
149      }
150    }
151
152  }

243  /**
244   * Start a transaction with 'REQUIRED' semantics.
245   * <p>
246   * With REQUIRED semantics if an active transaction already exists that transaction will be used.
247   * </p>
248   * <p>
249   * The transaction is stored in a ThreadLocal variable and typically you only
250   * need to use the returned Transaction <em>IF</em> you wish to do things like
251   * use batch mode, change the transaction isolation level, use savepoints or
252   * log comments to the transaction log.
253   * </p>
254   * <p>
255   * Example of using a transaction to span multiple calls to find(), save()
256   * etc.
257   * </p>
258   * <pre>{@code
259   *
260   *   // start a transaction (stored in a ThreadLocal)
261   *   Ebean.beginTransaction();
262   *   try {
263   *       Order order = Ebean.find(Order.class,10); ...
264   *
265   *       Ebean.save(order);
266   *
267   *       Ebean.commitTransaction();
268   *
269   *   } finally {
270   *       // rollback if we didn't commit
271   *       // i.e. an exception occurred before commitTransaction().
272   *       Ebean.endTransaction();
273   *   }
274   *
275   * }</pre>
276   * <p>
277   * If you want to externalise the transaction management then you should be
278   * able to do this via EbeanServer. Specifically with EbeanServer you can pass
279   * the transaction to the various find() and save() execute() methods. This
280   * gives you the ability to create the transactions yourself externally from
281   * Ebean and pass those transactions through to the various methods available
282   * on EbeanServer.
283   * </p>
284   */
285  public static Transaction beginTransaction() {
286    return serverMgr.getDefaultServer().beginTransaction();
287  }

298  /**
299   * Start a transaction typically specifying REQUIRES_NEW or REQUIRED semantics.
300   * <p>
301   * Note that this provides an try finally alternative to using {@link #executeCall(TxScope, Callable)} or
302   * {@link #execute(TxScope, Runnable)}.
303   * </p>
304   * <p>
305   * <h3>REQUIRES_NEW example:</h3>
306   * <pre>{@code
307   * // Start a new transaction. If there is a current transaction
308   * // suspend it until this transaction ends
309   * Transaction txn = Ebean.beginTransaction(TxScope.requiresNew());
310   * try {
311   *
312   *   ...
313   *
314   *   // commit the transaction
315   *   txn.commit();
316   *
317   * } finally {
318   *   // end this transaction which:
319   *   //  A) will rollback transaction if it has not been committed already
320   *   //  B) will restore a previously suspended transaction
321   *   txn.end();
322   * }
323   *
324   * }</pre>
325   * <h3>REQUIRED example:</h3>
326   * <pre>{@code
327   *
328   * // start a new transaction if there is not a current transaction
329   * Transaction txn = Ebean.beginTransaction(TxScope.required());
330   * try {
331   *
332   *   ...
333   *
334   *   // commit the transaction if it was created or
335   *   // do nothing if there was already a current transaction
336   *   txn.commit();
337   *
338   * } finally {
339   *   // end this transaction which will rollback the transaction
340   *   // if it was created for this try finally scope and has not
341   *   // already been committed
342   *   txn.end();
343   * }
344   *
345   * }</pre>
346   */
347  public static Transaction beginTransaction(TxScope scope) {
348    return serverMgr.getDefaultServer().beginTransaction(scope);
349  }
