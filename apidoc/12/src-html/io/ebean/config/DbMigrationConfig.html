374  /**
375   * Returns a DDL header prepend for each DDL. E.g. for copyright headers
376   * You can use placeholders like ${version} or ${timestamp} in properties file.
377   */
378  public String getDdlHeader() {
379    if (ddlHeader != null && !ddlHeader.isEmpty()) {
380      ddlHeader = replace(ddlHeader, "${version}", EbeanVersion.getVersion());
381      ddlHeader = replace(ddlHeader, "${timestamp}", ZonedDateTime.now().format( DateTimeFormatter.ISO_INSTANT ));
382    }
383    return ddlHeader;
384  }

428  /**
429   * Load the settings from the PropertiesWrapper.
430   */
431  public void loadSettings(PropertiesWrapper properties, String serverName) {
432
433    migrationPath = properties.get("migration.migrationPath", migrationPath);
434    migrationInitPath = properties.get("migration.migrationInitPath", migrationInitPath);
435    modelPath = properties.get("migration.modelPath", modelPath);
436    applyPrefix = properties.get("migration.applyPrefix", applyPrefix);
437    applySuffix = properties.get("migration.applySuffix", applySuffix);
438    modelSuffix = properties.get("migration.modelSuffix", modelSuffix);
439
440    platform = properties.getEnum(Platform.class, "migration.platform", platform);
441    patchInsertOn = properties.get("migration.patchInsertOn", patchInsertOn);
442    patchResetChecksumOn = properties.get("migration.patchResetChecksumOn", patchResetChecksumOn);
443
444    runMigration = properties.getBoolean("migration.run", runMigration);
445    metaTable = properties.get("migration.metaTable", metaTable);
446    runPlaceholders = properties.get("migration.placeholders", runPlaceholders);
447    dbSchema = properties.get("migration.dbSchema", dbSchema);
448
449    //Do not set user and pass from "datasource.db.username"
450    //There is a null test in MigrationRunner::getConnection to handle this
451    //String adminUser = properties.get("datasource." + serverName + ".username", dbUsername);
452    String adminUser = properties.get("datasource." + serverName + ".adminusername", dbUsername);
453    dbUsername = properties.get("migration.dbusername", adminUser);
454
455    //String adminPwd = properties.get("datasource." + serverName + ".password", dbPassword);
456    String adminPwd = properties.get("datasource." + serverName + ".adminpassword", dbPassword);
457    dbPassword = properties.get("migration.dbpassword", adminPwd);
458    ddlHeader = properties.get("ddl.header", ddlHeader);
459  }

461  /**

463   */
464  protected String readEnvironment(String key) {
465
466    String val = System.getProperty(key);
467    if (val == null) {
468      val = System.getenv(key);
469    }
470    return val;
471  }

480  /**
481   * Create the MigrationRunner to run migrations if necessary.
482   */
483  public MigrationRunner createRunner(ClassLoader classLoader, Properties properties) {
484
485    runnerConfig.setMetaTable(metaTable);
486    runnerConfig.setApplySuffix(applySuffix);
487    runnerConfig.setMigrationPath(migrationPath);
488    runnerConfig.setMigrationInitPath(migrationInitPath);
489    runnerConfig.setRunPlaceholderMap(runPlaceholderMap);
490    runnerConfig.setRunPlaceholders(runPlaceholders);
491    runnerConfig.setDbUsername(getDbUsername());
492    runnerConfig.setDbPassword(getDbPassword());
493    runnerConfig.setDbSchema(getDbSchema());
494    if (defaultDbSchema) {
495      runnerConfig.setSetCurrentSchema(false);
496    }
497    runnerConfig.setClassLoader(classLoader);
498    if (patchInsertOn != null) {
499      runnerConfig.setPatchInsertOn(patchInsertOn);
500    }
501    if (patchResetChecksumOn != null) {
502      runnerConfig.setPatchResetChecksumOn(patchResetChecksumOn);
503    }
504    if (properties != null) {
505      runnerConfig.load(properties);
506    }
507    return new MigrationRunner(runnerConfig);
508  }
