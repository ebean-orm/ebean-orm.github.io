046/**
047 * The configuration used for creating a Database.
048 * <p>
049 * Used to programmatically construct a Database and optionally register it
050 * with the DB singleton.
051 * </p>
052 * <p>
053 * If you just use DB without this programmatic configuration DB will read
054 * the application.properties file and take the configuration from there. This usually
055 * includes searching the class path and automatically registering any entity
056 * classes and listeners etc.
057 * </p>
058 * <pre>{@code
059 *
060 * ServerConfig config = new ServerConfig();
061 *
062 * // read the ebean.properties and load
063 * // those settings into this serverConfig object
064 * config.loadFromProperties();
065 *
066 * // explicitly register the entity beans to avoid classpath scanning
067 * config.addClass(Customer.class);
068 * config.addClass(User.class);
069 *
070 * Database database = DatabaseFactory.create(config);
071 *
072 * }</pre>
073 *
074 * <p>
075 * Note that ServerConfigProvider provides a standard Java ServiceLoader mechanism that can
076 * be used to apply configuration to the ServerConfig.
077 * </p>
078 *
079 * @author emcgreal
080 * @author rbygrave
081 * @see DatabaseFactory
082 */

629  private String serviceObjectKey(Object configObject) {
630    return serviceObjectKey(configObject.getClass());
631  }

633  private String serviceObjectKey(Class<?> cls) {
634    String simpleName = cls.getSimpleName();
635    return Character.toLowerCase(simpleName.charAt(0)) + simpleName.substring(1);
636  }

639   * Used by plugins to obtain service objects.

641   * <pre>{@code
642   *

644   *
645   * }</pre>
646   *
647   * @param cls The type of the service object to obtain
648   * @return The service object given the class type
649   */
650  @SuppressWarnings("unchecked")
651  public <P> P getServiceObject(Class<P> cls) {
652    return (P) serviceObject.get(serviceObjectKey(cls));
