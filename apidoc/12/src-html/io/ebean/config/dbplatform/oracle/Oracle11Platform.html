001package io.ebean.config.dbplatform.oracle;
002
003import io.ebean.annotation.Platform;
004import io.ebean.config.dbplatform.IdType;
005
006/**
007 * Oracle 11 platform using rownum sql limiting.
008 */
009public class Oracle11Platform extends OraclePlatform {
010
011  public Oracle11Platform() {
012    super();
013    this.platform = Platform.ORACLE11;
014    this.columnAliasPrefix = "c";
015    this.sqlLimiter = new OracleRownumSqlLimiter();
016    this.basicSqlLimiter = new OracleRownumBasicLimiter();
017    dbIdentity.setIdType(IdType.SEQUENCE);
018  }
019}
