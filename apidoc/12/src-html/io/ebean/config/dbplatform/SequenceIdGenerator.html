001package io.ebean.config.dbplatform;
002
003import io.ebean.BackgroundExecutor;
004import io.ebean.Transaction;
005import io.ebean.util.JdbcClose;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import javax.persistence.PersistenceException;
010import javax.sql.DataSource;
011import java.sql.Connection;
012import java.sql.PreparedStatement;
013import java.sql.ResultSet;
014import java.sql.SQLException;
015import java.util.Collections;
016import java.util.List;
017import java.util.NavigableSet;
018import java.util.TreeSet;
019import java.util.concurrent.locks.ReentrantLock;
020
021/**
022 * Database sequence based IdGenerator.
023 */
024public abstract class SequenceIdGenerator implements PlatformIdGenerator {
025
026  protected static final Logger logger = LoggerFactory.getLogger("io.ebean.SEQ");
027
028  private final ReentrantLock lock = new ReentrantLock(false);
029
030  private final ReentrantLock loadLock = new ReentrantLock(false);
031
032  /**
033   * The actual sequence name.
034   */
035  protected final String seqName;
036
037  protected final DataSource dataSource;
038
039  protected final BackgroundExecutor backgroundExecutor;
040
041  protected final NavigableSet<Long> idList = new TreeSet<>();
042
043  protected final int allocationSize;
044
045  protected boolean currentlyBackgroundLoading;
046
047  /**
048   * Construct given a dataSource and sql to return the next sequence value.
049   */
050  protected SequenceIdGenerator(BackgroundExecutor be, DataSource ds, String seqName, int allocationSize) {
051    this.backgroundExecutor = be;
052    this.dataSource = ds;
053    this.seqName = seqName;
054    this.allocationSize = allocationSize;
055  }
056
057  public abstract String getSql(int batchSize);
058
059  /**
060   * Returns the sequence name.
061   */
062  @Override
063  public String getName() {
064    return seqName;
065  }
066
067  /**
068   * Returns true.
069   */
070  @Override
071  public boolean isDbSequence() {
072    return true;
073  }
074
075  /**
076   * If allocateSize is large load some sequences in a background thread.
077   * <p>
078   * For example, when inserting a bean with a cascade on a OneToMany with many
079   * beans Ebean can call this to ensure .
080   * </p>
081   */
082  @Override
083  public void preAllocateIds(int requestSize) {
084    // do nothing by default
085  }
086
087  /**
088   * Return the next Id.
089   * <p>
090   * If a Transaction has been passed in use the Connection from it.
091   * </p>
092   */
093  @Override
094  public Object nextId(Transaction t) {
095    lock.lock();
096    try {
097      int size = idList.size();
098      if (size > 0) {
099        maybeLoadMoreInBackground(size);
100      } else {
101        loadMore(allocationSize);
102      }
103      return idList.pollFirst();
104    } finally {
105      lock.unlock();
106    }
107  }
108
109  private void maybeLoadMoreInBackground(int currentSize) {
110    if (allocationSize > 1) {
111      if (currentSize <= allocationSize / 2) {
112        loadInBackground(allocationSize);
113      }
114    }
115  }
116
117  private void loadMore(int requestSize) {
118    List<Long> newIds = getMoreIds(requestSize);
119    lock.lock();
120    try {
121      idList.addAll(newIds);
122    } finally {
123      lock.unlock();
124    }
125  }
126
127  /**
128   * Load another batch of Id's using a background thread.
129   */
130  protected void loadInBackground(final int requestSize) {
131    // single threaded processing...
132    loadLock.lock();
133    try {
134      if (currentlyBackgroundLoading) {
135        // skip as already background loading
136        logger.debug("... skip background sequence load (another load in progress)");
137        return;
138      }
139      currentlyBackgroundLoading = true;
140      backgroundExecutor.execute(() -> {
141        loadMore(requestSize);
142        loadLock.lock();
143        try {
144          currentlyBackgroundLoading = false;
145        } finally {
146          loadLock.unlock();
147        }
148      });
149    } finally {
150      loadLock.lock();
151    }
152  }
153
154  /**
155   * Read the resultSet returning the list of Id values.
156   */
157  protected abstract List<Long> readIds(ResultSet resultSet, int loadSize) throws SQLException;
158
159  /**
160   * Get more Id's by executing a query and reading the Id's returned.
161   */
162  protected List<Long> getMoreIds(int requestSize) {
163
164    String sql = getSql(requestSize);
165
166    Connection connection = null;
167    PreparedStatement statement = null;
168    ResultSet resultSet = null;
169    try {
170      connection = dataSource.getConnection();
171
172      statement = connection.prepareStatement(sql);
173      resultSet = statement.executeQuery();
174
175      List<Long> newIds = readIds(resultSet, requestSize);
176      if (logger.isTraceEnabled()) {
177        logger.trace("seq:{} loaded:{} sql:{}", seqName, newIds.size(), sql);
178      }
179      if (newIds.isEmpty()) {
180        throw new PersistenceException("Always expecting more than 1 row from " + sql);
181      }
182
183      return newIds;
184
185    } catch (SQLException e) {
186      if (e.getMessage().contains("Database is already closed")) {
187        String msg = "Error getting SEQ when DB shutting down " + e.getMessage();
188        logger.error(msg);
189        System.out.println(msg);
190        return Collections.emptyList();
191      } else {
192        throw new PersistenceException("Error getting sequence nextval", e);
193      }
194    } finally {
195      closeResources(connection, statement, resultSet);
196    }
197  }
198
199  /**
200   * Close the JDBC resources.
201   */
202  private void closeResources(Connection connection, PreparedStatement statement, ResultSet resultSet) {
203    JdbcClose.close(resultSet);
204    JdbcClose.close(statement);
205    JdbcClose.close(connection);
206  }
207
208}
