001package com.avaje.ebean;
002
003import com.avaje.ebean.annotation.DocStoreMode;
004import com.avaje.ebean.config.DocStoreConfig;
005import com.avaje.ebean.config.PersistBatch;
006import com.avaje.ebean.config.ServerConfig;

008import javax.persistence.PersistenceException;
009import javax.persistence.RollbackException;
010import java.io.Closeable;
011import java.sql.Connection;
012
013/**
014 * The Transaction object. Typically representing a JDBC or JTA transaction.
015 */
016public interface Transaction extends Closeable {
017
018  /**
019   * Read Committed transaction isolation. Same as
020   * java.sql.Connection.TRANSACTION_READ_COMMITTED.
021   */
022  int READ_COMMITTED = java.sql.Connection.TRANSACTION_READ_COMMITTED;
023
024  /**
025   * Read Uncommitted transaction isolation. Same as
026   * java.sql.Connection.TRANSACTION_READ_UNCOMMITTED.
027   */
028  int READ_UNCOMMITTED = java.sql.Connection.TRANSACTION_READ_UNCOMMITTED;
029
030  /**
031   * Repeatable read transaction isolation. Same as
032   * java.sql.Connection.TRANSACTION_REPEATABLE_READ.
033   */
034  int REPEATABLE_READ = java.sql.Connection.TRANSACTION_REPEATABLE_READ;
035
036  /**
037   * Serializable transaction isolation. Same as
038   * java.sql.Connection.TRANSACTION_SERIALIZABLE.
039   */
040  int SERIALIZABLE = java.sql.Connection.TRANSACTION_SERIALIZABLE;
041
042  /**
043   * Register a TransactionCallback with this transaction.
044   */
045  void register(TransactionCallback callback);
046
047  /**
048   * Return true if this transaction is read only.
049   */
050  boolean isReadOnly();
051
052  /**
053   * Set whether this transaction should be readOnly.
054   */
055  void setReadOnly(boolean readOnly);
056
057  /**
058   * Commits the transaction at this point with the expectation that another
059   * commit (or rollback or end) will occur later to complete the transaction.
060   * <p>
061   * This is similar to commit() but leaves the transaction "Active".
062   * </p>
063   * <h3>Functions/h3>
064   * <ul>
065   *   <li>Flush the JDBC batch buffer</li>
066   *   <li>Call commit on the underlying JDBC connection</li>
067   *   <li>Trigger any registered TransactionCallbacks</li>
068   *   <li>Perform post-commit processing updating L2 cache, ElasticSearch etc</li>
069   * </ul>
070   */
071  void commitAndContinue() throws RollbackException;
072
073  /**
074   * Commit the transaction.
075   * <p>
076   * This performs commit and completes the transaction closing underlying resources and
077   * marking the transaction as "In active".
078   * </p>
079   * <h3>Functions/h3>
080   * <ul>
081   *   <li>Flush the JDBC batch buffer</li>
082   *   <li>Call commit on the underlying JDBC connection</li>
083   *   <li>Trigger any registered TransactionCallbacks</li>
084   *   <li>Perform post-commit processing updating L2 cache, ElasticSearch etc</li>
085   *   <li>Close any underlying resources, closing the underlying JDBC connection</li>
086   *   <li>Mark the transaction as "Inactive"</li>
087   * </ul>
088   */
089  void commit() throws RollbackException;
090
091  /**
092   * Rollback the transaction.
093   * <p>
094   * This performs rollback, closes underlying resources and marks the transaction as "In active".
095   * </p>
096   * <h3>Functions/h3>
097   * <ul>
098   *   <li>Call rollback on the underlying JDBC connection</li>
099   *   <li>Trigger any registered TransactionCallbacks</li>
100   *   <li>Close any underlying resources, closing the underlying JDBC connection</li>
101   *   <li>Mark the transaction as "Inactive"</li>
102   * </ul>
103   */
104  void rollback() throws PersistenceException;
105
106  /**
107   * Rollback the transaction specifying a throwable that caused the rollback to
108   * occur.
109   * <p>
110   * If you are using transaction logging this will log the throwable in the
111   * transaction logs.
112   * </p>
113   */
114  void rollback(Throwable e) throws PersistenceException;
115
116  /**
117   * Mark the transaction for rollback only.
118   */
119  void setRollbackOnly();
120
121  /**
122   * Return true if the transaction is marked as rollback only.
123   */
124  boolean isRollbackOnly();
125
126  /**
127   * If the transaction is active then perform rollback. Otherwise do nothing.
128   */
129  void end() throws PersistenceException;
130
131  /**
132   * Return true if the transaction is active.
133   */
134  boolean isActive();
135
136  /**
137   * Set the behavior for document store updates on this transaction.
138   * <p>
139   *   For example, set the mode to DocStoreEvent.IGNORE for this transaction and
140   *   then any changes via this transaction are not sent to the doc store. This
141   *   would be used when doing large bulk inserts into the database and we want
142   *   to control how that is sent to the document store.
143   * </p>
144   */
145  void setDocStoreMode(DocStoreMode mode);
146
147  /**
148   * Set the batch size to use for sending messages to the document store.
149   * <p>
150   *   You might set this if you know the changes in this transaction result in especially large or
151   *   especially small payloads and want to adjust the batch size to match.
152   * </p>
153   * <p>
154   *   Setting this overrides the default of {@link DocStoreConfig#getBulkBatchSize()}
155   * </p>
156   */
157  void setDocStoreBatchSize(int batchSize);
158
159  /**
160   * Explicitly turn off or on the cascading nature of save() and delete(). This
161   * gives the developer exact control over what beans are saved and deleted
162   * rather than Ebean cascading detecting 'dirty/modified' beans etc.
163   * <p>
164   * This is useful if you can getting back entity beans from a layer of code
165   * (potentially remote) and you prefer to have exact control.
166   * </p>
167   * <p>
168   * This may also be useful if you are using jdbc batching with jdbc drivers
169   * that do not support getGeneratedKeys.
170   * </p>
171   */
172  void setPersistCascade(boolean persistCascade);
173
174  /**
175   * Set to true when you want all loaded properties to be included in the update
176   * (rather than just the changed properties).
177   * <p>
178   * You might set this when using JDBC batch in order to get multiple updates
179   * with slightly different sets of changed properties into the same statement
180   * and hence better JDBC batch performance.
181   * </p>
182   */
183  void setUpdateAllLoadedProperties(boolean updateAllLoadedProperties);
184
185  /**
186   * Set if the L2 cache should be skipped for "find by id" and "find by natural key" queries.
187   * <p>
188   * By default {@link ServerConfig#isSkipCacheAfterWrite()} is true and that means that for
189   * "find by id" and "find by natural key" queries which normally hit L2 bean cache automatically
190   * - will not do so after a persist/write on the transaction.
191   * </p>
192   * <p>
193   * This method provides explicit control over whether "find by id" and "find by natural key"
194   * will skip the L2 bean cache or not (regardless of whether the transaction is considered "read only").
195   * </p>
196   * <p>
197   * Refer to {@link ServerConfig#setSkipCacheAfterWrite(boolean)} for configuring the default behavior
198   * for using the L2 bean cache in transactions spanning multiple query/persist requests.
199   * </p>
200   *
201   * <pre>{@code
202   *
203   *   // assume Customer has L2 bean caching enabled ...
204   *
205   *   Transaction transaction = Ebean.beginTransaction();
206   *   try {
207   *
208   *     // this uses L2 bean cache as the transaction
209   *     // ... is considered "query only" at this point
210   *     Customer.find.byId(42);
211   *
212   *     // transaction no longer "query only" once
213   *     // ... a bean has been saved etc
214   *     Ebean.save(someBean);
215   *
216   *     // will NOT use L2 bean cache as the transaction
217   *     // ... is no longer considered "query only"
218   *     Customer.find.byId(55);
219   *
220   *
221   *
222   *     // explicit control - please use L2 bean cache
223   *
224   *     transaction.setSkipCache(false);
225   *     Customer.find.byId(77); // hit the l2 bean cache
226   *
227   *
228   *     // explicit control - please don't use L2 bean cache
229   *
230   *     transaction.setSkipCache(true);
231   *     Customer.find.byId(99); // skips l2 bean cache
232   *
233   *
234   *   } finally {
235   *     transaction.end();
236   *   }
237   *
238   * }</pre>
239   *
240   * @see ServerConfig#isSkipCacheAfterWrite()
241   */
242  void setSkipCache(boolean skipCache);
243
244  /**
245   * Return true if the L2 cache should be skipped. More accurately if true then find by id
246   * and find by natural key queries should NOT automatically use the L2 bean cache.
247   */
248  boolean isSkipCache();
249
250  /**
251   * Turn on or off statement batching. Statement batching can be transparent
252   * for drivers and databases that support getGeneratedKeys. Otherwise you may
253   * wish to specifically control when batching is used via this method.
254   * <p>
255   * Refer to <code>java.sql.PreparedStatement.addBatch();</code>
256   * <p>
257   * Note that you may also wish to use the setPersistCascade method to stop
258   * save and delete cascade behaviour. You may do this to have full control
259   * over the order of execution rather than the normal cascading fashion.
260   * </p>
261   * <p>
262   * Note that the <em>execution order</em> in batch mode may be different from
263   * non batch mode execution order. Also note that <em>insert behaviour</em>
264   * may be different depending on the JDBC driver and its support for
265   * getGeneratedKeys. That is, for JDBC drivers that do not support
266   * getGeneratedKeys you may not get back the generated IDs (used for inserting
267   * associated detail beans etc).
268   * </p>
269   * <p>
270   * Calls to save(), delete(), insert() and execute() all support batch
271   * processing. This includes normal beans, MapBean, CallableSql and UpdateSql.
272   * </p>
273   * <p>
274   * The flushing of the batched statements is automatic but you can call
275   * batchFlush when you like. Note that flushing occurs when a query is
276   * executed or when you mix UpdateSql and CallableSql with save and delete of
277   * beans.
278   * </p>
279   * <p>
280   * Example: batch processing executing every 3 rows
281   * </p>
282   * 
283   * <pre>{@code
284   *
285   * String data = "This is a simple test of the batch processing"
286   *             + " mode and the transaction execute batch method";
287   * 
288   * String[] da = data.split(" ");
289   * 
290   * String sql = "{call sp_t3(?,?)}";
291   * 
292   * CallableSql cs = new CallableSql(sql);
293   * cs.registerOut(2, Types.INTEGER);
294   * 
295   * // (optional) inform eBean this stored procedure
296   * // inserts into a table called sp_test
297   * cs.addModification("sp_test", true, false, false);
298   * 
299   * Transaction txn = ebeanServer.beginTransaction();
300   * txn.setBatchMode(true);
301   * txn.setBatchSize(3);
302   * try {
303   *   for (int i = 0; i < da.length;) {
304   *     cs.setParameter(1, da[i]);
305   *     ebeanServer.execute(cs);
306   *   }
307   * 
308   *   // NB: commit implicitly flushes
309   *   txn.commit();
310   * 
311   * } finally {
312   *   txn.end();
313   * }
314   *
315   * }</pre>
316   * 
317   */
318  void setBatchMode(boolean useBatch);
319
320  /**
321   * The JDBC batch mode to use for this transaction.
322   * <p>
323   * If this is NONE then JDBC batch can still be used for each request - save(), insert(), update() or delete()
324   * and this would be useful if the request cascades to detail beans.
325   * </p>
326   *
327   * @param persistBatchMode the batch mode to use for this transaction
328   *
329   * @see com.avaje.ebean.config.ServerConfig#setPersistBatch(com.avaje.ebean.config.PersistBatch)
330   */
331  void setBatch(PersistBatch persistBatchMode);
332
333  /**
334   * Return the batch mode at the transaction level.
335   */
336  PersistBatch getBatch();
337
338  /**
339   * Set the JDBC batch mode to use for a save() or delete() request.
340   * <p>
341   * This only takes effect when batch mode on the transaction has not already meant that
342   * JDBC batch mode is being used.
343   * </p>
344   * <p>
345   * This is useful when the single save() or delete() cascades. For example, inserting a 'master' cascades
346   * and inserts a collection of 'detail' beans. The detail beans can be inserted using JDBC batch.
347   * </p>
348   *
349   * @param batchOnCascadeMode the batch mode to use per save(), insert(), update() or delete()
350   *
351   * @see com.avaje.ebean.config.ServerConfig#setPersistBatchOnCascade(com.avaje.ebean.config.PersistBatch)
352   */
353  void setBatchOnCascade(PersistBatch batchOnCascadeMode);
354
355  /**
356   * Return the batch mode at the request level (for each save(), insert(), update() or delete()).
357   */
358  PersistBatch getBatchOnCascade();
359
360  /**
361   * Specify the number of statements before a batch is flushed automatically.
362   */
363  void setBatchSize(int batchSize);
364
365  /**
366   * Return the current batch size.
367   */
368  int getBatchSize();
369
370  /**
371   * Specify if you want batched inserts to use getGeneratedKeys.
372   * <p>
373   * By default batched inserts will try to use getGeneratedKeys if it is
374   * supported by the underlying jdbc driver and database.
375   * </p>
376   * <p>
377   * You may want to turn getGeneratedKeys off when you are inserting a large
378   * number of objects and you don't care about getting back the ids.
379   * </p>
380   */
381  void setBatchGetGeneratedKeys(boolean getGeneratedKeys);
382
383  /**
384   * By default when mixing UpdateSql (or CallableSql) with Beans the batch is
385   * automatically flushed when you change (between persisting beans and
386   * executing UpdateSql or CallableSql).
387   * <p>
388   * If you want to execute both WITHOUT having the batch automatically flush
389   * you need to call this with batchFlushOnMixed = false.
390   * </p>
391   * <p>
392   * Note that UpdateSql and CallableSql are ALWAYS executed first (before the
393   * beans are executed). This is because the UpdateSql and CallableSql have
394   * already been bound to their PreparedStatements. The beans on the other hand
395   * have a 2 step process (delayed binding).
396   * </p>
397   */
398  void setBatchFlushOnMixed(boolean batchFlushOnMixed);
399
400  /**
401   * By default executing a query will automatically flush any batched
402   * statements (persisted beans, executed UpdateSql etc).
403   * <p>
404   * Calling this method with batchFlushOnQuery = false means that you can
405   * execute a query and the batch will not be automatically flushed.
406   * </p>
407   */
408  void setBatchFlushOnQuery(boolean batchFlushOnQuery);
409
410  /**
411   * Return true if the batch (of persisted beans or executed UpdateSql etc)
412   * should be flushed prior to executing a query.
413   * <p>
414   * The default is for this to be true.
415   * </p>
416   */
417  boolean isBatchFlushOnQuery();
418
419  /**
420   * The batch will be flushing automatically but you can use this to explicitly
421   * flush the batch if you like.
422   * <p>
423   * Flushing occurs automatically when:
424   * </p>
425   * <ul>
426   * <li>the batch size is reached</li>
427   * <li>A query is executed on the same transaction</li>
428   * <li>UpdateSql or CallableSql are mixed with bean save and delete</li>
429   * <li>Transaction commit occurs</li>
430   * </ul>
431   */
432  void flushBatch() throws PersistenceException;
433
434  /**
435   * Return the underlying Connection object.
436   * <p>
437   * Useful where a Developer wishes to use the JDBC API directly. Note that the
438   * commit() rollback() and end() methods on the Transaction should still be
439   * used. Calling these methods on the Connection would be a big no no unless
440   * you know what you are doing.
441   * </p>
442   * <p>
443   * Examples of when a developer may wish to use the connection directly are:
444   * Savepoints, advanced CLOB BLOB use and advanced stored procedure calls.
445   * </p>
446   */
447  Connection getConnection();
448
449  /**
450   * Add table modification information to the TransactionEvent.
451   * <p>
452   * Use this in conjunction with getConnection() and raw JDBC.
453   * </p>
454   * <p>
455   * This effectively informs Ebean of the data that has been changed by the
456   * transaction and this information is normally automatically handled by Ebean
457   * when you save entity beans or use UpdateSql etc.
458   * </p>
459   * <p>
460   * If you use raw JDBC then you can use this method to inform Ebean for the
461   * tables that have been modified. Ebean uses this information to keep its
462   * caches in synch and maintain text indexes.
463   * </p>
464   */
465  void addModification(String tableName, boolean inserts, boolean updates, boolean deletes);
466
467  /**
468   * Add an arbitrary user object to the transaction. The objects added have no
469   * impact on any internals of ebean and are solely meant as a convenient
470   * method push user information (although somewhat replaced by TransactionCallback).
471   */
472  void putUserObject(String name, Object value);
473
474  /**
475   * Get an object added with {@link #putUserObject(String, Object)}.
476   */
477  Object getUserObject(String name);
