001package com.avaje.ebean.config;
002
003import com.avaje.ebean.Query;
004
005/**
006 * The mode for determining if Autofetch will be used for a given query when
007 * {@link Query#setAutofetch(boolean)} has not been explicitly set on a query.
008 * <p>
009 * The explicit control of {@link Query#setAutofetch(boolean)} will always take
010 * precedence. This mode is used when this has not been explicitly set on a
011 * query.
012 * </p>
013 */
014public enum AutofetchMode {
015
016  /**
017   * Don't implicitly use Autofetch. Must explicitly turn it on.
018   */
019  DEFAULT_OFF,
020
021  /**
022   * Use Autofetch implicitly. Must explicitly turn it off.
023   */
024  DEFAULT_ON,
025
026  /**
027   * Implicitly use Autofetch if the query has not got either select() or join()
028   * defined.
029   */
030  DEFAULT_ONIFEMPTY
031
032}
