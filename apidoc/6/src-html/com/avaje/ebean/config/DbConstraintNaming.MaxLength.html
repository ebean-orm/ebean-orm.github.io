001package com.avaje.ebean.config;
002
003/**
004 * Naming convention used for constraint names.
005 * <p>
006 * Note that these constraint names are trimmed in the PlatformDdl which can be overridden
007 * but provides a decent default implementation.
008 * </p>
009 */
010public class DbConstraintNaming {
011
012  /**
013   * Defines how constraint names are shortened if required based on platform limitations.
014   */
015  public interface MaxLength {
016
017    /**
018     * Truncate or shorten the constraint name to support DB platform limitations.
019     * <p>
020     * There is a default implementation of this which is used if an implementation is
021     * not specified.
022     * </p>
023     */
024    String maxLength(String constraintName, int count);
025  }
026
027  protected String pkPrefix = "pk_";
028  protected String pkSuffix = "";
029
030  protected String fkPrefix = "fk_";
031  protected String fkMiddle = "_";
032  protected String fkSuffix = "";
033
034  protected String fkIndexPrefix = "ix_";
035  protected String fkIndexMiddle = "_";
036  protected String fkIndexSuffix = "";
037
038  protected String indexPrefix = "ix_";
039  protected String indexMiddle = "_";
040  protected String indexSuffix = "";
041
042  protected String uqPrefix = "uq_";
043  protected String uqSuffix = "";
044
045  protected String ckPrefix = "ck_";
046  protected String ckSuffix = "";
047
048  protected boolean lowerCaseNames = true;
049
050  protected MaxLength maxLength;
051
052  protected DbConstraintNormalise normalise = new DbConstraintNormalise();
053
054  public DbConstraintNaming() {
055  }
056
057  /**
058   * Return the MaxLength implementation used to truncate/shorten db constraint names as necessary.
059   */
060  public MaxLength getMaxLength() {
061    return maxLength;
062  }
063
064  /**
065   * Set the MaxLength implementation used to truncate/shorten db constraint names as necessary.
066   */
067  public void setMaxLength(MaxLength maxLength) {
068    this.maxLength = maxLength;
069  }
070
071  /**
072   * Return the primary key constraint name.
073   */
074  public String primaryKeyName(String tableName) {
075
076    return pkPrefix + normaliseTable(tableName) + pkSuffix;
077  }
078
079  /**
080   * Return the foreign key constraint name given a single column foreign key.
081   */
082  public String foreignKeyConstraintName(String tableName, String columnName) {
083    return fkPrefix + normaliseTable(tableName) + fkMiddle + normaliseColumn(columnName) + fkSuffix;
084  }
085
086  /**
087   * Return the index name associated with a foreign key constraint given multiple columns.
088   */
089  public String foreignKeyIndexName(String tableName, String[] columns) {
090
091    String colPart = joinColumnNames(columns);
092    return fkIndexPrefix + normaliseTable(tableName) + fkIndexMiddle + colPart + fkIndexSuffix;
093  }
094
095  /**
096   * Return the index name associated with a foreign key constraint given a single column foreign key.
097   */
098  public String foreignKeyIndexName(String tableName, String column) {
099
100    String colPart = normaliseTable(column);
101    return fkIndexPrefix + normaliseTable(tableName) + fkIndexMiddle + colPart + fkIndexSuffix;
102  }
103
104  /**
105   * Return the index name for a general index (not associated with a foreign key).
106   */
107  public String indexName(String tableName, String column) {
108
109    String colPart = normaliseTable(column);
110    return indexPrefix + normaliseTable(tableName) + indexMiddle + colPart + indexSuffix;
111  }
112
113  /**
114   * Return the index name for a general index (not associated with a foreign key).
115   */
116  public String indexName(String tableName, String[] columns) {
117
118    String colPart = joinColumnNames(columns);
119    return indexPrefix + normaliseTable(tableName) + indexMiddle + colPart + indexSuffix;
120  }
121
122  /**
123   * Join the column names together with underscores.
124   */
125  protected String joinColumnNames(String[] columns) {
126
127    if (columns.length == 1) {
128      return normaliseColumn(columns[0]);
129    }
130    StringBuilder sb = new StringBuilder();
131    for (int i = 0; i < columns.length; i++) {
132      if (i > 0) {
133        sb.append("_");
134      }
135      sb.append(normaliseColumn(columns[i]));
136    }
137    return sb.toString();
138  }
139
140  /**
141   * Return the unique constraint name.
142   */
143  public String uniqueConstraintName(String tableName, String columnName) {
144
145    return uqPrefix + normaliseTable(tableName) + "_" + normaliseColumn(columnName) + uqSuffix;
146  }
147
148  /**
149   * Return the unique constraint name.
150   */
151  public String uniqueConstraintName(String tableName, String[] columns) {
152
153    String colPart = joinColumnNames(columns);
154    return uqPrefix + normaliseTable(tableName) + "_" + colPart + uqSuffix;
155  }
156
157  /**
158   * Return the check constraint name.
159   */
160  public String checkConstraintName(String tableName, String columnName) {
161
162    return ckPrefix + normaliseTable(tableName) + "_" + normaliseColumn(columnName) + ckSuffix;
163  }
164
165  /**
166   * Normalise the table name by trimming catalog and schema and removing any
167   * quoted identifier characters (",',[,] etc).
168   */
169  public String normaliseTable(String tableName) {
170
171    return normalise.normaliseTable(tableName);
172  }
173
174  /**
175   * Normalise the column name by removing any quoted identifier characters (",',[,] etc).
176   */
177  public String normaliseColumn(String tableName) {
178
179    return normalise.normaliseColumn(tableName);
180  }
181
182  /**
183   * Lower case the table or column name checking for quoted identifiers.
184   */
185  public String lowerName(String tableName) {
186    if (lowerCaseNames && normalise.notQuoted(tableName)) {
187      return tableName.toLowerCase();
188    }
189    return tableName;
190  }
191
192}
