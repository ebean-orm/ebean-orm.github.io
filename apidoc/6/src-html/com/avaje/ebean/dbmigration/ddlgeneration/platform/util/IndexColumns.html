001package com.avaje.ebean.dbmigration.ddlgeneration.platform.util;
002
003import java.util.ArrayList;
004import java.util.List;
005
006/**
007 * Set of columns making up a particular index (column order is important).
008 */
009public class IndexColumns {
010
011  List<String> columns = new ArrayList<String>(4);
012
013  /**
014   * Construct representing as a single column index.
015   */
016  public IndexColumns(String column) {
017    columns.add(column);
018  }
019
020  /**
021   * Construct representing index.
022   */
023  public IndexColumns(String[] columnNames) {
024    for (int i = 0; i < columnNames.length; i++) {
025      columns.add(columnNames[i]);
026    }
027  }
028
029  /**
030   * Return true if this index matches (same single column).
031   */
032  public boolean isMatch(String singleColumn) {
033    return columns.size() == 1 && columns.get(0).equals(singleColumn);
034  }
035
036  /**
037   * Return true if this index matches (same single column).
038   */
039  public boolean isMatch(List<String> columnNames) {
040    if (columns.size() != columnNames.size()) {
041      return false;
042    }
043    for (int i = 0; i <columns.size() ; i++) {
044      if (!columns.get(i).equals(columnNames.get(i))) {
045        return false;
046      }
047    }
048    return true;
049  }
050
051  /**
052   * Return true if this index matches (same columns same order).
053   */
054  public boolean isMatch(IndexColumns other) {
055    return columns.equals(other.columns);
056  }
057
058  /**
059   * Add a unique index based on the single column.
060   */
061  protected void add(String column) {
062    columns.add(column);
063  }
064
065}
