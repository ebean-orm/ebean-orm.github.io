092  /**
093   * Generate the appropriate 'create table' and matching 'drop table' statements
094   * and add them to the appropriate 'apply' and 'rollback' buffers.
095   */
096  @Override
097  public void generate(DdlWrite writer, CreateTable createTable) throws IOException {
098
099    reset();
100
101    String tableName = lowerName(createTable.getName());
102    List<Column> columns = createTable.getColumn();
103    List<Column> pk = determinePrimaryKeyColumns(columns);
104
105    boolean singleColumnPrimaryKey = (pk.size() == 1);
106    boolean useIdentity = false;
107    boolean useSequence = false;
108
109    if (singleColumnPrimaryKey) {
110      IdType useDbIdentityType = platformDdl.useIdentityType(createTable.getIdentityType());
111      useIdentity = (IdType.IDENTITY == useDbIdentityType);
112      useSequence = (IdType.SEQUENCE == useDbIdentityType);
113    }
114
115    DdlBuffer apply = writer.apply();
116    apply.append("create table ").append(tableName).append(" (");
117    writeTableColumns(apply, columns, useIdentity);
118    writeCheckConstraints(apply, createTable);
119    writeUniqueConstraints(apply, createTable);
120    writeCompoundUniqueConstraints(apply, createTable);
121    if (!pk.isEmpty()) {
122      // defined on the columns
123      writePrimaryKeyConstraint(apply, createTable.getPkName(), toColumnNames(pk));
124    }
125
126    apply.newLine().append(")").endOfStatement();
127
128    writeUniqueOneToOneConstraints(writer, createTable);
129
130    if (isTrue(createTable.isWithHistory())) {
131      // create history with rollback before the
132      // associated drop table is written to rollback
133      createWithHistory(writer, createTable.getName());
134    }
135
136    // add drop table to the rollback buffer - do this before
137    // we drop the related sequence (if sequences are used)
138    dropTable(writer.rollback(), tableName);
139
140    if (useSequence) {
141      String pkCol = pk.get(0).getName();
142      writeSequence(writer, createTable, pkCol);
143    }
144
145    // add blank line for a bit of whitespace between tables
146    apply.end();
147    writer.rollback().end();
148
149    writeAddForeignKeys(writer, createTable);
150
151  }
152
153  private void writeTableColumns(DdlBuffer apply, List<Column> columns, boolean useIdentity) throws IOException {
154    platformDdl.writeTableColumns(apply, columns, useIdentity);
155  }

157  /**
158   * Specific handling of OneToOne unique constraints for MsSqlServer.
159   * For all other DB platforms these unique constraints are done inline as per normal.
160   */
161  protected void writeUniqueOneToOneConstraints(DdlWrite write, CreateTable createTable) throws IOException {
162
163    String tableName = createTable.getName();
164    for (Column col : externalUnique) {
165      String uqName = col.getUniqueOneToOne();
166      String[] columnNames = {col.getName()};
167      write.apply()
168          .append(platformDdl.alterTableAddUniqueConstraint(tableName, uqName, columnNames))
169          .endOfStatement();
170
171      write.rollbackForeignKeys()
172          .append(platformDdl.dropIndex(uqName, tableName))
173          .endOfStatement();
174    }
175  }
176
177  protected void writeSequence(DdlWrite writer, CreateTable createTable, String pk) throws IOException {
178
179    // explicit sequence use or platform decides
180    String explicitSequenceName = createTable.getSequenceName();
181    int initial = toInt(createTable.getSequenceInitial());
182    int allocate = toInt(createTable.getSequenceAllocate());
183
184    String seqName = explicitSequenceName;
185    if (seqName == null) {
186      seqName = namingConvention.getSequenceName(createTable.getName(), pk);
187    }
188
189    String createSeq = platformDdl.createSequence(seqName, initial, allocate);
190    if (createSeq != null) {
191      writer.apply().append(createSeq).newLine();
192      writer.rollback().append(platformDdl.dropSequence(seqName)).endOfStatement();
193    }
194  }

202  protected void writeAddForeignKeys(DdlWrite write, CreateTable createTable) throws IOException {
203
204    String tableName = createTable.getName();
205    List<Column> columns = createTable.getColumn();
206    for (Column column : columns) {
207      String references = column.getReferences();
208      if (hasValue(references)) {
209        writeForeignKey(write, tableName, column);
210      }
211    }
212
213    writeAddCompoundForeignKeys(write, createTable);
214  }
215
216  protected void writeAddCompoundForeignKeys(DdlWrite write, CreateTable createTable) throws IOException {
217
218    String tableName = createTable.getName();
219
220    List<ForeignKey> foreignKey = createTable.getForeignKey();
221    for (ForeignKey key : foreignKey) {
222
223      String refTableName = key.getRefTableName();
224      String fkName = key.getName();
225      String[] cols = toColumnNamesSplit(key.getColumnNames());
226      String[] refColumns = toColumnNamesSplit(key.getRefColumnNames());
227
228      writeForeignKey(write, fkName, tableName, cols, refTableName, refColumns, key.getIndexName());
229    }
230  }
231
232  protected void writeForeignKey(DdlWrite write, String tableName, Column column) throws IOException {
233
234    String fkName = column.getForeignKeyName();
235    String references = column.getReferences();
236    int pos = references.lastIndexOf('.');
237    if (pos == -1) {
238      throw new IllegalStateException("Expecting period '.' character for table.column split but not found in [" + references + "]");
239    }
240    String refTableName = references.substring(0, pos);
241    String refColumnName = references.substring(pos + 1);
242
243    String[] cols = {column.getName()};
244    String[] refCols = {refColumnName};
245
246    writeForeignKey(write, fkName, tableName, cols, refTableName, refCols, column.getForeignKeyIndex());
247  }
248
249  protected void writeForeignKey(DdlWrite write, String fkName, String tableName, String[] columns, String refTable, String[] refColumns, String indexName) throws IOException {
250
251    tableName = lowerName(tableName);
252    DdlBuffer fkeyBuffer = write.applyForeignKeys();
253    alterTableAddForeignKey(fkeyBuffer, fkName, tableName, columns, refTable, refColumns);
254
255    if (indexName != null) {
256      // no matching unique constraint so add the index
257      fkeyBuffer.append(platformDdl.createIndex(indexName, tableName, columns)).endOfStatement();
258    }
259
260    fkeyBuffer.end();
261
262    write.rollbackForeignKeys()
263        .append(platformDdl.alterTableDropForeignKey(tableName, fkName)).endOfStatement();
264
265    if (indexName != null) {
266      write.rollbackForeignKeys()
267          .append(platformDdl.dropIndex(indexName, tableName)).endOfStatement();
268    }
269
270    write.rollbackForeignKeys().end();
271
272  }

275
276    buffer.append(platformDdl.alterTableAddForeignKey(tableName, fkName, columns, refTable, refColumns)).endOfStatement();
277  }

279  protected void appendColumns(String[] columns, DdlBuffer buffer) throws IOException {
280    buffer.append(" (");
281    for (int i = 0; i < columns.length; i++) {
282      if (i > 0) {
283        buffer.append(",");
284      }
285      buffer.append(lowerName(columns[i].trim()));
286    }
287    buffer.append(")");
288  }

305    for (Column column : columns) {

313  /**
314   * Write a check constraint.
315   */
316  protected void writeCheckConstraint(DdlBuffer buffer, Column column, String checkConstraint) throws IOException {
317
318    String ckName = column.getCheckConstraintName();
319
320    buffer.append(",").newLine();
321    buffer.append("  constraint ").append(ckName);
322    buffer.append(" ").append(checkConstraint);

325  protected void writeCompoundUniqueConstraints(DdlBuffer apply, CreateTable createTable) throws IOException {
326
327    List<UniqueConstraint> uniqueConstraints = createTable.getUniqueConstraint();
328    for (UniqueConstraint uniqueConstraint : uniqueConstraints) {
329      String uqName = uniqueConstraint.getName();
330      String[] columns = toColumnNamesSplit(uniqueConstraint.getColumnNames());
331      apply.append(",").newLine();
332      apply.append("  constraint ").append(uqName).append(" unique");
333      appendColumns(columns, apply);
334    }
335  }
336
337  /**
338   * Write the unique constraints inline with the create table statement.
339   */
340  protected void writeUniqueConstraints(DdlBuffer apply, CreateTable createTable) throws IOException {
341
342    boolean inlineUniqueOneToOne = platformDdl.isInlineUniqueOneToOne();
343
344    List<Column> columns = createTable.getColumn();
345    for (Column column : columns) {
346      if (hasValue(column.getUnique()) || (inlineUniqueOneToOne && hasValue(column.getUniqueOneToOne()))) {
347        // normal mechanism for adding unique constraint
348        inlineUniqueConstraintSingle(apply, column);
349
350      } else if (!inlineUniqueOneToOne && hasValue(column.getUniqueOneToOne())) {
351        // MsSqlServer specific mechanism for adding unique constraints (that allow nulls)
352        externalUnique.add(column);
353      }
354    }
355  }
356
357  /**
358   * Write the unique constraint inline with the create table statement.
359   */
360  protected void inlineUniqueConstraintSingle(DdlBuffer buffer, Column column) throws IOException {
361
362    String uqName = column.getUnique();
363    if (uqName == null) {
364      uqName = column.getUniqueOneToOne();
365    }
366
367    buffer.append(",").newLine();
368    buffer.append("  constraint ").append(uqName).append(" unique ");
369    buffer.append("(");
370    buffer.append(lowerName(column.getName()));
371    buffer.append(")");
372  }

374  /**
375   * Write the primary key constraint inline with the create table statement.

378
379    buffer.append(",").newLine();
380    buffer.append("  constraint ").append(pkName).append(" primary key");
381    appendColumns(pkColumns, buffer);

414  /**
415   * Return the list of columns that make the primary key.
416   */
417  protected List<Column> determinePrimaryKeyColumns(List<Column> columns) {
418    List<Column> pk = new ArrayList<Column>(3);
419    for (Column column : columns) {
420      if (isTrue(column.isPrimaryKey())) {
421        pk.add(column);
422      }
423    }
424    return pk;

474  /**
475   * Add add column DDL.
476   */
477  @Override
478  public void generate(DdlWrite writer, AddColumn addColumn) throws IOException {
479
480    String tableName = addColumn.getTableName();
481    List<Column> columns = addColumn.getColumn();
482    for (Column column : columns) {
483      alterTableAddColumn(writer.apply(), tableName, column, false);
484      alterTableDropColumn(writer.rollback(), tableName, column.getName());
485    }
486
487    if (isTrue(addColumn.isWithHistory())) {
488      // make same changes to the history table
489      String historyTable = historyTable(tableName);
490      for (Column column : columns) {
491        regenerateHistoryTriggers(tableName, HistoryTableUpdate.Change.ADD, column.getName());
492        alterTableAddColumn(writer.apply(), historyTable, column, true);
493        alterTableDropColumn(writer.rollback(), historyTable, column.getName());
494      }
495    }
496
497    // add a bit of whitespace
498    writer.apply().end();
499    writer.rollback().end();
500  }

515  public void generate(DdlWrite writer, DropColumn dropColumn) throws IOException {

517    String tableName = dropColumn.getTableName();

520    if (isTrue(dropColumn.isWithHistory())) {

529  /**
530   * Add all the appropriate changes based on the column changes.
531   */
532  @Override
533  public void generate(DdlWrite writer, AlterColumn alterColumn) throws IOException {
534
535    if (isTrue(alterColumn.isHistoryExclude())) {
536      regenerateHistoryTriggers(alterColumn.getTableName(), HistoryTableUpdate.Change.EXCLUDE, alterColumn.getColumnName());
537    } else if (isFalse(alterColumn.isHistoryExclude())) {
538      regenerateHistoryTriggers(alterColumn.getTableName(), HistoryTableUpdate.Change.INCLUDE, alterColumn.getColumnName());
539    }
540
541    if (hasValue(alterColumn.getDropForeignKey())) {
542      alterColumnDropForeignKey(writer, alterColumn);
543    }
544    if (hasValue(alterColumn.getReferences())) {
545      alterColumnAddForeignKey(writer, alterColumn);
546    }
547
548    if (hasValue(alterColumn.getDropUnique())) {
549      alterColumnDropUniqueConstraint(writer, alterColumn);
550    }
551    if (hasValue(alterColumn.getUnique())) {
552      alterColumnAddUniqueConstraint(writer, alterColumn);
553    }
554    if (hasValue(alterColumn.getUniqueOneToOne())) {
555      alterColumnAddUniqueOneToOneConstraint(writer, alterColumn);
556    }
557
558    boolean alterBaseAttributes = false;
559    if (hasValue(alterColumn.getType())) {
560      alterColumnType(writer, alterColumn);
561      alterBaseAttributes = true;
562    }
563    if (hasValue(alterColumn.getDefaultValue())) {
564      alterColumnDefaultValue(writer, alterColumn);
565      alterBaseAttributes = true;
566    }
567    if (alterColumn.isNotnull() != null) {
568      alterColumnNotnull(writer, alterColumn);
569      alterBaseAttributes = true;
570    }
571
572    if (alterBaseAttributes) {
573      alterColumnBaseAttributes(writer, alterColumn);
574    }
575  }

584  /**
585   * Register the base table that we need to regenerate the history triggers on.
586   */
587  protected void regenerateHistoryTriggers(String baseTableName, HistoryTableUpdate.Change change, String column) {
588
589    HistoryTableUpdate update = regenerateHistoryTriggers.get(baseTableName);
590    if (update == null) {
591      update = new HistoryTableUpdate(baseTableName);
592      regenerateHistoryTriggers.put(baseTableName, update);
593    }
594    update.add(change, column);
595  }

597  /**
598   * This is mysql specific - alter all the base attributes of the column together.
599   */
600  protected void alterColumnBaseAttributes(DdlWrite writer, AlterColumn alter) throws IOException {
601
602    String ddl = platformDdl.alterColumnBaseAttributes(alter);
603    if (hasValue(ddl)) {
604      writer.apply().append(ddl).endOfStatement();
605
606      // reverse and generate the rollback statement
607      String currentType = alter.getCurrentType();
608      String type = alter.getType();
609      Boolean currentNotnull = alter.isCurrentNotnull();
610      Boolean notnull = alter.isNotnull();
611
612      alter.setCurrentType(type);
613      alter.setType(currentType);
614      alter.setNotnull(currentNotnull);
615      alter.setCurrentNotnull(notnull);
616
617      // write the rollback
618      ddl = platformDdl.alterColumnBaseAttributes(alter);
619      writer.rollback().append(ddl).endOfStatement();
620
621      if (isTrue(alter.isWithHistory()) && alter.getType() != null) {
622        // mysql and sql server column type change allowing nulls in the history table column
623        AlterColumn alterHistoryColumn = new AlterColumn();
624        alterHistoryColumn.setTableName(historyTable(alter.getTableName()));
625        alterHistoryColumn.setColumnName(alter.getColumnName());
626        alterHistoryColumn.setType(alter.getType());
627        String histColumnDdl = platformDdl.alterColumnBaseAttributes(alterHistoryColumn);
628
629        // write the apply to history table
630        writer.apply().append(histColumnDdl).endOfStatement();
631
632        // write the rollback from history table
633        alterHistoryColumn.setType(currentType);
634        histColumnDdl = platformDdl.alterColumnBaseAttributes(alterHistoryColumn);
635        writer.rollback().append(histColumnDdl).endOfStatement();
636      }
637    }
638  }

648  protected void alterColumnNotnull(DdlWrite writer, AlterColumn alter) throws IOException {

651    if (hasValue(ddl)) {

653      ddl = platformDdl.alterColumnNotnull(alter.getTableName(), alter.getColumnName(), alter.isCurrentNotnull());
654      writer.rollback().append(ddl).endOfStatement();

658  protected void alterColumnType(DdlWrite writer, AlterColumn alter) throws IOException {
659
660    String ddl = platformDdl.alterColumnType(alter.getTableName(), alter.getColumnName(), alter.getType());
661    if (hasValue(ddl)) {
662      writer.apply().append(ddl).endOfStatement();
663      ddl = platformDdl.alterColumnType(alter.getTableName(), alter.getColumnName(), alter.getCurrentType());
664      writer.rollback().append(ddl).endOfStatement();
665      if (isTrue(alter.isWithHistory())) {
666        // apply same type change to matching column in the history table
667        ddl = platformDdl.alterColumnType(historyTable(alter.getTableName()), alter.getColumnName(), alter.getType());
668        writer.apply().append(ddl).endOfStatement();
669        ddl = platformDdl.alterColumnType(historyTable(alter.getTableName()), alter.getColumnName(), alter.getCurrentType());
670        writer.rollback().append(ddl).endOfStatement();
671      }
672    }
673  }

676  protected void alterColumnAddForeignKey(DdlWrite writer, AlterColumn alterColumn) throws IOException {
677
678    String tableName = alterColumn.getTableName();
679    String fkName = alterColumn.getForeignKeyName();
680    String[] cols = {alterColumn.getColumnName()};
681    String references = alterColumn.getReferences();
682    int pos = references.lastIndexOf('.');
683    if (pos == -1) {
684      throw new IllegalStateException("Expecting period '.' character for table.column split but not found in [" + references + "]");
685    }
686    String refTableName = references.substring(0, pos);
687    String refColumnName = references.substring(pos + 1);
688    String[] refCols = {refColumnName};
689

691  }

735        .endOfStatement();

738  protected void alterTableAddColumn(DdlBuffer buffer, String tableName, Column column, boolean onHistoryTable) throws IOException {
739
740    buffer.append("alter table ").append(tableName)
741        .append(" add column ").append(column.getName())
742        .append(" ").append(column.getType());
743
744    if (!onHistoryTable) {
745      if (isTrue(column.isNotnull())) {
746        buffer.append(" not null");
747      }
748      if (hasValue(column.getCheckConstraint())) {
749        buffer.append(" ").append(column.getCheckConstraint());
750      }
751    }
752    buffer.endOfStatement();
753  }

755  protected boolean isFalse(Boolean value) {
756    return value != null && !value;
757  }
758
759  /**
760   * Return true if null or trimmed string is empty.
761   */
762  protected boolean hasValue(String value) {
763    return value != null && !value.trim().isEmpty();
764  }
765
766  /**
767   * Null safe Boolean true test.
768   */
769  protected boolean isTrue(Boolean value) {
770    return Boolean.TRUE.equals(value);
771  }

773  /**
774   * Return as an int value with 0 when it is null.
775   */
776  protected int toInt(BigInteger value) {
777    return (value == null) ? 0 : value.intValue();
