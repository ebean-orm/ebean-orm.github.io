001package com.avaje.ebean.dbmigration.model.visitor;
002
003import com.avaje.ebeaninternal.server.deploy.BeanProperty;
004import com.avaje.ebeaninternal.server.deploy.BeanPropertyAssocMany;
005import com.avaje.ebeaninternal.server.deploy.BeanPropertyAssocOne;
006import com.avaje.ebeaninternal.server.deploy.BeanPropertyCompound;
007
008/**
009 * Used to help mark PropertyVisitor methods that need to be implemented
010 * to visit base table properties.
011 */
012public abstract class BaseTablePropertyVisitor implements BeanPropertyVisitor {
013
014  /**
015   * Not required in that you can use the visitEmbeddedScalar.
016   */
017  public void visitEmbedded(BeanPropertyAssocOne<?> p) {
018  }
019
020  /**
021   * Override this method.
022   */
023  public abstract void visitEmbeddedScalar(BeanProperty p, BeanPropertyAssocOne<?> embedded);
024
025  /**
026   * Not part of base table.
027   */
028  public void visitMany(BeanPropertyAssocMany<?> p) {
029  }
030
031  /**
032   * Not part of base table.
033   */
034  public void visitOneExported(BeanPropertyAssocOne<?> p) {
035  }
036
037  /**
038   * Override this method for the foreign key.
039   */
040  public abstract void visitOneImported(BeanPropertyAssocOne<?> p);
041
042  /**
043   * Override this method for normal scalar property.
044   */
045  public abstract void visitScalar(BeanProperty p);
046
047  /**
048   * Not required in that the scalar properties map to the columns.
049   */
050  public void visitCompound(BeanPropertyCompound p) {
051  }
052
053  /**
054   * Override this method for scalar property inside a Immutable Compound Value object.
055   */
056  public abstract void visitCompoundScalar(BeanPropertyCompound compound, BeanProperty p);
057
058
059}
