001package io.ebean.meta;
002
003import java.util.List;
004
005/**
006 * Provides access to the meta data in EbeanServer such as query execution statistics.
007 */
008public interface MetaInfoManager {
009
010  /**
011   * Visit the metrics resetting and collecting/reporting as desired.
012   */
013  void visitMetrics(MetricVisitor visitor);
014
015  /**
016   * Run a visit collecting all the metrics and returning BasicMetricVisitor
017   * which holds all the metrics in simple lists.
018   */
019  BasicMetricVisitor visitBasic();
020
021  /**
022   * Just reset all the metrics. Maybe only useful for testing purposes.
023   */
024  void resetAllMetrics();
025
026  /**
027   * Collect and return the ObjectGraphNode statistics.
028   * <p>
029   * These show query executions based on an origin point and relative path.
030   * This is used to look at the amount of lazy loading occurring for a given
031   * query origin point and highlight potential for tuning a query.
032   * </p>
033   *
034   * @param reset Set to true to reset the underlying statistics after collection.
035   */
036  List<MetaOrmQueryNode> collectNodeStatistics(boolean reset);
037
038}
