001package io.ebean;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.Arrays;
006import java.util.List;
007
008/**
009 * Represents an Order By for a Query.
010 * <p>
011 * Is a ordered list of OrderBy.Property objects each specifying a property and
012 * whether it is ascending or descending order.
013 * </p>
014 * <p>
015 * Typically you will not construct an OrderBy yourself but use one that exists
016 * on the Query object.
017 * </p>
018 */
019public final class OrderBy<T> implements Serializable {
020
021  private static final long serialVersionUID = 9157089257745730539L;
022
023  private transient Query<T> query;
024
025  private final List<Property> list;
026
027  /**
028   * Create an empty OrderBy with no associated query.
029   */
030  public OrderBy() {
031    this.list = new ArrayList<>(3);
032  }
033
034  private OrderBy(List<Property> list) {
035    this.list = list;
036  }
037
038  /**
039   * Create an orderBy parsing the order by clause.
040   * <p>
041   * The order by clause follows SQL order by clause with comma's between each
042   * property and optionally "asc" or "desc" to represent ascending or
043   * descending order respectively.
044   * </p>
045   */
046  public OrderBy(String orderByClause) {
047    this(null, orderByClause);
048  }
049
050  /**
051   * Construct with a given query and order by clause.
052   */
053  public OrderBy(Query<T> query, String orderByClause) {
054    this.query = query;
055    this.list = new ArrayList<>(3);
056    parse(orderByClause);
057  }
058
059  /**
060   * Reverse the ascending/descending order on all the properties.
061   */
062  public void reverse() {
063    for (Property aList : list) {
064      aList.reverse();
065    }
066  }
067
068  /**
069   * Add a property with ascending order to this OrderBy.
070   */
071  public Query<T> asc(String propertyName) {
072
073    list.add(new Property(propertyName, true));
074    return query;
075  }
076
077  /**
078   * Add a property with descending order to this OrderBy.
079   */
080  public Query<T> desc(String propertyName) {
081
082    list.add(new Property(propertyName, false));
083    return query;
084  }
085
086  /**
087   * Return true if the property is known to be contained in the order by clause.
088   */
089  public boolean containsProperty(String propertyName) {
090
091    for (Property aList : list) {
092      if (propertyName.equals(aList.getProperty())) {
093        return true;
094      }
095    }
096    return false;
097  }
098
099  /**
100   * Return a copy of this OrderBy with the path trimmed.
101   */
102  public OrderBy<T> copyWithTrim(String path) {
103    List<Property> newList = new ArrayList<>(list.size());
104    for (Property aList : list) {
105      newList.add(aList.copyWithTrim(path));
106    }
107    return new OrderBy<>(newList);
108  }
109
110  /**
111   * Return the properties for this OrderBy.
112   */
113  public List<Property> getProperties() {
114    // not returning an Immutable list at this point
115    return list;
116  }
117
118  /**
119   * Return true if this OrderBy does not have any properties.
120   */
121  public boolean isEmpty() {
122    return list.isEmpty();
123  }
124
125  /**
126   * Return the associated query if there is one.
127   */
128  public Query<T> getQuery() {
129    return query;
130  }
131
132  /**
133   * Associate this OrderBy with a query.
134   */
135  public void setQuery(Query<T> query) {
136    this.query = query;
137  }
138
139  /**
140   * Return a copy of the OrderBy.
141   */
142  public OrderBy<T> copy() {
143
144    OrderBy<T> copy = new OrderBy<>();
145    for (Property aList : list) {
146      copy.add(aList.copy());
147    }
148    return copy;
149  }
150
151  /**
152   * Add to the order by by parsing a raw expression.
153   */
154  public void add(String rawExpression) {
155    parse(rawExpression);
156  }
157
158  /**
159   * Add a property to the order by.
160   */
161  public void add(Property p) {
162    list.add(p);
163  }
164
165  @Override
166  public String toString() {
167    return list.toString();
168  }
169
170  /**
171   * Returns the OrderBy in string format.
172   */
173  public String toStringFormat() {
174    if (list.isEmpty()) {
175      return null;
176    }
177    StringBuilder sb = new StringBuilder();
178    for (int i = 0; i < list.size(); i++) {
179      Property property = list.get(i);
180      if (i > 0) {
181        sb.append(", ");
182      }
183      sb.append(property.toStringFormat());
184    }
185    return sb.toString();
186  }
187
188  @Override
189  public boolean equals(Object obj) {
190    if (obj == this) {
191      return true;
192    }
193    if (!(obj instanceof OrderBy<?>)) {
194      return false;
195    }
196
197    OrderBy<?> e = (OrderBy<?>) obj;
198    return e.list.equals(list);
199  }
200
201  /**
202   * Return a hash value for this OrderBy. This can be to determine logical
203   * equality for OrderBy clauses.
204   */
205  @Override
206  public int hashCode() {
207    return list.hashCode();
208  }
209
210  /**
211   * Clear the orderBy removing any current order by properties.
212   * <p>
213   * This is intended to be used when some code creates a query with a
214   * 'default' order by clause and some other code may clear the 'default'
215   * order by clause and replace.
216   * </p>
217   */
218  public OrderBy<T> clear() {
219    list.clear();
220    return this;
221  }
222
223  /**
224   * A property and its ascending descending order.
225   */
226  public static final class Property implements Serializable {
227
228    private static final long serialVersionUID = 1546009780322478077L;
229
230    private String property;
231
232    private boolean ascending;
233
234    private String nulls;
235
236    private String highLow;
237
238    public Property(String property, boolean ascending) {
239      this.property = property;
240      this.ascending = ascending;
241    }
242
243    public Property(String property, boolean ascending, String nulls, String highLow) {
244      this.property = property;
245      this.ascending = ascending;
246      this.nulls = nulls;
247      this.highLow = highLow;
248    }
249
250    /**
251     * Return a copy of this Property with the path trimmed.
252     */
253    public Property copyWithTrim(String path) {
254      return new Property(property.substring(path.length() + 1), ascending, nulls, highLow);
255    }
256
257    @Override
258    public int hashCode() {
259      int hc = property.hashCode();
260      hc = hc * 92821 + (ascending ? 0 : 1);
261      hc = hc * 92821 + (nulls == null ? 0 : nulls.hashCode());
262      hc = hc * 92821 + (highLow == null ? 0 : highLow.hashCode());
263      return hc;
264    }
265
266    @Override
267    public boolean equals(Object obj) {
268      if (obj == this) {
269        return true;
270      }
271      if (!(obj instanceof Property)) {
272        return false;
273      }
274      Property e = (Property) obj;
275      if (ascending != e.ascending) return false;
276      if (!property.equals(e.property)) return false;
277      if (nulls != null ? !nulls.equals(e.nulls) : e.nulls != null) return false;
278      return highLow != null ? highLow.equals(e.highLow) : e.highLow == null;
279    }
280
281    @Override
282    public String toString() {
283      return toStringFormat();
284    }
285
286    public String toStringFormat() {
287      if (nulls == null) {
288        if (ascending) {
289          return property;
290        } else {
291          return property + " desc";
292        }
293      } else {
294        StringBuilder sb = new StringBuilder();
295        sb.append(property);
296        if (!ascending) {
297          sb.append(" ").append("desc");
298        }
299        sb.append(" ").append(nulls).append(" ").append(highLow);
300        return sb.toString();
301      }
302    }
303
304    /**
305     * Reverse the ascending/descending order for this property.
306     */
307    public void reverse() {
308      this.ascending = !ascending;
309    }
310
311    /**
312     * Trim off the pathPrefix.
313     */
314    public void trim(String pathPrefix) {
315      property = property.substring(pathPrefix.length() + 1);
316    }
317
318    /**
319     * Return a copy of this property.
320     */
321    public Property copy() {
322      return new Property(property, ascending, nulls, highLow);
323    }
324
325    /**
326     * Return the property name.
327     */
328    public String getProperty() {
329      return property;
330    }
331
332    /**
333     * Set the property name.
334     */
335    public void setProperty(String property) {
336      this.property = property;
337    }
338
339    /**
340     * Return true if the order is ascending.
341     */
342    public boolean isAscending() {
343      return ascending;
344    }
345
346    /**
347     * Set to true if the order is ascending.
348     */
349    public void setAscending(boolean ascending) {
350      this.ascending = ascending;
351    }
352
353  }
354
355  private void parse(String orderByClause) {
356
357    if (orderByClause == null) {
358      return;
359    }
360
361    String[] chunks = orderByClause.split(",");
362    for (String chunk : chunks) {
363      String[] pairs = chunk.split(" ");
364      Property p = parseProperty(pairs);
365      if (p != null) {
366        list.add(p);
367      }
368    }
369  }
370
371  private Property parseProperty(String[] pairs) {
372    if (pairs.length == 0) {
373      return null;
374    }
375
376    ArrayList<String> wordList = new ArrayList<>(pairs.length);
377    for (String pair : pairs) {
378      if (!isEmptyString(pair)) {
379        wordList.add(pair);
380      }
381    }
382    if (wordList.isEmpty()) {
383      return null;
384    }
385    if (wordList.size() == 1) {
386      return new Property(wordList.get(0), true);
387    }
388    if (wordList.size() == 2) {
389      boolean asc = isAscending(wordList.get(1));
390      return new Property(wordList.get(0), asc);
391    }
392    if (wordList.size() == 4) {
393      // nulls high or nulls low as 3rd and 4th
394      boolean asc = isAscending(wordList.get(1));
395      return new Property(wordList.get(0), asc, wordList.get(2), wordList.get(3));
396    }
397    String m = "Expecting a 1, 2 or 4 words in [" + Arrays.toString(pairs) + "] but got " + wordList;
398    throw new RuntimeException(m);
399  }
400
401  private boolean isAscending(String s) {
402    s = s.toLowerCase();
403    if (s.startsWith("asc")) {
404      return true;
405    }
406    if (s.startsWith("desc")) {
407      return false;
408    }
409    String m = "Expecting [" + s + "] to be asc or desc?";
410    throw new RuntimeException(m);
411  }
412
413  private boolean isEmptyString(String s) {
414    return s == null || s.isEmpty();
415  }
