001package io.ebean;
002
003/**
004 * An update query typically intended to perform a bulk update of many rows that match the query.
005 * <p>
006 * Also note that you can also just use a raw SQL update via {@link SqlUpdate} which is pretty light and simple.
007 * This UpdateQuery is more for the cases where we want to build the where expression of the update using the
008 * {@link ExpressionList} "Criteria API" that is used with a normal ORM query.
009 * </p>
010 * <p>
011 * <h4>Example: Simple update</h4>
012 * <p>
013 * <pre>{@code
014 *
015 *  int rows = ebeanServer
016 *      .update(Customer.class)
017 *      .set("status", Customer.Status.ACTIVE)
018 *      .set("updtime", new Timestamp(System.currentTimeMillis()))
019 *      .where()
020 *      .gt("id", 1000)
021 *      .update();
022 *
023 * }</pre>
024 * <pre>{@code sql
025 *
026 *   update o_customer set status=?, updtime=? where id > ?
027 *
028 * }</pre>
029 * <p>
030 * Note that if the where() clause contains a join then the SQL update changes to use a
031 * <code> WHERE ID IN () </code> form.
032 * </p>
033 * <p>
034 * <h4>Example: Update with a JOIN</h4>
035 * <p>
036 * In this example the expression <code>.eq("billingAddress.country", nz)</code> requires a join
037 * to the address table.
038 * </p>
039 * <p>
040 * <pre>{@code
041 *
042 *   int rows = ebeanServer
043 *       .update(Customer.class)
044 *       .set("status", Customer.Status.ACTIVE)
045 *       .set("updtime", new Timestamp(System.currentTimeMillis()))
046 *       .where()
047 *         .eq("status", Customer.Status.NEW)
048 *         .eq("billingAddress.country", nz)
049 *         .gt("id", 1000)
050 *         .update();
051 * }</pre>
052 * <p>
053 * <pre>{@code sql
054 *
055 *   update o_customer set status=?, updtime=?
056 *   where id in (
057 *     select t0.id c0
058 *     from o_customer t0
059 *     left join o_address t1 on t1.id = t0.billing_address_id
060 *     where t0.status = ?
061 *       and t1.country_code = ?
062 *       and t0.id > ? )
063 *
064 * }</pre>
065 *
066 * @param <T> The type of entity bean being updated
067 * @see SqlUpdate
068 */
069public interface UpdateQuery<T> {
070
071  /**
072   * Set the value of a property.
073   * <p>
074   * <pre>{@code
075   *
076   *   int rows = ebeanServer
077   *      .update(Customer.class)
078   *      .set("status", Customer.Status.ACTIVE)
079   *      .set("updtime", new Timestamp(System.currentTimeMillis()))
080   *      .where()
081   *      .gt("id", 1000)
082   *      .update();
083   *
084   * }</pre>
085   *
086   * @param property The bean property to be set
087   * @param value    The value to set the property to
088   */
089  UpdateQuery<T> set(String property, Object value);
090
091  /**
092   * Set the property to be null.
093   * <p>
094   * <pre>{@code
095   *
096   *   int rows = ebeanServer
097   *      .update(Customer.class)
098   *      .setNull("notes")
099   *      .where()
100   *      .gt("id", 1000)
101   *      .update();
102   *
103   * }</pre>
104   *
105   * @param property The property to be set to null.
106   */
107  UpdateQuery<T> setNull(String property);
108
109  /**
110   * Set using a property expression that does not need any bind values.
111   * <p>
112   * The property expression typically contains database functions.
113   * </p>
114   * <p>
115   * <pre>{@code
116   *
117   *   int rows = ebeanServer
118   *      .update(Customer.class)
119   *      .setRaw("status = coalesce(status, 'A')")
120   *      .where()
121   *      .gt("id", 1000)
122   *      .update();
123   *
124   * }</pre>
125   *
126   * @param propertyExpression A property expression
127   */
128  UpdateQuery<T> setRaw(String propertyExpression);
129
130  /**
131   * Set using a property expression that can contain <code>?</code> bind value placeholders.
132   * <p>
133   * For each <code>?</code> in the property expression there should be a matching bind value supplied.
134   * </p>
135   * <pre>{@code
136   *
137   *   int rows = ebeanServer
138   *      .update(Customer.class)
139   *      .setRaw("status = coalesce(status, ?)", Customer.Status.ACTIVE)
140   *      .where()
141   *      .gt("id", 1000)
142   *      .update();
143   *
144   * }</pre>
145   *
146   * @param propertyExpression A raw property expression
147   * @param values             The values to bind with the property expression
148   */
149  UpdateQuery<T> setRaw(String propertyExpression, Object... values);
150
151  /**
152   * Set the profile location of this update query. This is used to relate query execution metrics
153   * back to a location like a specific line of code.
154   */
155  UpdateQuery<T> setProfileLocation(ProfileLocation profileLocation);
156
157  /**
158   * Return the query expression list to add predicates to.
159   */
160  ExpressionList<T> where();
161
162}
