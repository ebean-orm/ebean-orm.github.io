001package io.ebean.config.dbplatform;
002
003import io.ebean.config.PlatformConfig;
004
005import java.util.EnumMap;
006import java.util.Map;
007
008/**
009 * Used to map bean property types to DB specific types for DDL generation.
010 */
011public class DbPlatformTypeMapping {
012
013  /**
014   * Boolean type used for logical model.
015   */
016  private static class BooleanLogicalType extends DbPlatformType {
017    BooleanLogicalType() {
018      super("boolean", false);
019    }
020    @Override
021    protected void renderLengthScale(int deployLength, int deployScale, StringBuilder sb) {
022      // do not have length - even if platform boolean type does like integer(1)
023    }
024  }
025
026  private static DbPlatformTypeLookup lookup = new DbPlatformTypeLookup();
027
028  private static final DbPlatformType BOOLEAN_LOGICAL = new BooleanLogicalType();
029
030  private static final DbPlatformType UUID_NATIVE = new DbPlatformType("uuid", false);
031  @SuppressWarnings("unused")
032  private static final DbPlatformType UUID_PLACEHOLDER = new DbPlatformType("uuidPlaceholder");
033  private static final DbPlatformType JSON_CLOB_PLACEHOLDER = new DbPlatformType("jsonClobPlaceholder");
034  private static final DbPlatformType JSON_BLOB_PLACEHOLDER = new DbPlatformType("jsonBlobPlaceholder");
035  private static final DbPlatformType JSON_VARCHAR_PLACEHOLDER = new DbPlatformType("jsonVarcharPlaceholder");
036
037  private static final DbPlatformType POINT = new DbPlatformType("point");
038  private static final DbPlatformType POLYGON = new DbPlatformType("polygon");
039  private static final DbPlatformType LINESTRING = new DbPlatformType("linestring");
040  private static final DbPlatformType MULTIPOINT = new DbPlatformType("multipoint");
041  private static final DbPlatformType MULTILINESTRING = new DbPlatformType("multilinestring");
042  private static final DbPlatformType MULTIPOLYGON = new DbPlatformType("multipolygon");
043
044  private final Map<DbType, DbPlatformType> typeMap = new EnumMap<>(DbType.class);
045
046  /**
047   * Return the DbTypeMap with standard (not platform specific) types.
048   * <p>
049   * This has some extended JSON types (JSON, JSONB, JSONVarchar, JSONClob, JSONBlob).
050   * These types get translated to specific database platform types during DDL generation.
051   */
052  public static DbPlatformTypeMapping logicalTypes() {
053    return new DbPlatformTypeMapping(true);
054  }
055
056  public DbPlatformTypeMapping() {
057    loadDefaults(false);
058  }
059
060  private DbPlatformTypeMapping(boolean logicalTypes) {
061    loadDefaults(logicalTypes);
062  }
063
064  /**
065   * Load the standard types. These can be overridden by DB specific platform.
066   */
067  private void loadDefaults(boolean logicalTypes) {
068
069    put(DbType.BOOLEAN, BOOLEAN_LOGICAL);
070    put(DbType.BIT);
071    put(DbType.INTEGER);
072    put(DbType.BIGINT);
073    put(DbType.DOUBLE);
074    put(DbType.SMALLINT);
075    put(DbType.TINYINT);
076    put(DbType.BLOB);
077    put(DbType.CLOB);
078    put(DbType.ARRAY);
079    put(DbType.DATE);
080    put(DbType.TIME);
081    put(DbType.TIMESTAMP);
082    put(DbType.LONGVARBINARY);
083    put(DbType.LONGVARCHAR);
084    // most commonly real maps to db float
085    put(DbType.REAL, new DbPlatformType("float"));
086
087    put(DbType.POINT, POINT);
088    put(DbType.POLYGON, POLYGON);
089    put(DbType.LINESTRING, LINESTRING);
090    put(DbType.MULTIPOINT, MULTIPOINT);
091    put(DbType.MULTILINESTRING, MULTILINESTRING);
092    put(DbType.MULTIPOLYGON, MULTIPOLYGON);
093
094    if (logicalTypes) {
095      // keep it logical for 2 layer DDL generation
096      put(DbType.VARCHAR, new DbPlatformType("varchar"));
097      put(DbType.DECIMAL, new DbPlatformType("decimal"));
098      put(DbType.VARBINARY, new DbPlatformType("varbinary"));
099      put(DbType.BINARY, new DbPlatformType("binary"));
100      put(DbType.CHAR, new DbPlatformType("char"));
101
102      put(DbType.HSTORE, new DbPlatformType("hstore", false));
103      put(DbType.JSON, new DbPlatformType("json", false));
104      put(DbType.JSONB, new DbPlatformType("jsonb", false));
105      put(DbType.JSONCLOB, new DbPlatformType("jsonclob"));
106      put(DbType.JSONBLOB, new DbPlatformType("jsonblob"));
107      put(DbType.JSONVARCHAR, new DbPlatformType("jsonvarchar", 1000));
108      put(DbType.UUID, UUID_NATIVE);
109
110    } else {
111      put(DbType.VARCHAR, new DbPlatformType("varchar", 255));
112      put(DbType.DECIMAL, new DbPlatformType("decimal", 38));
113      put(DbType.VARBINARY, new DbPlatformType("varbinary", 255));
114      put(DbType.BINARY, new DbPlatformType("binary", 255));
115      put(DbType.CHAR, new DbPlatformType("char", 1));
116
117      put(DbType.JSON, JSON_CLOB_PLACEHOLDER); // Postgres maps this to JSON
118      put(DbType.JSONB, JSON_CLOB_PLACEHOLDER); // Postgres maps this to JSONB
119      put(DbType.JSONCLOB, JSON_CLOB_PLACEHOLDER);
120      put(DbType.JSONBLOB, JSON_BLOB_PLACEHOLDER);
121      put(DbType.JSONVARCHAR, JSON_VARCHAR_PLACEHOLDER);
122      // default to native UUID and override on platform configure()
123      put(DbType.UUID, UUID_NATIVE);
124    }
125  }
126
127  /**
128   * Lookup the platform specific DbType given the standard sql type name.
129   */
130  public DbPlatformType lookup(String name, boolean withScale) {
131
132    DbType type = lookup.byName(name);
133    if (type == null) {
134      throw new IllegalArgumentException("Unknown type [" + name + "] - not standard sql type");
135    }
136    // handle JSON types mapped to clob, blob and varchar
137    switch (type) {
138      case JSONBLOB:
139        return get(DbType.BLOB);
140      case JSONCLOB:
141        return get(DbType.CLOB);
142      case JSONVARCHAR:
143        return get(DbType.VARCHAR);
144      case JSON:
145        return getJsonType(DbType.JSON, withScale);
146      case JSONB:
147        return getJsonType(DbType.JSONB, withScale);
148      default:
149        return get(type);
150    }
151  }
152
153  private DbPlatformType getJsonType(DbType type, boolean withScale) {
154
155    DbPlatformType dbType = get(type);
156    if (dbType == JSON_CLOB_PLACEHOLDER) {
157      // if we have scale that implies this maps to varchar
158      return withScale ? get(DbType.VARCHAR) : get(DbType.CLOB);
159    }
160    if (dbType == JSON_BLOB_PLACEHOLDER) {
161      return get(DbType.BLOB);
162    }
163    if (dbType == JSON_VARCHAR_PLACEHOLDER) {
164      return get(DbType.VARCHAR);
165    }
166    // Postgres has specific type
167    return get(type);
168  }
169
170  /**
171   * Override the type for a given JDBC type.
172   */
173  private void put(DbType type) {
174    typeMap.put(type, type.createPlatformType());
175  }
176
177  /**
178   * Override the type for a given JDBC type.
179   */
180  public void put(DbType type, DbPlatformType platformType) {
181    typeMap.put(type, platformType);
182  }
183
184  /**
185   * Return the type for a given jdbc type.
186   */
187  public DbPlatformType get(int jdbcType) {
188    return get(lookup.byId(jdbcType));
189  }
190
191  /**
192   * Return the type for a given jdbc type.
193   */
194  public DbPlatformType get(DbType dbType) {
195    return typeMap.get(dbType);
196  }
197
198  /**
199   * Map the UUID appropriately based on native DB support and ServerConfig.DbUuid.
200   */
201  public void config(boolean nativeUuidType, PlatformConfig.DbUuid dbUuid) {
202    if (nativeUuidType && dbUuid.useNativeType()) {
203      // native UUID already set by default
204    } else if (dbUuid.useBinary()) {
205      put(DbType.UUID, get(DbType.BINARY).withLength(16));
206    } else {
207      put(DbType.UUID, get(DbType.VARCHAR).withLength(40));
208    }
209  }
