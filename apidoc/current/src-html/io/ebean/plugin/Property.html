001package io.ebean.plugin;
002
003/**
004 * Property of a entity bean that can be read.
005 */
006public interface Property {
007
008  /**
009   * Return the name of the property.
010   */
011  String getName();
012
013  /**
014   * Return the value of the property on the given bean.
015   */
016  Object getVal(Object bean);
017
018  /**
019   * Return true if this is a OneToMany or ManyToMany property.
020   */
021  boolean isMany();
022}
