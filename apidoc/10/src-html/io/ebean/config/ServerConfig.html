051 * classes and listeners etc.
052 * </p>
053 * <pre>{@code
054 *
055 * ServerConfig c = new ServerConfig();
056 * c.setName("db");
057 *
058 * // read the ebean.properties and load
059 * // those settings into this serverConfig object
060 * c.loadFromProperties();
061 *
062 * // add any classes found in the app.data package
063 * c.addPackage("com.myapp.domain");

067 *
068 * EbeanServer server = EbeanServerFactory.create(c);

2595   * @param properties the properties
2596   * @return the classes

2598  private List<Class<?>> getClasses(PropertiesWrapper properties) {
2599
2600    String classNames = properties.get("classes", null);
2601    if (classNames == null) {
2602      return classes;
2603    }
2604
2605    List<Class<?>> classes = new ArrayList<>();
2606
2607    String[] split = classNames.split("[ ,;]");
2608    for (String aSplit : split) {
2609      String cn = aSplit.trim();
2610      if (!cn.isEmpty() && !"class".equalsIgnoreCase(cn)) {
2611        try {
2612          classes.add(Class.forName(cn));
2613        } catch (ClassNotFoundException e) {
2614          String msg = "Error registering class [" + cn + "] from [" + classNames + "]";
2615          throw new RuntimeException(msg, e);
2616        }
2617      }
2618    }
2619    return classes;
2620  }
