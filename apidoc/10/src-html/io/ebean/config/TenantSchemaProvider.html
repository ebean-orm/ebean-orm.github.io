001package io.ebean.config;
002
003/**
004 * For multi-tenancy via DB SCHEMA supply the schema given the tenantId.
005 */
006public interface TenantSchemaProvider {
007
008  /**
009   * Return the DB schema for the given tenantId.
010   *
011   * @param tenantId The current tenant id.
012   * @return The DB schema to use for the given tenant
013   */
014  String schema(Object tenantId);
015}
