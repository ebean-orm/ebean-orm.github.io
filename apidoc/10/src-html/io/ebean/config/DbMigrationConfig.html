001package io.ebean.config;

003import io.ebean.Platform;
004import io.ebean.dbmigration.MigrationConfig;

011/**
012 * Configuration for the DB migration processing.
013 */
014public class DbMigrationConfig {
015
016  protected static final Logger logger = LoggerFactory.getLogger(DbMigrationConfig.class);
017
018  /**
019   * The database platform to generate migration DDL for.
020   */
021  protected Platform platform;
022
023  /**
024   * Set to true if the DB migration should be generated on server start.
025   */
026  protected boolean generate;
027
028  /**
029   * The migration version name (typically FlywayDb compatible).
030   * <p>
031   * Example: 1.1.1_2
032   * <p>
033   * The version is expected to be the combination of the current pom version plus
034   * a 'feature' id. The combined version must be unique and ordered to work with
035   * FlywayDb so each developer sets a unique version so that the migration script
036   * generated is unique (typically just prior to being submitted as a merge request).
037   */
038  protected String version;
039
040  /**
041   * Description text that can be appended to the version to become the ddl script file name.
042   * <p>
043   * So if the name is "a foo table" then the ddl script file could be:
044   * "1.1.1_2__a-foo-table.sql"
045   * <p>
046   * When the DB migration relates to a git feature (merge request) then this description text
047   * is a short description of the feature.
048   */
049  protected String name;
050
051  /**
052   * Resource path for the migration xml and sql.
053   */
054  protected String migrationPath = "dbmigration";
055
056  /**
057   * Subdirectory the model xml files go into.
058   */
059  protected String modelPath = "model";
060
061  protected String applySuffix = ".sql";
062
063  /**
064   * Set this to "V" to be compatible with FlywayDB.
065   */
066  protected String applyPrefix = "";
067
068  protected String modelSuffix = ".model.xml";
069
070  protected boolean includeGeneratedFileComment;
071
072  /**
073   * The version of a pending drop that should be generated as the next migration.
074   */
075  protected String generatePendingDrop;
076
077  /**
078   * For running migration the DB table that holds migration execution status.
079   */
080  protected String metaTable = "db_migration";
081
082  /**
083   * Flag set to true means to run any outstanding migrations on startup.
084   */
085  protected boolean runMigration;
086
087  /**
088   * Comma and equals delimited key/value placeholders to replace in DDL scripts.
089   */
090  protected String runPlaceholders;
091
092  /**
093   * Map of key/value placeholders to replace in DDL scripts.
094   */
095  protected Map<String, String> runPlaceholderMap;
096
097  /**
098   * DB user used to run the DB migration.
099   */
100  protected String dbUsername;
101
102  /**
103   * DB password used to run the DB migration.
104   */
105  protected String dbPassword;
106

298  /**
299   * Return true if the DB migration should be run on startup.
300   */
301  public boolean isRunMigration() {
302    // environment properties take precedence
303    String run = readEnvironment("ddl.migration.run");
304    if (run != null) {
305      return "true".equalsIgnoreCase(run.trim());
306    }
307    return runMigration;
308  }

354  /**
355   * Load the settings from the PropertiesWrapper.
356   */
357  public void loadSettings(PropertiesWrapper properties, String serverName) {
358
359    migrationPath = properties.get("migration.migrationPath", migrationPath);
360    modelPath = properties.get("migration.modelPath", modelPath);
361    applyPrefix = properties.get("migration.applyPrefix", applyPrefix);
362    applySuffix = properties.get("migration.applySuffix", applySuffix);
363    modelSuffix = properties.get("migration.modelSuffix", modelSuffix);
364    includeGeneratedFileComment = properties.getBoolean("migration.includeGeneratedFileComment", includeGeneratedFileComment);
365    generatePendingDrop = properties.get("migration.generatePendingDrop", generatePendingDrop);
366
367    platform = properties.getEnum(Platform.class, "migration.platform", platform);
368
369    generate = properties.getBoolean("migration.generate", generate);
370    version = properties.get("migration.version", version);
371    name = properties.get("migration.name", name);
372
373    runMigration = properties.getBoolean("migration.run", runMigration);
374    metaTable = properties.get("migration.metaTable", metaTable);
375    runPlaceholders = properties.get("migration.placeholders", runPlaceholders);
376
377    String adminUser = properties.get("datasource." + serverName + ".username", dbUsername);
378    adminUser = properties.get("datasource." + serverName + ".adminusername", adminUser);
379    dbUsername = properties.get("migration.dbusername", adminUser);
380
381    String adminPwd = properties.get("datasource." + serverName + ".password", dbPassword);
382    adminPwd = properties.get("datasource." + serverName + ".adminpassword", adminPwd);
383    dbPassword = properties.get("migration.dbpassword", adminPwd);
384  }

386  /**
387   * Return the migration version (typically FlywayDb compatible).
388   * <p>
389   * Example: 1.1.1_2
390   * <p>
391   * The version is expected to be the combination of the current pom version plus
392   * a 'feature' id. The combined version must be unique and ordered to work with
393   * FlywayDb so each developer sets a unique version so that the migration script
394   * generated is unique (typically just prior to being submitted as a merge request).
395   */
396  public String getVersion() {
397    String envVersion = readEnvironment("ddl.migration.version");
398    if (!isEmpty(envVersion)) {
399      return envVersion.trim();
400    }
401    return version;
402  }
403
404  /**
405   * Return the migration name which is short description text that can be appended to
406   * the migration version to become the ddl script file name.
407   * <p>
408   * So if the name is "a foo table" then the ddl script file could be:
409   * "1.1.1_2__a-foo-table.sql"
410   * </p>
411   * <p>
412   * When the DB migration relates to a git feature (merge request) then this description text
413   * is a short description of the feature.
414   * </p>
415   */
416  public String getName() {
417    String envName = readEnvironment("ddl.migration.name");
418    if (!isEmpty(envName)) {
419      return envName.trim();
420    }
421    return name;
422  }
423
424  /**
425   * Return the system or environment property.
426   */
427  protected String readEnvironment(String key) {
428
429    String val = System.getProperty(key);
430    if (val == null) {
431      val = System.getenv(key);
432    }
433    return val;
434  }
435
436  /**
437   * Return true if the string is null or empty.
438   */
439  protected boolean isEmpty(String val) {
440    return val == null || val.trim().isEmpty();
441  }

443  /**
444   * Create the MigrationRunner to run migrations if necessary.
445   */
446  public MigrationRunner createRunner(ClassLoader classLoader) {
447
448    MigrationConfig runnerConfig = new MigrationConfig();
449    runnerConfig.setMetaTable(metaTable);
450    runnerConfig.setApplySuffix(applySuffix);
451    runnerConfig.setMigrationPath(migrationPath);
452    runnerConfig.setRunPlaceholderMap(runPlaceholderMap);
453    runnerConfig.setRunPlaceholders(runPlaceholders);
454    runnerConfig.setDbUsername(getDbUsername());
455    runnerConfig.setDbPassword(getDbPassword());
456    runnerConfig.setClassLoader(classLoader);
457    return new MigrationRunner(runnerConfig);
458  }
