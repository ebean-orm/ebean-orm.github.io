001package io.ebean.util;
002
003public class CamelCaseHelper {
004
005  /**
006   * To camel from underscore.
007   *
008   * @param underscore the underscore
009   * @return the string
010   */
011  public static String toCamelFromUnderscore(String underscore) {
012
013    String[] vals = underscore.split("_");
014    if (vals.length == 1) {
015      return underscore;
016    }
017
018    StringBuilder result = new StringBuilder();
019    for (int i = 0; i < vals.length; i++) {
020      String lower = vals[i].toLowerCase();
021      if (i > 0) {
022        char c = Character.toUpperCase(lower.charAt(0));
023        result.append(c);
024        result.append(lower.substring(1));
025      } else {
026        result.append(lower);
027      }
028    }
029
030    return result.toString();
031  }
032}
