016  /**
017   * parses a String of the form name1='value1' name2='value2'. Note that you
018   * can use either single or double quotes for any particular name value pair
019   * and the end quote must match the begin quote.
020   */
021  public static HashMap<String, String> parseNameQuotedValue(String tag) throws RuntimeException {
022
023    if (tag == null || tag.length() < 1) {
024      return null;
025    }
026
027    // make sure that the quotes are matched...
028    // int remainer = countOccurances(tag, ""+quote) % 2;
029    // if (remainer == 1) {
030    // dp("remainder = "+remainer);
031    // throw new StringParsingException("Unmatched quote in "+tag);
032    // }
033
034    // make sure that th last character is not an equals...
035    // (check now so I don't need to check this every time..)
036    if (tag.charAt(tag.length() - 1) == '=') {
037      throw new RuntimeException("missing quoted value at the end of " + tag);
038    }
039
040    HashMap<String, String> map = new HashMap<>();
041    // recursively parse out the name value pairs...
042    return parseNameQuotedValue(map, tag, 0);
043  }

045  /**
046   * recursively parse out name value pairs (where the value is quoted, with
047   * either single or double quotes).
048   */
049  private static HashMap<String, String> parseNameQuotedValue(HashMap<String, String> map,
050                                                              String tag, int pos) throws RuntimeException {
051
052    int equalsPos = tag.indexOf("=", pos);
053    if (equalsPos > -1) {
054      // check for begin quote...
055      char firstQuote = tag.charAt(equalsPos + 1);
056      if (firstQuote != SINGLE_QUOTE && firstQuote != DOUBLE_QUOTE) {
057        throw new RuntimeException("missing begin quote at " + (equalsPos) + "["
058          + tag.charAt(equalsPos + 1) + "] in [" + tag + "]");
059      }
060
061      // check for end quote...
062      int endQuotePos = tag.indexOf(firstQuote, equalsPos + 2);
063      if (endQuotePos == -1) {
064        throw new RuntimeException("missing end quote [" + firstQuote + "] after " + pos + " in [" + tag + "]");
065      }
066
067      // we have a valid name and value...
068      // dp("pos="+pos+" equalsPos="+equalsPos+"
069      // endQuotePos="+endQuotePos);
070      String name = tag.substring(pos, equalsPos);
071      String value = tag.substring(equalsPos + 2, endQuotePos);
072      // dp("name="+name+"; value="+value+";");
073
074      // trim off any whitespace from the front of name...
075      name = trimFront(name, " ");
076      if ((name.indexOf(SINGLE_QUOTE) > -1) || (name.indexOf(DOUBLE_QUOTE) > -1)) {
077        throw new RuntimeException("attribute name contains a quote [" + name + "]");
078      }
079      map.put(name, value);
080
081      return parseNameQuotedValue(map, tag, endQuotePos + 1);
082
083    } else {
084      // no more equals... stop parsing...
085      return map;
086    }
087  }

089  /**
090   * Returns the number of times a particular String occurs in another String.
091   * e.g. count the number of single quotes.
092   */
093  public static int countOccurances(String content, String occurs) {
094    return countOccurances(content, occurs, 0, 0);
095  }
096
097  private static int countOccurances(String content, String occurs, int pos, int countSoFar) {
098    int equalsPos = content.indexOf(occurs, pos);
099    if (equalsPos > -1) {
100      countSoFar = countSoFar + 1;
101      pos = equalsPos + occurs.length();
102      // dp("countSoFar="+countSoFar+" pos="+pos);
103      return countOccurances(content, occurs, pos, countSoFar);
104    } else {
105      return countSoFar;
106    }
107  }

109  /**
110   * Parses out a list of Name Value pairs that are delimited together. Will
111   * always return a StringMap. If allNameValuePairs is null, or no name values
112   * can be parsed out an empty StringMap is returned.
113   *
114   * @param allNameValuePairs  the entire string to be parsed.
115   * @param listDelimiter      (typically ';') the delimited between the list
116   * @param nameValueSeparator (typically '=') the separator between the name and value
117   */
118  public static Map<String, String> delimitedToMap(String allNameValuePairs,
119                                                   String listDelimiter, String nameValueSeparator) {
120
121    HashMap<String, String> params = new HashMap<>();
122    if ((allNameValuePairs == null) || (allNameValuePairs.isEmpty())) {
123      return params;
124    }
125    // trim off any leading listDelimiter...
126    allNameValuePairs = trimFront(allNameValuePairs, listDelimiter);
127    return getKeyValue(params, 0, allNameValuePairs, listDelimiter, nameValueSeparator);
128  }

130  /**
131   * Trims off recurring strings from the front of a string.
132   *
133   * @param source the source string
134   * @param trim   the string to trim off the front
135   */
136  public static String trimFront(String source, String trim) {
137    if (source == null) {
138      return null;
139    }
140    if (source.indexOf(trim) == 0) {
141      // dp("trim ...");
142      return trimFront(source.substring(trim.length()), trim);
143    } else {
144      return source;
145    }
146  }

155  /**
156   * Recursively pulls out the key value pairs from a raw string.
157   */
158  private static HashMap<String, String> getKeyValue(HashMap<String, String> map, int pos,
159                                                     String allNameValuePairs, String listDelimiter, String nameValueSeparator) {
160
161    if (pos >= allNameValuePairs.length()) {
162      // dp("end as "+pos+" >= "+allNameValuePairs.length() );
163      return map;
164    }
165
166    int equalsPos = allNameValuePairs.indexOf(nameValueSeparator, pos);
167    int delimPos = allNameValuePairs.indexOf(listDelimiter, pos);
168
169    if (delimPos == -1) {
170      delimPos = allNameValuePairs.length();
171    }
172    if (equalsPos == -1) {
173      // dp("no more equals...");
174      return map;
175    }
176    if (delimPos == (equalsPos + 1)) {
177      // dp("Ignoring as nothing between delim and equals...
178      // delim:"+delimPos+" eq:"+equalsPos);
179      return getKeyValue(map, delimPos + 1, allNameValuePairs, listDelimiter,
180        nameValueSeparator);
181    }
182    if (equalsPos > delimPos) {
183      // there is a key without a value?
184      String key = allNameValuePairs.substring(pos, delimPos);
185      key = key.trim();
186      if (!key.isEmpty()) {
187        map.put(key, null);
188      }
189      return getKeyValue(map, delimPos + 1, allNameValuePairs, listDelimiter,
190        nameValueSeparator);
191
192    }
193    String key = allNameValuePairs.substring(pos, equalsPos);
194
195    if (delimPos > -1) {
196      String value = allNameValuePairs.substring(equalsPos + 1, delimPos);
197      // dp("cont "+key+","+value+" pos:"+pos+"
198      // len:"+allNameValuePairs.length());
199      key = key.trim();
200
201      map.put(key, value);
202      pos = delimPos + 1;
203
204      // recurse the rest of the values...
205      return getKeyValue(map, pos, allNameValuePairs, listDelimiter, nameValueSeparator);
206    } else {
207      // dp("ERROR: delimPos < 0 ???");
208      return map;
209    }
210  }

212  /**
213   * Convert a string that has delimited values (say comma delimited) in a
214   * String[]. You must explicitly choose whether or not to include empty values
215   * (say two commas that a right beside each other.
216   * <p>
217   * e.g. "alpha,beta,,theta"<br>
218   * With keepEmpties true, this results in a String[] of size 4 with the third
219   * one having a String of 0 length. With keepEmpties false, this results in a
220   * String[] of size 3.
221   * <p>
222   * e.g. ",alpha,beta,,theta,"<br>
223   * With keepEmpties true, this results in a String[] of size 6 with the
224   * 1st,4th and 6th one having a String of 0 length. With keepEmpties false,
225   * this results in a String[] of size 3.
226   */
227  public static String[] delimitedToArray(String str, String delimiter, boolean keepEmpties) {
228
229    ArrayList<String> list = new ArrayList<>();
230    int startPos = 0;
231    delimiter(str, delimiter, keepEmpties, startPos, list);
232    String[] result = new String[list.size()];
233    return list.toArray(result);
234  }
235
236  private static void delimiter(String str, String delimiter, boolean keepEmpties, int startPos,
237                                ArrayList<String> list) {
238
239    int endPos = str.indexOf(delimiter, startPos);
240    if (endPos == -1) {
241      if (startPos <= str.length()) {
242        String lastValue = str.substring(startPos, str.length());
243        if (keepEmpties || !lastValue.isEmpty()) {
244          list.add(lastValue);
245        }
246      }
247      // we have finished parsing the string...
248
249    } else {
250      // get the delimited value... add it..
251      String value = str.substring(startPos, endPos);
252      if (keepEmpties || !value.isEmpty()) {
253        list.add(value);
254      }
255      // recursively search as we are not at the end yet...
256      delimiter(str, delimiter, keepEmpties, endPos + 1, list);
257    }
258  }

260  /**
261   * This returns the FIRST string in str that is bounded on the left by
262   * leftBound, and bounded on the right by rightBound. This will return null if
263   * the leftBound is not found within str.
264   * <p>
265   * If leftBound can't be found this returns null.
266   * <p>
267   * This rightBound can't be found then this throws a
268   * StringIndexOutOfBoundsException.
269   *
270   * @param str        the base string that we will search for the bounded string.
271   * @param leftBound  the left bound of the string.
272   * @param rightBound the right bound of the string.
273   */
274  public static String getBoundedString(String str, String leftBound, String rightBound)
275    throws RuntimeException {
276
277    if (str == null) {
278      throw new RuntimeException("string to parse is null?");
279    }
280    int startPos = str.indexOf(leftBound);
281    if (startPos > -1) {
282      startPos = startPos + leftBound.length();
283      int endPos = str.indexOf(rightBound, startPos);
284      // dp(str+" start:"+startPos+" end:"+endPos);
285      if (endPos == -1) {
286        throw new RuntimeException("Can't find rightBound: " + rightBound);
287      }
288      return str.substring(startPos, endPos);
289    } else {
290      // if no leftBound can be found.. return null... could be in a
291      // search n parse type loop?
292      // this keeps "no tag"==null different from "tag not formed
293      // properly"==StringParsingException
294      return null;
295    }
296  }

298  /**
299   * Takes the String bounded by leftBound & rightBound, and replaces it with
300   * replaceString. Actually removes the left and right bound strings aswell.
301   */
302  public static String setBoundedString(String str, String leftBound, String rightBound,
303                                        String replaceString) {
304
305    int startPos = str.indexOf(leftBound);
306    if (startPos > -1) {
307      // startPos = startPos;
308      int endPos = str.indexOf(rightBound, startPos + leftBound.length());
309      if (endPos > -1) {
310        String toReplace = str.substring(startPos, endPos + 1);
311        return replaceString(str, toReplace, replaceString);
312      } else {
313        return str;
314      }
315    } else {
316      return str;
317    }
318  }

320  /**
321   * This method takes a String and will replace all occurrences of the match
322   * String with that of the replace String.
323   *
324   * @param source  the source string
325   * @param match   the string used to find a match
326   * @param replace the string used to replace match with
327   * @return the source string after the search and replace
328   */
329  public static String replaceString(String source, String match, String replace) {
330    if (source == null) {
331      return null;
332    }
333    if (replace == null) {
334      return source;
335    }
336    if (match == null) {
337      throw new NullPointerException("match is null?");
338    }
339    if (match.equals(replace)) {
340      return source;
341    }
342    return replaceString(source, match, replace, 30, 0, source.length());
343  }

345  /**
346   * Additionally specify the additionalSize to add to the buffer. This will
347   * make the buffer bigger so that it doesn't have to grow when replacement
348   * occurs.
349   */
350  public static String replaceString(String source, String match, String replace,
351                                     int additionalSize, int startPos, int endPos) {
352
353    if (source == null) {
354      return null;
355    }
356
357    char match0 = match.charAt(0);
358
359    int matchLength = match.length();
360
361    if (matchLength == 1 && replace.length() == 1) {
362      char replace0 = replace.charAt(0);
363      return source.replace(match0, replace0);
364    }
365    if (matchLength >= replace.length()) {
366      additionalSize = 0;
367    }
368
369    int sourceLength = source.length();
370    int lastMatch = endPos - matchLength;
371
372    StringBuilder sb = new StringBuilder(sourceLength + additionalSize);
373
374    if (startPos > 0) {
375      sb.append(source.substring(0, startPos));
376    }
377
378    char sourceChar;
379    boolean isMatch;
380    int sourceMatchPos;
381
382    for (int i = startPos; i < sourceLength; i++) {
383      sourceChar = source.charAt(i);
384      if (i > lastMatch || sourceChar != match0) {
385        sb.append(sourceChar);
386
387      } else {
388        // check to see if this is a match
389        isMatch = true;
390        sourceMatchPos = i;
391
392        // check each following character...
393        for (int j = 1; j < matchLength; j++) {
394          sourceMatchPos++;
395          if (source.charAt(sourceMatchPos) != match.charAt(j)) {
396            isMatch = false;
397            break;
398          }
399        }
400        if (isMatch) {
401          i = i + matchLength - 1;
402          sb.append(replace);
403        } else {
404          // was not a match
405          sb.append(sourceChar);
406        }
407      }
408    }
409
410    return sb.toString();
411  }

428  /**
429   * Additionally specify an additional size estimate for the buffer plus start
430   * and end positions.
431   * <p>
432   * The start and end positions can limit the search and replace. Otherwise
433   * these default to startPos = 0 and endPos = source.length().
434   * </p>
435   */
436  public static String replaceStringMulti(String source, String[] match, String replace,
437                                          int additionalSize, int startPos, int endPos) {
438
439    int shortestMatch = match[0].length();
440
441    char[] match0 = new char[match.length];
442    for (int i = 0; i < match0.length; i++) {
443      match0[i] = match[i].charAt(0);
444      if (match[i].length() < shortestMatch) {
445        shortestMatch = match[i].length();
446      }
447    }
448
449    StringBuilder sb = new StringBuilder(source.length() + additionalSize);
450
451    char sourceChar;
452
453    int len = source.length();
454    int lastMatch = endPos - shortestMatch;
455
456    if (startPos > 0) {
457      sb.append(source.substring(0, startPos));
458    }
459
460    int matchCount;
461
462    for (int i = startPos; i < len; i++) {
463      sourceChar = source.charAt(i);
464      if (i > lastMatch) {
465        sb.append(sourceChar);
466      } else {
467        matchCount = 0;
468        for (int k = 0; k < match0.length; k++) {
469          if (matchCount == 0 && sourceChar == match0[k]) {
470            if (match[k].length() + i <= len) {
471
472              ++matchCount;
473              int j = 1;
474              for (; j < match[k].length(); j++) {
475                if (source.charAt(i + j) != match[k].charAt(j)) {
476                  --matchCount;
477                  break;
478                }
479              }
480              if (matchCount > 0) {
481                i = i + j - 1;
482                sb.append(replace);
483                break;
484              }
485            }
486          }
487        }
488        if (matchCount == 0) {
489          sb.append(sourceChar);
490        }
491      }
492    }
493
494    return sb.toString();
495  }

497  /**
498   * This method takes a String as an argument and removes all occurrences of
499   * the supplied Char. It returns the resulting String.
500   */
501  public static String removeChar(String s, char chr) {
502
503    StringBuilder sb = new StringBuilder(s.length());
504
505    for (int i = 0; i < s.length(); i++) {
506      char c = s.charAt(i);
507      if (c != chr) {
508        sb.append(c);
509      }
510    }
511
512    return sb.toString();
513  }

515  /**
516   * This method takes a String as an argument and removes all occurrences of
517   * the supplied Chars. It returns the resulting String.
518   */
519  public static String removeChars(String s, char[] chr) {
520
521    StringBuilder sb = new StringBuilder(s.length());
522
523    for (int i = 0; i < s.length(); i++) {
524      char c = s.charAt(i);
525      if (!charMatch(c, chr)) {
526        sb.append(c);
527      }
528    }
529
530    return sb.toString();
531  }

533  private static boolean charMatch(int iChr, char[] chr) {
534    for (char aChr : chr) {
535      if (iChr == aChr) {
536        return true;
537      }
538    }
539    return false;
