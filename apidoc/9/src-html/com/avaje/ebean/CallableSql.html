001package com.avaje.ebean;
002
003import java.sql.CallableStatement;
004import java.sql.SQLException;
005
006/**
007 * For making calls to stored procedures. Refer to the Ebean execute() method.
008 * <p>
009 * Note that UpdateSql is designed for general DML sql and CallableSql is
010 * designed for use with stored procedures. Also note that when using this in
011 * batch mode the out parameters are not read.
012 * </p>
013 * <p>
014 * Example 1:
015 * </p>
016 * <pre>{@code
017 *
018 * String sql = "{call sp_order_mod(?,?)}";
019 *
020 * CallableSql cs = Ebean.createCallableSql(sql);
021 * cs.setParameter(1, "turbo");
022 * cs.registerOut(2, Types.INTEGER);
023 *
024 * Ebean.execute(cs);
025 *
026 * // read the out parameter
027 * Integer returnValue = (Integer) cs.getObject(2);
028 *
029 * }</pre>
030 * <p>
031 * Example 2:<br>
032 * Includes batch mode, table modification information and label. Note that the
033 * label is really only to help people reading the transaction logs to identify
034 * the procedure called etc.
035 * </p>
036 *
037 * <pre>{@code
038 *
039 * String sql = "{call sp_insert_order(?,?)}";
040 *
041 * CallableSql cs = Ebean.createCallableSql(sql);
042 *
043 * // Inform Ebean this stored procedure inserts into the
044 * // oe_order table and inserts + updates the oe_order_detail table.
045 * // this is used to invalidate objects in the cache
046 * cs.addModification("oe_order", true, false, false);
047 * cs.addModification("oe_order_detail", true, true, false);
048 *
049 * Transaction t = Ebean.startTransaction();
050 *
051 * // execute using JDBC batching 10 statements at a time
052 * t.setBatchMode(true);
053 * t.setBatchSize(10);
054 * try {
055 *   cs.setParameter(1, "Was");
056 *   cs.setParameter(2, "Banana");
057 *   Ebean.execute(cs);
058 *
059 *   cs.setParameter(1, "Here");
060 *   cs.setParameter(2, "Kumera");
061 *   Ebean.execute(cs);
062 *
063 *   cs.setParameter(1, "More");
064 *   cs.setParameter(2, "Apple");
065 *   Ebean.execute(cs);
066 *
067 *   // Ebean.externalModification("oe_order",true,false,false);
068 *   // Ebean.externalModification("oe_order_detail",true,true,false);
069 *   Ebean.commitTransaction();
070 *
071 * } finally {
072 *   Ebean.endTransaction();
073 * }
074 * }</pre>
075 *
076 * @see com.avaje.ebean.SqlUpdate
077 * @see com.avaje.ebean.Ebean#execute(CallableSql)
078 */
079public interface CallableSql {
080
081  /**
082   * Return the label that is put into the transaction log.
083   */
084  String getLabel();
085
086  /**
087   * Set the label that is put in the transaction log.
088   */
089  CallableSql setLabel(String label);
090
091  /**
092   * Return the statement execution timeout.
093   */
094  int getTimeout();
095
096  /**
097   * Return the callable sql.
098   */
099  String getSql();
100
101  /**
102   * Set the statement execution timeout. Zero implies unlimited time.
103   * <p>
104   * This is set to the underlying CallableStatement.
105   * </p>
106   */
107  CallableSql setTimeout(int secs);
108
109  /**
110   * Set the callable sql.
111   */
112  CallableSql setSql(String sql);
113
114  /**
115   * Bind a parameter that is bound as a IN parameter.
116   * <p>
117   * position starts at value 1 (not 0) to be consistent with CallableStatement.
118   * </p>
119   * <p>
120   * This is designed so that you do not need to set params in index order. You
121   * can set/register param 2 before param 1 etc.
122   * </p>
123   *
124   * @param position the index position of the parameter.
125   * @param value    the value of the parameter.
126   */
127  CallableSql bind(int position, Object value);
128
129  /**
130   * Bind a positioned parameter (same as bind method).
131   *
132   * @param position the index position of the parameter.
133   * @param value    the value of the parameter.
134   */
135  CallableSql setParameter(int position, Object value);
136
137  /**
138   * Register an OUT parameter.
139   * <p>
140   * Note that position starts at value 1 (not 0) to be consistent with
141   * CallableStatement.
142   * </p>
143   * <p>
144   * This is designed so that you do not need to register params in index order.
145   * You can set/register param 2 before param 1 etc.
146   * </p>
147   *
148   * @param position the index position of the parameter (starts with 1).
149   * @param type     the jdbc type of the OUT parameter that will be read.
150   */
151  CallableSql registerOut(int position, int type);
152
153  /**
154   * Return an OUT parameter value.
155   * <p>
156   * position starts at value 1 (not 0) to be consistent with CallableStatement.
157   * </p>
158   * <p>
159   * This can only be called after the CallableSql has been executed. When run
160   * in batch mode you effectively can't use this method.
161   * </p>
162   */
163  Object getObject(int position);
164
165  /**
166   * You can extend this object and override this method for more advanced
167   * stored procedure calls. This would be the case when ResultSets are returned
168   * etc.
169   */
170  boolean executeOverride(CallableStatement cstmt) throws SQLException;
171
172  /**
173   * Add table modification information to the TransactionEvent.
174   * <p>
175   * This would be similar to using the
176   * <code>Ebean.externalModification()</code> method. It may be easier and make
177   * more sense to set it here with the CallableSql.
178   * </p>
179   * <p>
180   * For UpdateSql the table modification information is derived by parsing the
181   * sql to determine the table name and whether it was an insert, update or
182   * delete.
183   * </p>
184   */
185  CallableSql addModification(String tableName, boolean inserts, boolean updates, boolean deletes);
186
187}
