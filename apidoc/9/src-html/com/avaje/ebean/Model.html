001package com.avaje.ebean;
002
003import com.avaje.ebean.bean.EntityBean;
004
005import javax.persistence.MappedSuperclass;
006
007/**
008 * A MappedSuperclass base class that provides convenience methods for inserting, updating and
009 * deleting beans.
010 * <p>
011 * By having your entity beans extend this it provides a 'Active Record' style programming model for
012 * Ebean users.
013 * <p>
014 * Note that there is a ebean-mocker project that enables you to use Mockito or similar
015 * tools to still mock out the underlying 'default EbeanServer' for testing purposes.
016 * <p>
017 * You may choose not use this Model mapped superclass if you don't like the 'Active Record' style
018 * or if you believe it 'pollutes' your entity beans.
019 * <p>
020 * You can use Dependency Injection like Guice or Spring to construct and wire a EbeanServer instance
021 * and have that same instance used with this Model and Finder. The way that works is that when the
022 * DI container creates the EbeanServer instance it can be registered with the Ebean singleton. In this
023 * way the EbeanServer instance can be injected as per normal Guice / Spring dependency injection and
024 * that same instance also used to support the Model and Finder active record style.
025 * <p>
026 * If you choose to use the Model mapped superclass you will probably also chose to additionally add
027 * a {@link Finder} as a public static field to complete the active record pattern and provide a
028 * relatively nice clean way to write queries.
029 * <p>
030 * <h3>Typical common @MappedSuperclass</h3>
031 * <pre>{@code
032 *
033 *     // Typically there is a common base model that has some
034 *     // common properties like the ones below
035 *
036 *     @MappedSuperclass
037 *     public class BaseModel extends Model {
038 *
039 *       @Id Long id;
040 *
041 *       @Version Long version;
042 *
043 *       @WhenCreated Timestamp whenCreated;
044 *
045 *       @WhenUpdated Timestamp whenUpdated;
046 *
047 *       ...
048 *
049 * }</pre>
050 * <p>
051 * <h3>Extend the Model</h3>
052 * <pre>{@code
053 *
054 *     // Extend the mappedSuperclass
055 *
056 *     @Entity @Table(name="o_account")
057 *     public class Customer extends BaseModel {
058 *
059 *       String name;
060 *       ...
061 *     }
062 *
063 * }</pre>
064 * <p>
065 * <h3>Modal: save()</h3>
066 * <pre>{@code
067 *
068 *     // Active record style ... save(), delete() etc
069 *     Customer customer = new Customer();
070 *     customer.setName("AC234");
071 *
072 *     // save() method inherited from Model
073 *     customer.save();
074 *
075 * }</pre>
076 */
077@MappedSuperclass
078public abstract class Model {
079
080  /**
081   * Return the underlying 'default' EbeanServer.
082   * <p>
083   * This provides full access to the API such as explicit transaction demarcation etc.
084   * <p>
085   * Example:
086   * <pre>{@code
087   *
088   * Transaction transaction = Customer.db().beginTransaction();
089   * try {
090   *
091   *   // turn off cascade persist for this transaction
092   *   transaction.setPersistCascade(false);
093   *
094   *   // extra control over jdbc batching for this transaction
095   *   transaction.setBatchGetGeneratedKeys(false);
096   *   transaction.setBatchMode(true);
097   *   transaction.setBatchSize(20);
098   *
099   *   Customer customer = new Customer();
100   *   customer.setName(&quot;Roberto&quot;);
101   *   customer.save();
102   *
103   *   Customer otherCustomer = new Customer();
104   *   otherCustomer.setName("Franko");
105   *   otherCustomer.save();
106   *
107   *   transaction.commit();
108   *
109   * } finally {
110   *   transaction.end();
111   * }
112   *
113   * }</pre>
114   */
115  public static EbeanServer db() {
116    return Ebean.getDefaultServer();
117  }
118
119  /**
120   * Return a named EbeanServer that is typically different to the default server.
121   * <p>
122   * If you are using multiple databases then each database has a name and maps to a single
123   * EbeanServer. You can use this method to get an EbeanServer for another database.
124   *
125   * @param server The name of the EbeanServer. If this is null then the default EbeanServer is returned.
126   */
127  public static EbeanServer db(String server) {
128    return Ebean.getServer(server);
129  }
130
131  /**
132   * Marks the entity bean as dirty.
133   * <p>
134   * This is used so that when a bean that is otherwise unmodified is updated the version
135   * property is updated.
136   * <p>
137   * An unmodified bean that is saved or updated is normally skipped and this marks the bean as
138   * dirty so that it is not skipped.
139   * <p>
140   * <pre>{@code
141   *
142   * Customer customer = Customer.find.byId(id);
143   *
144   * // mark the bean as dirty so that a save() or update() will
145   * // increment the version property
146   * customer.markAsDirty();
147   * customer.save();
148   *
149   * }</pre>
150   *
151   * @see EbeanServer#markAsDirty(Object)
152   */
153  public void markAsDirty() {
154    db().markAsDirty(this);
155  }
156
157  /**
158   * Mark the property as unset or 'not loaded'.
159   * <p>
160   * This would be used to specify a property that we did not wish to include in a stateless update.
161   * </p>
162   * <pre>{@code
163   *
164   *   // populate an entity bean from JSON or whatever
165   *   User user = ...;
166   *
167   *   // mark the email property as 'unset' so that it is not
168   *   // included in a 'stateless update'
169   *   user.markPropertyUnset("email");
170   *
171   *   user.update();
172   *
173   * }</pre>
174   *
175   * @param propertyName the name of the property on the bean to be marked as 'unset'
176   */
177  public void markPropertyUnset(String propertyName) {
178    ((EntityBean) this)._ebean_getIntercept().setPropertyLoaded(propertyName, false);
179  }
180
181  /**
182   * Insert or update this entity depending on its state.
183   * <p>
184   * Ebean will detect if this is a new bean or a previously fetched bean and perform either an
185   * insert or an update based on that.
186   *
187   * @see EbeanServer#save(Object)
188   */
189  public void save() {
190    db().save(this);
191  }
192
193  /**
194   * Update this entity.
195   *
196   * @see EbeanServer#update(Object)
197   */
198  public void update() {
199    db().update(this);
200  }
201
202  /**
203   * Insert this entity.
204   *
205   * @see EbeanServer#insert(Object)
206   */
207  public void insert() {
208    db().insert(this);
209  }
210
211  /**
212   * Delete this bean.
213   * <p>
214   * This will return true if the bean was deleted successfully or JDBC batch is being used.
215   * </p>
216   * <p>
217   * If there is no current transaction one will be created and committed for
218   * you automatically.
219   * </p>
220   * <p>
221   * If the Bean does not have a version property (or loaded version property) and
222   * the bean does not exist then this returns false indicating that nothing was
223   * deleted. Note that, if JDBC batch mode is used then this always returns true.
224   * </p>
225   *
226   * @see EbeanServer#delete(Object)
227   */
228  public boolean delete() {
229    return db().delete(this);
230  }
231
232  /**
233   * Delete a bean permanently without soft delete.
234   * <p>
235   * This is used when the bean contains a <code>@SoftDelete</code> property and we
236   * want to perform a hard/permanent delete.
237   * </p>
238   *
239   * @see EbeanServer#deletePermanent(Object)
240   */
241  public boolean deletePermanent() {
242    return db().deletePermanent(this);
243  }
244
245  /**
246   * Perform an update using this entity against the specified server.
247   */
248  public void update(String server) {
249    db(server).update(this);
250  }
251
252  /**
253   * Perform an insert using this entity against the specified server.
254   */
255  public void insert(String server) {
256    db(server).insert(this);
257  }
258
259  /**
260   * Perform a delete using this entity against the specified server.
261   */
262  public boolean delete(String server) {
263    return db(server).delete(this);
264  }
265
266  /**
267   * Refreshes this entity from the database.
268   *
269   * @see EbeanServer#refresh(Object)
270   */
271  public void refresh() {
272    db().refresh(this);
273  }
274
275}
