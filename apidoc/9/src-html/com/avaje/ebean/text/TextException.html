001package com.avaje.ebean.text;
002
003/**
004 * An exception occurred typically in processing CSV, JSON or XML.
005 */
006public class TextException extends RuntimeException {
007
008  private static final long serialVersionUID = 1601310159486033148L;
009
010  /**
011   * Construct with an error message.
012   */
013  public TextException(String msg) {
014    super(msg);
015  }
016
017  /**
018   * Construct with a message and cause.
019   */
020  public TextException(String msg, Exception e) {
021    super(msg, e);
022  }
023
024  /**
025   * Construct with a cause.
026   */
027  public TextException(Exception e) {
028    super(e);
029  }
030}
