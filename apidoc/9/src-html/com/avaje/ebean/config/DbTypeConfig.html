001package com.avaje.ebean.config;
002
003import com.avaje.ebean.config.dbplatform.DbType;
004import com.avaje.ebean.config.dbplatform.IdType;
005
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * Configuration for DB types such as UUID, Geometry etc.
011 */
012public class DbTypeConfig {
013
014  /**
015   * The Geometry SRID value (default 4326).
016   */
017  private int geometrySRID = 4326;
018
019  /**
020   * Set for DB's that support both Sequence and Identity (and the default choice is not desired).
021   */
022  private IdType idType;
023
024  /**
025   * Setting to indicate if UUID should be stored as binary(16) or varchar(40) or native DB type (for H2 and Postgres).
026   */
027  private ServerConfig.DbUuid dbUuid = ServerConfig.DbUuid.AUTO_VARCHAR;
028
029  /**
030   * Modify the default mapping of standard types such as default precision for DECIMAL etc.
031   */
032  private List<CustomDbTypeMapping> customDbTypeMappings = new ArrayList<>();
033
034  /**
035   * Return the Geometry SRID.
036   */
037  public int getGeometrySRID() {
038    return geometrySRID;
039  }
040
041  /**
042   * Set the Geometry SRID.
043   */
044  public void setGeometrySRID(int geometrySRID) {
045    this.geometrySRID = geometrySRID;
046  }
047
048  /**
049   * Return the DB type used to store UUID.
050   */
051  public ServerConfig.DbUuid getDbUuid() {
052    return dbUuid;
053  }
054
055  /**
056   * Set the DB type used to store UUID.
057   */
058  public void setDbUuid(ServerConfig.DbUuid dbUuid) {
059    this.dbUuid = dbUuid;
060  }
061
062  /**
063   * Return the IdType to use (or null for the default choice).
064   */
065  public IdType getIdType() {
066    return idType;
067  }
068
069  /**
070   * Set the IdType to use (when the DB supports both SEQUENCE and IDENTITY and the default is not desired).
071   */
072  public void setIdType(IdType idType) {
073    this.idType = idType;
074  }
075
076  /**
077   * Add a custom type mapping.
078   * <p>
079   * <pre>{@code
080   *
081   *   // set the default mapping for BigDecimal.class/decimal
082   *   serverConfig.addCustomMapping(DbType.DECIMAL, "decimal(18,6)");
083   *
084   *   // set the default mapping for String.class/varchar but only for Postgres
085   *   serverConfig.addCustomMapping(DbType.VARCHAR, "text", Platform.POSTGRES);
086   *
087   * }</pre>
088   *
089   * @param type             The DB type this mapping should apply to
090   * @param columnDefinition The column definition that should be used
091   * @param platform         Optionally specify the platform this mapping should apply to.
092   */
093  public void addCustomMapping(DbType type, String columnDefinition, Platform platform) {
094    customDbTypeMappings.add(new CustomDbTypeMapping(type, columnDefinition, platform));
095  }
096
097  /**
098   * Add a custom type mapping that applies to all platforms.
099   * <p>
100   * <pre>{@code
101   *
102   *   // set the default mapping for BigDecimal/decimal
103   *   serverConfig.addCustomMapping(DbType.DECIMAL, "decimal(18,6)");
104   *
105   *   // set the default mapping for String/varchar
106   *   serverConfig.addCustomMapping(DbType.VARCHAR, "text");
107   *
108   * }</pre>
109   *
110   * @param type             The DB type this mapping should apply to
111   * @param columnDefinition The column definition that should be used
112   */
113  public void addCustomMapping(DbType type, String columnDefinition) {
114    customDbTypeMappings.add(new CustomDbTypeMapping(type, columnDefinition));
115  }
116
117  /**
118   * Return the list of custom type mappings.
119   */
120  public List<CustomDbTypeMapping> getCustomTypeMappings() {
121    return customDbTypeMappings;
122  }
123}
