001package com.avaje.ebean.config.dbplatform.hsqldb;
002
003import com.avaje.ebean.BackgroundExecutor;
004import com.avaje.ebean.config.Platform;
005import com.avaje.ebean.config.dbplatform.DatabasePlatform;
006import com.avaje.ebean.config.dbplatform.DbPlatformType;
007import com.avaje.ebean.config.dbplatform.DbType;
008import com.avaje.ebean.config.dbplatform.IdType;
009import com.avaje.ebean.config.dbplatform.PlatformIdGenerator;
010import com.avaje.ebean.config.dbplatform.h2.H2DbEncrypt;
011import com.avaje.ebean.config.dbplatform.h2.H2SequenceIdGenerator;
012import com.avaje.ebean.dbmigration.ddlgeneration.platform.HsqldbDdl;
013
014import javax.sql.DataSource;
015
016/**
017 * H2 specific platform.
018 */
019public class HsqldbPlatform extends DatabasePlatform {
020
021  public HsqldbPlatform() {
022    super();
023    this.platform = Platform.HSQLDB;
024    this.dbEncrypt = new H2DbEncrypt();
025    this.platformDdl = new HsqldbDdl(this);
026
027    this.dbIdentity.setIdType(IdType.IDENTITY);
028    this.dbIdentity.setSupportsGetGeneratedKeys(true);
029    this.dbIdentity.setSupportsSequence(true);
030    this.dbIdentity.setSupportsIdentity(true);
031
032    dbTypeMap.put(DbType.INTEGER, new DbPlatformType("integer", false));
033  }
034
035  @Override
036  public PlatformIdGenerator createSequenceIdGenerator(BackgroundExecutor be, DataSource ds, String seqName, int batchSize) {
037
038    return new H2SequenceIdGenerator(be, ds, seqName, batchSize);
039  }
040
041}
