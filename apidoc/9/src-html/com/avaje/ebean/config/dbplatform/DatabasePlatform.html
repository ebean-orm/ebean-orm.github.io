181  /**
182   * Configure UUID Storage etc based on ServerConfig settings.
183   */
184  public void configure(DbTypeConfig config) {
185    addGeoTypes(config.getGeometrySRID());
186    configureIdType(config.getIdType());
187    dbTypeMap.config(nativeUuidType, config.getDbUuid());
188    for (CustomDbTypeMapping mapping : config.getCustomTypeMappings()) {
189      if (platformMatch(mapping.getPlatform())) {
190        dbTypeMap.put(mapping.getType(), parse(mapping.getColumnDefinition()));
191      }
192    }
193  }
194
195  protected void configureIdType(IdType idType) {
196    if (idType != null) {
197      this.dbIdentity.setIdType(idType);
198    }
199  }

519  /**
520   * Convert backticks to the platform specific open quote and close quote
521   * <p>
522   * Specific plugins may implement this method to cater for platform specific
523   * naming rules.
524   * </p>
525   *
526   * @param dbName the db name
527   * @return the string
528   */
529  public String convertQuotedIdentifiers(String dbName) {
530    // Ignore null values e.g. schema name or catalog
531    if (dbName != null && !dbName.isEmpty()) {
532      if (dbName.charAt(0) == BACK_TICK) {
533        if (dbName.charAt(dbName.length() - 1) == BACK_TICK) {
534
535          String quotedName = getOpenQuote();
536          quotedName += dbName.substring(1, dbName.length() - 1);
537          quotedName += getCloseQuote();
538
539          return quotedName;
540
541        } else {
542          logger.error("Missing backquote on [" + dbName + "]");
543        }
544      }
545    }
546    return dbName;
547  }

556  public String completeSql(String sql, Query<?> query) {
557    if (Boolean.TRUE.equals(query.isForUpdate())) {
558      sql = withForUpdate(sql);
559    }
560
561    return sql;
562  }
563
564  protected String withForUpdate(String sql) {
565    // silently assume the database does not support the "for update" clause.
566    logger.info("it seems your database does not support the 'for update' clause");
567
568    return sql;
569  }

587  /**
588   * Return true if the table exists.
589   */
590  public boolean tableExists(Connection connection, String catalog, String schema, String table) throws SQLException {
591
592    DatabaseMetaData metaData = connection.getMetaData();
593    ResultSet tables = metaData.getTables(catalog, schema, table, null);
594    try {
595      return tables.next();
596    } finally {
597      close(tables);
598    }
599  }

603   */
604  protected void close(ResultSet resultSet) {
605    try {
606      resultSet.close();
607    } catch (SQLException e) {
608      logger.error("Error closing resultSet", e);
609    }
