001package com.avaje.ebean.config.dbplatform;
002
003import com.avaje.ebean.Transaction;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import javax.persistence.PersistenceException;
008import javax.sql.DataSource;
009import java.sql.Connection;
010import java.sql.PreparedStatement;
011import java.sql.ResultSet;
012import java.sql.SQLException;
013
014/**
015 * A very simple Database sequence based IdGenerator.
016 * <p>
017 * One which batch requests sequence Id's would be better for performance.
018 * </p>
019 */
020public class SimpleSequenceIdGenerator implements PlatformIdGenerator {
021
022  private static final Logger logger = LoggerFactory.getLogger(SimpleSequenceIdGenerator.class);
023
024  private final String sql;
025
026  private final DataSource dataSource;
027
028  private final String seqName;
029
030  /**
031   * Construct given a dataSource and sql to return the next sequence value.
032   */
033  public SimpleSequenceIdGenerator(DataSource dataSource, String sql, String seqName) {
034    this.dataSource = dataSource;
035    this.sql = sql;
036    this.seqName = seqName;
037  }
038
039  public String getName() {
040    return seqName;
041  }
042
043  public boolean isDbSequence() {
044    return true;
045  }
046
047  public void preAllocateIds(int batchSize) {
048    // just ignore this
049  }
050
051  public Object nextId(Transaction t) {
052
053    boolean useTxnConnection = t != null;
054
055    Connection c = null;
056    PreparedStatement pstmt = null;
057    ResultSet rset = null;
058    try {
059      c = useTxnConnection ? t.getConnection() : dataSource.getConnection();
060      pstmt = c.prepareStatement(sql);
061      rset = pstmt.executeQuery();
062      if (rset.next()) {
063        return rset.getInt(1);
064      } else {
065        String m = "Always expecting 1 row from " + sql;
066        throw new PersistenceException(m);
067      }
068    } catch (SQLException e) {
069      throw new PersistenceException("Error getting sequence nextval", e);
070
071    } finally {
072      if (useTxnConnection) {
073        closeResources(rset, pstmt, null);
074      } else {
075        closeResources(rset, pstmt, c);
076      }
077    }
078  }
079
080  private void closeResources(ResultSet rset, PreparedStatement pstmt, Connection c) {
081    try {
082      if (rset != null) {
083        rset.close();
084      }
085    } catch (SQLException e) {
086      logger.error("Error closing ResultSet", e);
087    }
088    try {
089      if (pstmt != null) {
090        pstmt.close();
091      }
092    } catch (SQLException e) {
093      logger.error("Error closing PreparedStatement", e);
094    }
095    try {
096      if (c != null) {
097        c.close();
098      }
099    } catch (SQLException e) {
100      logger.error("Error closing Connection", e);
101    }
102  }
103
104}
